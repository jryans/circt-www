80 IntegerAttr direction::packAttribute(MLIRContext *context,
81  ArrayRef<Direction> directions) {
82  // Pack the array of directions into an APInt. Input is zero, output is one.
83  auto size = directions.size();
84  APInt portDirections(size, 0);
85  for (size_t i = 0; i != size; ++i)
86  if (directions[i] == Direction::Out)
87  portDirections.setBit(i);
88  return IntegerAttr::get(IntegerType::get(context, size), portDirections);
89 }
