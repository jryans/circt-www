130 /// Check if an `OMNode` is an `OMSRAM` and requires special treatment of its
131 /// instance path field. This returns the ID of the tracker stored in the
132 /// `instancePath` or `finalPath` field if the node has an array field `omType`
133 /// that contains a `OMString:OMSRAM` entry.
134 static IntegerAttr isOMSRAM(Attribute &node) {
135  auto dict = node.dyn_cast<DictionaryAttr>();
136  if (!dict)
137  return {};
138  auto idAttr = dict.getAs<StringAttr>("id");
139  if (!idAttr)
140  return {};
141  IntegerAttr id;
142  if (auto infoAttr = dict.getAs<DictionaryAttr>("fields")) {
143  auto finalPath = infoAttr.getAs<DictionaryAttr>("finalPath");
144  // The following is used prior to an upstream bump in Chisel.
145  if (!finalPath)
146  finalPath = infoAttr.getAs<DictionaryAttr>("instancePath");
147  if (finalPath)
148  if (auto v = finalPath.getAs<DictionaryAttr>("value"))
149  if (v.getAs<UnitAttr>("omir.tracker"))
150  id = v.getAs<IntegerAttr>("id");
151  if (auto omTy = infoAttr.getAs<DictionaryAttr>("omType"))
152  if (auto valueArr = omTy.getAs<ArrayAttr>("value"))
153  for (auto attr : valueArr)
154  if (auto str = attr.dyn_cast<StringAttr>())
155  if (str.getValue().equals("OMString:OMSRAM"))
156  return id;
157  }
158  return {};
159 }

165 void EmitOMIRPass::runOnOperation() {
166  MLIRContext *context = &getContext();
167  anyFailures = false;
168  symtbl = nullptr;
169  circuitNamespace = nullptr;
170  instancePaths = nullptr;
171  trackers.clear();
172  symbols.clear();
173  symbolIndices.clear();
174  removeTempNLAs.clear();
175  moduleNamespaces.clear();
176  CircuitOp circuitOp = getOperation();
177 
178  // Gather the relevant annotations from the circuit. On the one hand these are
179  // all the actual `OMIRAnnotation`s that need processing and emission, as well
180  // as an optional `OMIRFileAnnotation` that overrides the default OMIR output
181  // file. Also while we're at it, keep track of all OMIR nodes that qualify as
182  // an SRAM and that require their trackers to be turned into NLAs starting at
183  // the root of the hierarchy.
184  SmallVector<ArrayRef<Attribute>> annoNodes;
185  DenseSet<Attribute> sramIDs;
186  Optional<StringRef> outputFilename = {};
187 
188  AnnotationSet::removeAnnotations(circuitOp, [&](Annotation anno) {
189  if (anno.isClass(omirFileAnnoClass)) {
190  auto pathAttr = anno.getMember<StringAttr>("filename");
191  if (!pathAttr) {
192  circuitOp.emitError(omirFileAnnoClass)
193  << " annotation missing `filename` string attribute";
194  anyFailures = true;
195  return true;
196  }
197  LLVM_DEBUG(llvm::dbgs() << "- OMIR path: " << pathAttr << "\n");
198  outputFilename = pathAttr.getValue();
199  return true;
200  }
201  if (anno.isClass(omirAnnoClass)) {
202  auto nodesAttr = anno.getMember<ArrayAttr>("nodes");
203  if (!nodesAttr) {
204  circuitOp.emitError(omirAnnoClass)
205  << " annotation missing `nodes` array attribute";
206  anyFailures = true;
207  return true;
208  }
209  LLVM_DEBUG(llvm::dbgs() << "- OMIR: " << nodesAttr << "\n");
210  annoNodes.push_back(nodesAttr.getValue());
211  for (auto node : nodesAttr) {
212  if (auto id = isOMSRAM(node)) {
213  LLVM_DEBUG(llvm::dbgs() << " - is SRAM with tracker " << id << "\n");
214  sramIDs.insert(id);
215  }
216  }
217  return true;
218  }
219  return false;
220  });
221  if (anyFailures)
222  return signalPassFailure();
223 
224  // Establish some of the analyses we need throughout the pass.
225  SymbolTable currentSymtbl(circuitOp);
226  CircuitNamespace currentCircuitNamespace(circuitOp);
227  InstancePathCache currentInstancePaths(getAnalysis<InstanceGraph>());
228  symtbl = &currentSymtbl;
229  circuitNamespace = &currentCircuitNamespace;
230  instancePaths = &currentInstancePaths;
231 
232  // Traverse the IR and collect all tracker annotations that were previously
233  // scattered into the circuit.
234  circuitOp.walk([&](Operation *op) {
235  AnnotationSet::removeAnnotations(op, [&](Annotation anno) {
236  if (!anno.isClass(omirTrackerAnnoClass))
237  return false;
238  Tracker tracker;
239  tracker.op = op;
240  tracker.id = anno.getMember<IntegerAttr>("id");
241  if (!tracker.id) {
242  op->emitError(omirTrackerAnnoClass)
243  << " annotation missing `id` integer attribute";
244  anyFailures = true;
245  return true;
246  }
247  if (auto nlaSym = anno.getMember<FlatSymbolRefAttr>("circt.nonlocal"))
248  tracker.nla =
249  dyn_cast_or_null<NonLocalAnchor>(symtbl->lookup(nlaSym.getAttr()));
250  if (sramIDs.erase(tracker.id))
251  makeTrackerAbsolute(tracker);
252  trackers.insert({tracker.id, tracker});
253  return true;
254  });
255  });
256 
257  // If an OMIR output filename has been specified as a pass parameter, override
258  // whatever the annotations have configured. If neither are specified we just
259  // bail.
260  if (!this->outputFilename.empty())
261  outputFilename = this->outputFilename;
262  if (!outputFilename) {
263  LLVM_DEBUG(llvm::dbgs() << "Not emitting OMIR because no annotation or "
264  "pass parameter specified an output file\n");
265  markAllAnalysesPreserved();
266  return;
267  }
268 
269  // Build the output JSON.
270  std::string jsonBuffer;
271  llvm::raw_string_ostream jsonOs(jsonBuffer);
272  llvm::json::OStream json(jsonOs, 2);
273  json.array([&] {
274  for (auto nodes : annoNodes) {
275  for (auto node : nodes) {
276  emitOMNode(node, json);
277  if (anyFailures)
278  return;
279  }
280  }
281  });
282  if (anyFailures)
283  return signalPassFailure();
284 
285  // Delete the temporary NLAs. This requires us to visit all the nodes along
286  // the NLA's path and remove `circt.nonlocal` annotations referring to the
287  // NLA.
288  for (auto nla : removeTempNLAs) {
289  LLVM_DEBUG(llvm::dbgs() << "Removing " << nla << "\n");
290  for (auto modName : nla.modpath().getAsRange<FlatSymbolRefAttr>()) {
291  Operation *mod = symtbl->lookup(modName.getValue());
292  mod->walk([&](InstanceOp instOp) {
293  AnnotationSet::removeAnnotations(instOp, [&](Annotation anno) {
294  auto match =
295  anno.isClass("circt.nonlocal") &&
296  anno.getMember<FlatSymbolRefAttr>("circt.nonlocal").getAttr() ==
297  nla.sym_nameAttr();
298  if (match)
299  LLVM_DEBUG(llvm::dbgs()
300  << "- Removing " << anno.getDict() << " from " << modName
301  << "." << instOp.name() << "\n");
302  return match;
303  });
304  });
305  }
306  nla->erase();
307  }
308  removeTempNLAs.clear();
309 
310  // Emit the OMIR JSON as a verbatim op.
311  auto builder = OpBuilder(circuitOp);
312  builder.setInsertionPointAfter(circuitOp);
313  auto verbatimOp =
314  builder.create<sv::VerbatimOp>(builder.getUnknownLoc(), jsonBuffer);
315  auto fileAttr = hw::OutputFileAttr::getFromFilename(
316  context, *outputFilename, /*excludeFromFilelist=*/true);
317  verbatimOp->setAttr("output_file", fileAttr);
318  verbatimOp.symbolsAttr(ArrayAttr::get(context, symbols));
319 }

321 /// Make a tracker absolute by adding an NLA to it which starts at the root
322 /// module of the circuit. Generates an error if any module along the path is
323 /// instantiated multiple times.
324 void EmitOMIRPass::makeTrackerAbsolute(Tracker &tracker) {
325  auto *context = &getContext();
326  auto builder = OpBuilder::atBlockBegin(getOperation().getBody());
327 
328  // Pick a name for the NLA that doesn't collide with anything.
329  auto opName = tracker.op->getAttrOfType<StringAttr>("name");
330  auto nlaName = circuitNamespace->newName("omir_nla_" + opName.getValue());
331 
332  // Assemble the NLA annotation to be put on all the operations participating
333  // in the path.
334  auto nlaAttr = builder.getDictionaryAttr({
335  builder.getNamedAttr("circt.nonlocal",
336  FlatSymbolRefAttr::get(context, nlaName)),
337  builder.getNamedAttr("class", StringAttr::get(context, "circt.nonlocal")),
338  });
339 
340  // Get all the paths instantiating this module.
341  auto mod = tracker.op->getParentOfType<FModuleOp>();
342  auto paths = instancePaths->getAbsolutePaths(mod);
343  if (paths.empty()) {
344  tracker.op->emitError("OMIR node targets uninstantiated component `")
345  << opName.getValue() << "`";
346  anyFailures = true;
347  return;
348  }
349  if (paths.size() > 1) {
350  auto diag = tracker.op->emitError("OMIR node targets ambiguous component `")
351  << opName.getValue() << "`";
352  diag.attachNote(tracker.op->getLoc())
353  << "may refer to the following paths:";
354  for (auto path : paths)
355  formatInstancePath(diag.attachNote(tracker.op->getLoc()) << "- ", path);
356  anyFailures = true;
357  return;
358  }
359 
360  // Assemble the module and name path for the NLA. Also attach an NLA reference
361  // annotation to each instance participating in the path.
362  SmallVector<Attribute> modpath, namepath;
363  auto addToPath = [&](Operation *op, StringAttr name) {
364  AnnotationSet annos(op);
365  annos.addAnnotations(nlaAttr);
366  annos.applyToOperation(op);
367  modpath.push_back(FlatSymbolRefAttr::get(op->getParentOfType<FModuleOp>()));
368  namepath.push_back(name);
369  };
370  for (InstanceOp inst : paths[0])
371  addToPath(inst, inst.nameAttr());
372  addToPath(tracker.op, opName);
373 
374  // Add the NLA to the tracker and mark it to be deleted later.
375  tracker.nla = builder.create<NonLocalAnchor>(
376  builder.getUnknownLoc(), builder.getStringAttr(nlaName),
377  builder.getArrayAttr(modpath), builder.getArrayAttr(namepath));
378  removeTempNLAs.push_back(tracker.nla);
379 }

381 /// Emit a source locator into a string, for inclusion in the `info` field of
382 /// `OMNode` and `OMField`.
383 void EmitOMIRPass::emitSourceInfo(Location input, SmallString<64> &into) {
384  into.clear();
385  input->walk([&](Location loc) {
386  if (FileLineColLoc fileLoc = loc.dyn_cast<FileLineColLoc>()) {
387  into.append(into.empty() ? "@[" : " ");
388  (Twine(fileLoc.getFilename()) + " " + Twine(fileLoc.getLine()) + ":" +
389  Twine(fileLoc.getColumn()))
390  .toVector(into);
391  }
392  return WalkResult::advance();
393  });
394  if (!into.empty())
395  into.append("]");
396  else
397  into.append("UnlocatableSourceInfo");
398 }

400 /// Emit an entire `OMNode` as JSON.
401 void EmitOMIRPass::emitOMNode(Attribute node, llvm::json::OStream &jsonStream) {
402  auto dict = node.dyn_cast<DictionaryAttr>();
403  if (!dict) {
404  getOperation()
405  .emitError("OMNode must be a dictionary")
406  .attachNote(getOperation().getLoc())
407  << node;
408  anyFailures = true;
409  return;
410  }
411 
412  // Extract the `info` field and serialize the location.
413  SmallString<64> info;
414  if (auto infoAttr = dict.getAs<LocationAttr>("info"))
415  emitSourceInfo(infoAttr, info);
416  if (anyFailures)
417  return;
418 
419  // Extract the `id` field.
420  auto idAttr = dict.getAs<StringAttr>("id");
421  if (!idAttr) {
422  getOperation()
423  .emitError("OMNode missing `id` string field")
424  .attachNote(getOperation().getLoc())
425  << dict;
426  anyFailures = true;
427  return;
428  }
429 
430  // Extract and order the fields of this node.
431  SmallVector<std::tuple<unsigned, StringAttr, DictionaryAttr>> orderedFields;
432  auto fieldsDict = dict.getAs<DictionaryAttr>("fields");
433  if (fieldsDict) {
434  for (auto nameAndField : fieldsDict.getValue()) {
435  auto fieldDict = nameAndField.getValue().dyn_cast<DictionaryAttr>();
436  if (!fieldDict) {
437  getOperation()
438  .emitError("OMField must be a dictionary")
439  .attachNote(getOperation().getLoc())
440  << nameAndField.getValue();
441  anyFailures = true;
442  return;
443  }
444 
445  unsigned index = 0;
446  if (auto indexAttr = fieldDict.getAs<IntegerAttr>("index"))
447  index = indexAttr.getValue().getLimitedValue();
448 
449  orderedFields.push_back({index, nameAndField.getName(), fieldDict});
450  }
451  llvm::sort(orderedFields,
452  [](auto a, auto b) { return std::get<0>(a) < std::get<0>(b); });
453  }
454 
455  jsonStream.object([&] {
456  jsonStream.attribute("info", info);
457  jsonStream.attribute("id", idAttr.getValue());
458  jsonStream.attributeArray("fields", [&] {
459  for (auto &orderedField : orderedFields) {
460  emitOMField(std::get<1>(orderedField), std::get<2>(orderedField),
461  jsonStream);
462  if (anyFailures)
463  return;
464  }
465  if (auto node = fieldsDict.getAs<DictionaryAttr>("containingModule"))
466  if (auto value = node.getAs<DictionaryAttr>("value"))
467  emitOptionalRTLPorts(value, jsonStream);
468  });
469  });
470 }

475 void EmitOMIRPass::emitOMField(StringAttr fieldName, DictionaryAttr field,

494 // If the given `node` refers to a valid tracker in the IR, gather the
495 // additional port metadata of the module it refers to. Then emit this port
496 // metadata as a `ports` array field for the surrounding `OMNode`.
497 void EmitOMIRPass::emitOptionalRTLPorts(DictionaryAttr node,
498  llvm::json::OStream &jsonStream) {
499  // First make sure we actually have a valid tracker. If not, just silently
500  // abort and don't emit any port metadata.
501  auto idAttr = node.getAs<IntegerAttr>("id");
502  auto trackerIt = trackers.find(idAttr);
503  if (!idAttr || !node.getAs<UnitAttr>("omir.tracker") ||
504  trackerIt == trackers.end())
505  return;
506  auto tracker = trackerIt->second;
507 
508  // Lookup the module the tracker refers to. If it points at something *within*
509  // a module, go dig up the surrounding module. This is roughly what
510  // `Target.referringModule(...)` does on the Scala side.
511  auto module = dyn_cast<FModuleLike>(tracker.op);
512  if (!module)
513  module = tracker.op->getParentOfType<FModuleLike>();
514  if (!module) {
515  LLVM_DEBUG(llvm::dbgs() << "Not emitting RTL ports since tracked operation "
516  "does not have a FModuleLike parent: "
517  << *tracker.op << "\n");
518  return;
519  }
520  LLVM_DEBUG(llvm::dbgs() << "Emitting RTL ports for module `"
521  << module.moduleName() << "`\n");
522 
523  // Emit the JSON.
524  SmallString<64> buf;
525  jsonStream.object([&] {
526  buf.clear();
527  emitSourceInfo(module.getLoc(), buf);
528  jsonStream.attribute("info", buf);
529  jsonStream.attribute("name", "ports");
530  jsonStream.attributeArray("value", [&] {
531  for (auto port : llvm::enumerate(module.getPorts())) {
532  if (port.value().type.getBitWidthOrSentinel() == 0)
533  continue;
534  jsonStream.object([&] {
535  // Emit the `ref` field.
536  buf.assign("OMDontTouchedReferenceTarget:~");
537  buf.append(getOperation().name());
538  buf.push_back('|');
539  buf.append(addSymbol(module));
540  buf.push_back('>');
541  buf.append(addSymbol(getInnerRefTo(module, port.index())));
542  jsonStream.attribute("ref", buf);
543 
544  // Emit the `direction` field.
545  buf.assign("OMString:");
546  buf.append(port.value().isOutput() ? "Output" : "Input");
547  jsonStream.attribute("direction", buf);
548 
549  // Emit the `width` field.
550  buf.assign("OMBigInt:");
551  Twine(port.value().type.getBitWidthOrSentinel()).toVector(buf);
552  jsonStream.attribute("width", buf);
553  });
554  }
555  });
556  });
557 }

559 void EmitOMIRPass::emitValue(Attribute node, llvm::json::OStream &jsonStream) {
560  // Handle the null case.
561  if (!node || node.isa<UnitAttr>())
562  return jsonStream.value(nullptr);
563 
564  // Handle the trivial cases where the OMIR serialization simply uses the
565  // builtin JSON types.
566  if (auto attr = node.dyn_cast<BoolAttr>())
567  return jsonStream.value(attr.getValue()); // OMBoolean
568  if (auto attr = node.dyn_cast<IntegerAttr>()) {
569  // CAVEAT: We expect these integers to come from an OMIR file that is
570  // initially read in from JSON, where they are i32 or i64, so this should
571  // yield a valid value. However, a user could cook up an arbitrary precision
572  // integer attr in MLIR input and then subtly break the JSON spec.
573  SmallString<16> val;
574  attr.getValue().toStringSigned(val);
575  return jsonStream.rawValue(val); // OMInt
576  }
577  if (auto attr = node.dyn_cast<FloatAttr>()) {
578  // CAVEAT: We expect these floats to come from an OMIR file that is
579  // initially read in from JSON, where they are f32 or f64, so this should
580  // yield a valid value. However, a user could cook up an arbitrary precision
581  // float attr in MLIR input and then subtly break the JSON spec.
582  SmallString<16> val;
583  attr.getValue().toString(val);
584  return jsonStream.rawValue(val); // OMDouble
585  }
586 
587  // Handle aggregate types.
588  if (auto attr = node.dyn_cast<ArrayAttr>()) {
589  jsonStream.array([&] {
590  for (auto element : attr.getValue()) {
591  emitValue(element, jsonStream);
592  if (anyFailures)
593  return;
594  }
595  });
596  return;
597  }
598  if (auto attr = node.dyn_cast<DictionaryAttr>()) {
599  // Handle targets that have a corresponding tracker annotation in the IR.
600  if (attr.getAs<UnitAttr>("omir.tracker"))
601  return emitTrackedTarget(attr, jsonStream);
602 
603  // Handle regular dictionaries.
604  jsonStream.object([&] {
605  for (auto field : attr.getValue()) {
606  jsonStream.attributeBegin(field.getName());
607  emitValue(field.getValue(), jsonStream);
608  jsonStream.attributeEnd();
609  if (anyFailures)
610  return;
611  }
612  });
613  return;
614  }
615 
616  // The remaining types are all simple string-encoded pass-through cases.
617  if (auto attr = node.dyn_cast<StringAttr>()) {
618  StringRef val = attr.getValue();
619  if (isOMIRStringEncodedPassthrough(val.split(":").first))
620  return jsonStream.value(val);
621  }
622 
623  // If we get here, we don't know how to serialize the given MLIR attribute as
624  // a OMIR value.
625  jsonStream.value("<unsupported value>");

631 void EmitOMIRPass::emitTrackedTarget(DictionaryAttr node,
632  llvm::json::OStream &jsonStream) {
633  // Extract the `id` field.
634  auto idAttr = node.getAs<IntegerAttr>("id");
635  if (!idAttr) {
636  getOperation()
637  .emitError("tracked OMIR target missing `id` string field")
638  .attachNote(getOperation().getLoc())
639  << node;
640  anyFailures = true;
641  return jsonStream.value("<error>");
642  }
643 
644  // Extract the `type` field.
645  auto typeAttr = node.getAs<StringAttr>("type");
646  if (!typeAttr) {
647  getOperation()
648  .emitError("tracked OMIR target missing `type` string field")
649  .attachNote(getOperation().getLoc())
650  << node;
651  anyFailures = true;
652  return jsonStream.value("<error>");
653  }
654  StringRef type = typeAttr.getValue();
655 
656  // Find the tracker for this target, and handle the case where the tracker has
657  // been deleted.
658  auto trackerIt = trackers.find(idAttr);
659  if (trackerIt == trackers.end()) {
660  // Some of the target types indicate removal of the target through an
661  // `OMDeleted` node.
662  if (type == "OMReferenceTarget" || type == "OMMemberReferenceTarget" ||
663  type == "OMMemberInstanceTarget")
664  return jsonStream.value("OMDeleted:");
665 
666  // The remaining types produce an error upon removal of the target.
667  auto diag = getOperation().emitError("tracked OMIR target of type `")
668  << type << "` was deleted";
669  diag.attachNote(getOperation().getLoc())
670  << "`" << type << "` should never be deleted";
671  if (auto path = node.getAs<StringAttr>("path"))
672  diag.attachNote(getOperation().getLoc())
673  << "original path: `" << path.getValue() << "`";
674  anyFailures = true;
675  return jsonStream.value("<error>");
676  }
677  auto tracker = trackerIt->second;
678 
679  // In case this is an `OMMemberTarget`, handle the case where the component
680  // used to be a "reference target" (wire, register, memory, node) when the
681  // OMIR was read in, but has been change to an "instance target" during the
682  // execution of the compiler. This mainly occurs during mapping of
683  // `firrtl.mem` operations to a corresponding `firrtl.instance`.
684  if (type == "OMMemberReferenceTarget" && isa<InstanceOp, MemOp>(tracker.op))
685  type = "OMMemberInstanceTarget";
686 
687  // Serialize the target circuit first.
688  SmallString<64> target(type);
689  target.append(":~");
690  target.append(getOperation().name());
691  target.push_back('|');
692 
693  // Serialize the local or non-local module/instance hierarchy path.
694  if (tracker.nla) {
695  bool notFirst = false;
696  hw::InnerRefAttr instName;
697  for (auto modAndName : llvm::zip(tracker.nla.modpath().getValue(),
698  tracker.nla.namepath().getValue())) {
699  auto symAttr = std::get<0>(modAndName).cast<FlatSymbolRefAttr>();
700  auto nameAttr = std::get<1>(modAndName).cast<StringAttr>();
701  Operation *module = symtbl->lookup(symAttr.getValue());
702  assert(module);
703  if (notFirst)
704  target.push_back('/');
705  notFirst = true;
706  if (instName) {
707  target.append(addSymbol(instName));
708  target.push_back(':');
709  }
710  target.append(addSymbol(module));
711 
712  // Find an instance with the given name in this module. Ensure it has a
713  // symbol that we can refer to.
714  instName = {};
715  module->walk([&](InstanceOp instOp) {
716  if (instOp.nameAttr() != nameAttr)
717  return;
718  LLVM_DEBUG(llvm::dbgs()
719  << "Marking NLA-participating instance " << nameAttr
720  << " in module " << symAttr << " as dont-touch\n");
721  AnnotationSet::addDontTouch(instOp);
722  instName = getInnerRefTo(instOp);
723  });
724  }
725  } else {
726  FModuleOp module = dyn_cast<FModuleOp>(tracker.op);
727  if (!module)
728  module = tracker.op->getParentOfType<FModuleOp>();
729  assert(module);
730  target.append(addSymbol(module));
731  }
732 
733  // Serialize any potential component *inside* the module that this target may
734  // specifically refer to.
735  hw::InnerRefAttr componentName;
736  if (isa<WireOp, RegOp, RegResetOp, InstanceOp, NodeOp, MemOp>(tracker.op)) {
737  componentName = getInnerRefTo(tracker.op);
738  AnnotationSet::addDontTouch(tracker.op);
739  LLVM_DEBUG(llvm::dbgs() << "Marking OMIR-targeted " << componentName
740  << " as dont-touch\n");
741  } else if (!isa<FModuleOp>(tracker.op)) {
742  tracker.op->emitError("invalid target for `") << type << "` OMIR";
743  anyFailures = true;
744  return jsonStream.value("<error>");
745  }
746  if (componentName) {
747  // Check if the targeted component is going to be emitted as an instance.
748  // This is trivially the case for `InstanceOp`s, but also for `MemOp`s that
749  // get converted to an instance during lowering to HW dialect and generator
750  // callout.
751  [&] {
752  if (type == "OMMemberInstanceTarget") {
753  if (auto instOp = dyn_cast<InstanceOp>(tracker.op)) {
754  target.push_back('/');
755  target.append(addSymbol(componentName));
756  target.push_back(':');
757  target.append(addSymbol(instOp.moduleNameAttr()));
758  return;
759  }
760  if (auto memOp = dyn_cast<MemOp>(tracker.op)) {
761  target.push_back('/');
762  target.append(addSymbol(componentName));
763  target.push_back(':');
764  target.append(memOp.getSummary().getFirMemoryName());
765  return;
766  }
767  }
768  target.push_back('>');
769  target.append(addSymbol(componentName));
770  }();
771  }
772 
773  jsonStream.value(target);
774 }
