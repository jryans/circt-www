43 class PlacementDB {
44 public:
45  PlacementDB(MlirOperation top, PrimitiveDB *seed) {
46  db = circtMSFTCreatePlacementDB(top, seed ? seed->db
47  : CirctMSFTPrimitiveDB{nullptr});
48  }
49  ~PlacementDB() { circtMSFTDeletePlacementDB(db); }
50  size_t addDesignPlacements() {
51  return circtMSFTPlacementDBAddDesignPlacements(db);
52  }
53  bool addPlacement(MlirAttribute loc, MlirAttribute path, std::string subpath,
54  MlirOperation op) {
55  return mlirLogicalResultIsSuccess(circtMSFTPlacementDBAddPlacement(
56  db, loc,
57  CirctMSFTPlacedInstance{path, subpath.c_str(), subpath.size(), op}));
58  }
59  py::object getInstanceAt(MlirAttribute loc) {
60  CirctMSFTPlacedInstance inst;
61  if (!circtMSFTPlacementDBTryGetInstanceAt(db, loc, &inst))
62  return py::none();
63  std::string subpath(inst.subpath, inst.subpathLength);
64  return (py::tuple)py::cast(std::make_tuple(inst.path, subpath, inst.op));
65  }
66  py::object getNearestFreeInColumn(CirctMSFTPrimitiveType prim,
67  uint64_t column, uint64_t nearestToY) {
68  MlirAttribute nearest = circtMSFTPlacementDBGetNearestFreeInColumn(
69  db, prim, column, nearestToY);
70  if (!nearest.ptr)
71  return py::none();
72  return py::cast(nearest);
73  }
74  void walkPlacements(
75  py::function pycb,
76  std::tuple<py::object, py::object, py::object, py::object> bounds,
77  py::object prim) {
78 
79  auto handleNone = [](py::object o) {
80  return o.is_none() ? -1 : o.cast<int64_t>();
81  };
82  int64_t cBounds[4] = {
83  handleNone(std::get<0>(bounds)), handleNone(std::get<1>(bounds)),
84  handleNone(std::get<2>(bounds)), handleNone(std::get<3>(bounds))};
85  CirctMSFTPrimitiveType cPrim;
86  if (prim.is_none())
87  cPrim = -1;
88  else
89  cPrim = prim.cast<CirctMSFTPrimitiveType>();
90  circtMSFTPlacementDBWalkPlacements(
91  db,
92  [](MlirAttribute loc, CirctMSFTPlacedInstance p, void *userData) {
93  std::string subpath(p.subpath, p.subpathLength);
94  py::gil_scoped_acquire gil;
95  py::function pycb = *((py::function *)(userData));
96  if (!p.op.ptr) {
97  pycb(loc, py::none());
98  } else {
99  pycb(loc, std::make_tuple(p.path, subpath, p.op));
100  }
101  },
102  cBounds, cPrim, &pycb);
103  }
104 
105 private:
106  CirctMSFTPlacementDB db;
107 };
