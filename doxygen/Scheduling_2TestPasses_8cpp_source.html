26 static SmallVector<SmallVector<unsigned>> parseArrayOfArrays(ArrayAttr attr) {
27  SmallVector<SmallVector<unsigned>> result;
28  for (auto elemArr : attr.getAsRange<ArrayAttr>()) {
29  result.emplace_back();
30  for (auto elem : elemArr.getAsRange<IntegerAttr>())
31  result.back().push_back(elem.getInt());
32  }
33  return result;
34 }

36 static SmallVector<std::pair<llvm::StringRef, unsigned>>
37 parseArrayOfDicts(ArrayAttr attr, StringRef key) {
38  SmallVector<std::pair<llvm::StringRef, unsigned>> result;
39  for (auto dictAttr : attr.getAsRange<DictionaryAttr>()) {
40  auto name = dictAttr.getAs<StringAttr>("name");
41  auto value = dictAttr.getAs<IntegerAttr>(key);
42  if (name && value)
43  result.push_back(std::make_pair(name.getValue(), value.getInt()));
44  }
45  return result;
46 }

48 static void constructProblem(Problem &prob, FuncOp func) {
49  // set up catch-all operator type with unit latency
50  auto unitOpr = prob.getOrInsertOperatorType("unit");
51  prob.setLatency(unitOpr, 1);
52 
53  // parse additional operator type information attached to the test case
54  if (auto attr = func->getAttrOfType<ArrayAttr>("operatortypes")) {
55  for (auto &elem : parseArrayOfDicts(attr, "latency")) {
56  auto opr = prob.getOrInsertOperatorType(std::get<0>(elem));
57  prob.setLatency(opr, std::get<1>(elem));
58  }
59  }
60 
61  // construct problem (consider only the first block)
62  for (auto &op : func.getBlocks().front().getOperations()) {
63  prob.insertOperation(&op);
64 
65  if (auto oprRefAttr = op.getAttrOfType<StringAttr>("opr")) {
66  auto opr = prob.getOrInsertOperatorType(oprRefAttr.getValue());
67  prob.setLinkedOperatorType(&op, opr);
68  } else {
69  prob.setLinkedOperatorType(&op, unitOpr);
70  }
71  }
72 
73  // parse auxiliary dependences in the testcase, encoded as an array of arrays
74  // of integer attributes
75  if (auto attr = func->getAttrOfType<ArrayAttr>("auxdeps")) {
76  auto &ops = prob.getOperations();
77  unsigned nOps = ops.size();
78  for (auto &elemArr : parseArrayOfArrays(attr)) {
79  assert(elemArr.size() >= 2 && elemArr[0] < nOps && elemArr[1] < nOps);
80  Operation *from = ops[elemArr[0]];
81  Operation *to = ops[elemArr[1]];
82  auto res = prob.insertDependence(std::make_pair(from, to));
83  assert(succeeded(res));
84  }
85  }
86 }

88 static void constructCyclicProblem(CyclicProblem &prob, FuncOp func) {
89  // parse auxiliary dependences in the testcase (again), in order to set the
90  // optional distance in the cyclic problem
91  if (auto attr = func->getAttrOfType<ArrayAttr>("auxdeps")) {
92  auto &ops = prob.getOperations();
93  for (auto &elemArr : parseArrayOfArrays(attr)) {
94  if (elemArr.size() < 3)
95  continue; // skip this dependence, rather than setting the default value
96  Operation *from = ops[elemArr[0]];
97  Operation *to = ops[elemArr[1]];
98  unsigned dist = elemArr[2];
99  prob.setDistance(std::make_pair(from, to), dist);
100  }
101  }
102 }

104 static void constructSharedOperatorsProblem(SharedOperatorsProblem &prob,
105  FuncOp func) {
106  // parse operator type info (again) to extract optional operator limit
107  if (auto attr = func->getAttrOfType<ArrayAttr>("operatortypes")) {
108  for (auto &elem : parseArrayOfDicts(attr, "limit")) {
109  auto opr = prob.getOrInsertOperatorType(std::get<0>(elem));
110  prob.setLimit(opr, std::get<1>(elem));
111  }
112  }
113 }

115 static void emitSchedule(Problem &prob, StringRef attrName,
116  OpBuilder &builder) {
117  for (auto *op : prob.getOperations()) {
118  unsigned startTime = *prob.getStartTime(op);
119  op->setAttr(attrName, builder.getI32IntegerAttr(startTime));
120  }
121 }

137 void TestProblemPass::runOnFunction() {
138  auto func = getFunction();
139 
140  auto prob = Problem::get(func);
141  constructProblem(prob, func);
142 
143  if (failed(prob.check())) {
144  func->emitError("problem check failed");
145  return signalPassFailure();
146  }
147 
148  // get schedule from the test case
149  for (auto *op : prob.getOperations())
150  if (auto startTimeAttr = op->getAttrOfType<IntegerAttr>("problemStartTime"))
151  prob.setStartTime(op, startTimeAttr.getInt());
152 
153  if (failed(prob.verify())) {
154  func->emitError("problem verification failed");
155  return signalPassFailure();
156  }
157 }

339 void TestSimplexSchedulerPass::runOnFunction() {
340  auto func = getFunction();
341  Operation *lastOp = func.getBlocks().front().getTerminator();
342  OpBuilder builder(func.getContext());
343 
344  if (problemToTest == "Problem") {
345  auto prob = Problem::get(func);
346  constructProblem(prob, func);
347  assert(succeeded(prob.check()));
348 
349  if (failed(scheduleSimplex(prob, lastOp))) {
350  func->emitError("scheduling failed");
351  return signalPassFailure();
352  }
353 
354  if (failed(prob.verify())) {
355  func->emitError("schedule verification failed");
356  return signalPassFailure();
357  }
358 
359  emitSchedule(prob, "simplexStartTime", builder);
360  return;
361  }
362 
363  if (problemToTest == "CyclicProblem") {
364  auto prob = CyclicProblem::get(func);
365  constructProblem(prob, func);
366  constructCyclicProblem(prob, func);
367  assert(succeeded(prob.check()));
368 
369  if (failed(scheduleSimplex(prob, lastOp))) {
370  func->emitError("scheduling failed");
371  return signalPassFailure();
372  }
373 
374  if (failed(prob.verify())) {
375  func->emitError("schedule verification failed");
376  return signalPassFailure();
377  }
378 
379  func->setAttr("simplexInitiationInterval",
380  builder.getI32IntegerAttr(*prob.getInitiationInterval()));
381  emitSchedule(prob, "simplexStartTime", builder);
382  return;
383  }
384 
385  if (problemToTest == "SharedOperatorsProblem") {
386  auto prob = SharedOperatorsProblem::get(func);
387  constructProblem(prob, func);
388  constructSharedOperatorsProblem(prob, func);
389  assert(succeeded(prob.check()));
390 
391  if (failed(scheduleSimplex(prob, lastOp))) {
392  func->emitError("scheduling failed");
393  return signalPassFailure();
394  }
395 
396  if (failed(prob.verify())) {
397  func->emitError("schedule verification failed");
398  return signalPassFailure();
399  }
400 
401  emitSchedule(prob, "simplexStartTime", builder);
402  return;
403  }
404 
405  llvm_unreachable("Unsupported scheduling problem");
406 }
