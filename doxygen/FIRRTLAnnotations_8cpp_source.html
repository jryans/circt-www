109 static DictionaryAttr applyToDictionaryAttrImpl(const AnnotationSet &annoSet,
110  StringRef key,
111  ArrayRef<NamedAttribute> attrs,
112  bool sorted,
113  DictionaryAttr originalDict) {
114  // Find the location in the dictionary where the entry would go.
115  ArrayRef<NamedAttribute>::iterator it;
116  if (sorted) {
117  it = llvm::lower_bound(attrs, key);
118  if (it != attrs.end() && it->getName() != key)
119  it = attrs.end();
120  } else {
121  it = llvm::find_if(
122  attrs, [key](NamedAttribute attr) { return attr.getName() == key; });
123  }
124 
125  // Fast path in case there are no annotations in the dictionary and we are not
126  // supposed to add any.
127  if (it == attrs.end() && annoSet.empty())
128  return originalDict;
129 
130  // Fast path in case there already is an entry in the dictionary, it matches
131  // the set, and, in the case we're supposed to remove empty sets, we're not
132  // leaving an empty entry in the dictionary.
133  if (it != attrs.end() && it->getValue() == annoSet.getArrayAttr() &&
134  !annoSet.empty())
135  return originalDict;
136 
137  // If we arrive here, we are supposed to assemble a new dictionary.
138  SmallVector<NamedAttribute> newAttrs;
139  newAttrs.reserve(attrs.size() + 1);
140  newAttrs.append(attrs.begin(), it);
141  if (!annoSet.empty())
142  newAttrs.push_back(
143  {StringAttr::get(key, annoSet.getContext()), annoSet.getArrayAttr()});
144  if (it != attrs.end())
145  newAttrs.append(it + 1, attrs.end());
146  return sorted ? DictionaryAttr::getWithSorted(annoSet.getContext(), newAttrs)
147  : DictionaryAttr::get(annoSet.getContext(), newAttrs);
148 }

317 /// Remove all annotations from this annotation set for which `predicate`
318 /// returns true.
319 bool AnnotationSet::removeAnnotations(
320  llvm::function_ref<bool(Annotation)> predicate) {
321  // Fast path for empty sets.
322  auto attr = getArrayAttr();
323  if (!attr)
324  return false;
325 
326  // Search for the first match.
327  ArrayRef<Attribute> annos = getArrayAttr().getValue();
328  auto it = annos.begin();
329  while (it != annos.end() && !predicate(Annotation(*it)))
330  ++it;
331 
332  // Fast path for sets where the predicate never matched.
333  if (it == annos.end())
334  return false;
335 
336  // Build a filtered list of annotations.
337  SmallVector<Attribute> filteredAnnos;
338  filteredAnnos.reserve(annos.size());
339  filteredAnnos.append(annos.begin(), it);
340  ++it;
341  while (it != annos.end()) {
342  if (!predicate(Annotation(*it)))
343  filteredAnnos.push_back(*it);
344  ++it;
345  }
346  annotations = ArrayAttr::get(getContext(), filteredAnnos);
347  return true;
348 }

366 /// Remove all port annotations from a module or extmodule for which `predicate`
367 /// returns true.
368 bool AnnotationSet::removePortAnnotations(
369  Operation *module,
370  llvm::function_ref<bool(unsigned, Annotation)> predicate) {
371  auto ports = module->getAttr("portAnnotations").dyn_cast_or_null<ArrayAttr>();
372  if (!ports || ports.empty())
373  return false;
374 
375  // Collect results
376  SmallVector<Attribute> newAnnos;
377 
378  // Filter the annotations on each argument.
379  bool changed = false;
380  for (unsigned argNum = 0, argNumEnd = ports.size(); argNum < argNumEnd;
381  ++argNum) {
382  AnnotationSet annos(AnnotationSet(ports[argNum].cast<ArrayAttr>()));
383 
384  // Go through all annotations on this port and extract the interesting
385  // ones. If any modifications were done, keep a reduced set of attributes
386  // around for the port, otherwise just stick with the existing ones.
387  if (!annos.empty())
388  changed |= annos.removeAnnotations(
389  [&](Annotation anno) { return predicate(argNum, anno); });
390  newAnnos.push_back(annos.getArrayAttr());
391  }
392 
393  // If we have made any changes, apply them to the operation.
394  if (changed)
395  module->setAttr("portAnnotations",
396  ArrayAttr::get(module->getContext(), newAnnos));
397  return changed;
398 }
