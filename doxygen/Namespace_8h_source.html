23 /// A namespace that is used to store existing names and generate new names in
24 /// some scope within the IR. This exists to work around limitations of
25 /// SymbolTables. This acts as a base class providing facilities common to all
26 /// namespaces implementations.
27 class NamespaceBase {
28 public:
29  NamespaceBase() {}
30  NamespaceBase(const NamespaceBase &other) = default;
31  NamespaceBase(NamespaceBase &&other) : internal(std::move(other.internal)) {}
32 
33  NamespaceBase &operator=(const NamespaceBase &other) = default;
34  NamespaceBase &operator=(NamespaceBase &&other) {
35  internal = std::move(other.internal);
36  return *this;
37  }
38 
39  /// Empty the namespace.
40  void clear() { internal.clear(); }
41 
42  /// Return a unique name, derived from the input `name`, and add the new name
43  /// to the internal namespace. There are two possible outcomes for the
44  /// returned name:
45  ///
46  /// 1. The original name is returned.
47  /// 2. The name is given a `_<n>` suffix where `<n>` is a number starting from
48  /// `_0` and incrementing by one each time.
49  StringRef newName(const Twine &name) {
50  // Special case the situation where there is no name collision to avoid
51  // messing with the SmallString allocation below.
52  llvm::SmallString<64> tryName;
53  auto inserted = internal.insert(name.toStringRef(tryName));
54  if (inserted.second)
55  return inserted.first->getKey();
56 
57  // Try different suffixes until we get a collision-free one.
58  size_t i = 0;
59  if (tryName.empty())
60  name.toVector(tryName); // toStringRef may leave tryName unfilled
61  tryName.push_back('_');
62  size_t baseLength = tryName.size();
63  for (;;) {
64  tryName.resize(baseLength);
65  Twine(i++).toVector(tryName); // append integer to tryName
66  auto inserted = internal.insert(tryName);
67  if (inserted.second)
68  return inserted.first->getKey();
69  }
70  }
71 
72 protected:
73  llvm::StringSet<> internal;
74 };

76 /// The namespace of a `CircuitOp`, generally inhabited by modules.
77 struct CircuitNamespace : public NamespaceBase {
78  CircuitNamespace() {}
79  CircuitNamespace(CircuitOp circuit) { add(circuit); }
80 
81  /// Populate the namespace from a circuit operation. This namespace will be
82  /// composed of any operation in the first level of the circuit that contains
83  /// a symbol.
84  void add(CircuitOp circuit) {
85  for (auto &op : *circuit.getBody())
86  if (auto symbol = op.getAttrOfType<mlir::StringAttr>(
87  SymbolTable::getSymbolAttrName()))
88  internal.insert(symbol.getValue());
89  }
90 };
91 
92 /// The namespace of a `FModuleLike` operation, generally inhabited by its ports
93 /// and declarations.
94 struct ModuleNamespace : public NamespaceBase {
95  ModuleNamespace() {}
96  ModuleNamespace(FModuleLike module) { add(module); }
97 
98  /// Populate the namespace from a module-like operation. This namespace will
99  /// be composed of the `inner_sym`s of the module's ports and declarations.
100  void add(FModuleLike module) {
101  for (auto portSymbol : module.getPortSymbolsAttr().getAsRange<StringAttr>())
102  if (!portSymbol.getValue().empty())
103  internal.insert(portSymbol.getValue());
104  module.walk([&](Operation *op) {
105  auto attr = op->getAttrOfType<StringAttr>("inner_sym");
106  if (attr)
107  internal.insert(attr.getValue());
108  });
109  }
110 };
