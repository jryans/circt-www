26 hw::HWModuleExternOp circt::calyx::getExternHWModule(OpBuilder &builder,
27  ComponentOp op) {
28  SmallVector<hw::PortInfo, 8> hwPortInfos;
29  auto addHWPortInfo = [&](auto portInfos, hw::PortDirection direction) {
30  for (auto portInfo : enumerate(portInfos)) {
31  hw::PortInfo hwPortInfo;
32  hwPortInfo.direction = direction;
33  hwPortInfo.type = portInfo.value().type;
34  hwPortInfo.argNum = portInfo.index();
35  hwPortInfo.name = portInfo.value().name;
36  hwPortInfos.push_back(hwPortInfo);
37  }
38  };
39 
40  addHWPortInfo(op.getInputPortInfo(), hw::PortDirection::INPUT);
41  addHWPortInfo(op.getOutputPortInfo(), hw::PortDirection::OUTPUT);
42 
43  return builder.create<hw::HWModuleExternOp>(
44  op->getLoc(), builder.getStringAttr(op.getName()), hwPortInfos);
45 }
