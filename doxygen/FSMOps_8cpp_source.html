31  state.addAttribute(getTypeAttrName(), TypeAttr::get(type));
32  state.attributes.append(attrs.begin(), attrs.end());

61  hw::PortInfo port;
62  port.name = builder.getStringAttr("out" + std::to_string(i));
63  port.direction = circt::hw::PortDirection::OUTPUT;
64  port.type = machineType.getResult(i);
65  port.argNum = i;
66  ports.push_back(port);
67  }

87 static LogicalResult compareTypes(TypeRange rangeA, TypeRange rangeB) {
88  if (rangeA.size() != rangeB.size())
89  return failure();
90 
91  int64_t index = 0;
92  for (auto zip : llvm::zip(rangeA, rangeB)) {
93  if (std::get<0>(zip) != std::get<1>(zip))
94  return failure();
95  ++index;
96  }
97 
98  return success();
99 }

101 static LogicalResult verifyMachineOp(MachineOp op) {
102  // If this function is external there is nothing to do.
103  if (op.isExternal())
104  return success();
105 
106  if (!op.stateType().isa<IntegerType>())
107  return op.emitOpError("state must be integer type");
108 
109  // Verify that the argument list of the function and the arg list of the entry
110  // block line up. The trait already verified that the number of arguments is
111  // the same between the signature and the block.
112  if (failed(compareTypes(op.getType().getInputs(),
113  op.front().getArgumentTypes())))
114  return op.emitOpError(
115  "entry block argument types must match the machine input types");
116 
117  // Verify that the machine only has one block terminated with OutputOp.
118  if (!llvm::hasSingleElement(op))
119  return op.emitOpError("must only have a single block");
120 
121  return success();
122 }

152 template <typename OpType>
153 static LogicalResult verifyCallerTypes(OpType op) {
154  auto machine = op.getMachine();
155  if (!machine)
156  return op.emitError("cannot find machine definition");
157 
158  // Check operand types first.
159  if (failed(compareTypes(machine.getType().getInputs(),
160  op.inputs().getTypes()))) {
161  auto diag =
162  op.emitOpError("operand types must match the machine input types");
163  diag.attachNote(machine->getLoc()) << "original machine declared here";
164  return failure();
165  }
166 
167  // Check result types.
168  if (failed(compareTypes(machine.getType().getResults(),
169  op.outputs().getTypes()))) {
170  auto diag =
171  op.emitOpError("result types must match the machine output types");
172  diag.attachNote(machine->getLoc()) << "original machine declared here";
173  return failure();
174  }
175 
176  return success();
177 }

210 LogicalResult StateOp::canonicalize(StateOp op, PatternRewriter &rewriter) {
211  bool hasAlwaysTakenTransition = false;
212  SmallVector<TransitionOp, 4> transitionsToErase;
213  // Remove all transitions after an "always-taken" transition.
214  for (auto transition : op.transitions().getOps<TransitionOp>()) {
215  if (!hasAlwaysTakenTransition)
216  hasAlwaysTakenTransition = transition.isAlwaysTaken();
217  else
218  transitionsToErase.push_back(transition);
219  }
220 
221  for (auto transition : transitionsToErase)
222  rewriter.eraseOp(transition);
223 
224  return failure(transitionsToErase.empty());
225 }

231 static LogicalResult verifyOutputOp(OutputOp op) {
232  if (op->getParentRegion() == &op->getParentOfType<StateOp>().transitions()) {
233  if (op.getNumOperands() != 0)
234  op.emitOpError("transitions region must not output any value");
235  return success();
236  }
237 
238  // Verify that the result list of the machine and the operand list of the
239  // OutputOp line up.
240  auto machine = op->getParentOfType<MachineOp>();
241  if (failed(
242  compareTypes(machine.getType().getResults(), op.getOperandTypes())))
243  return op.emitOpError("operand types must match the machine output types");
244 
245  return success();
246 }

261 bool TransitionOp::isAlwaysTaken() {
262  auto guardReturn = getGuardReturn();
263  if (guardReturn.getNumOperands() == 0)
264  return true;
265 
266  if (auto constantOp =
267  guardReturn.getOperand(0).getDefiningOp<mlir::ConstantOp>())
268  return constantOp.value().cast<BoolAttr>().getValue();
269 
270  return false;
271 }

273 LogicalResult TransitionOp::canonicalize(TransitionOp op,
274  PatternRewriter &rewriter) {
275  auto guardReturn = op.getGuardReturn();
276  if (guardReturn.getNumOperands() == 1)
277  if (auto constantOp =
278  guardReturn.getOperand(0).getDefiningOp<mlir::ConstantOp>()) {
279  // Simplify when the guard region returns a constant value.
280  if (constantOp.value().cast<BoolAttr>().getValue()) {
281  // Replace the original return op with a new one without any operands
282  // if the constant is TRUE.
283  rewriter.setInsertionPoint(guardReturn);
284  rewriter.create<fsm::ReturnOp>(guardReturn.getLoc());
285  rewriter.eraseOp(guardReturn);
286  } else {
287  // Erase the whole transition op if the constant is FALSE, because the
288  // transition will never be taken.
289  rewriter.eraseOp(op);
290  }
291  return success();
292  }
293 
294  return failure();
295 }

297 static LogicalResult verifyTransitionOp(TransitionOp op) {
298  if (!op.getNextState())
299  return op.emitOpError("cannot find the definition of the next state `")
300  << op.nextState() << "`";
301 
302  // Verify the action region.
303  if (op.action().front().getTerminator()->getNumOperands() != 0)
304  return op.emitOpError("action region must not return any value");
305 
306  // Verify the transition is located in the correct region.
307  if (op->getParentRegion() != &op.getCurrentState().transitions())
308  return op.emitOpError("must only be located in the transitions region");
309 
310  return success();
311 }
