26 static Attribute parseRootedInstancePath(AsmParser &p) {
27  FlatSymbolRefAttr root;
28  if (p.parseAttribute(root) || p.parseLSquare())
29  return Attribute();
30  SmallVector<StringAttr, 16> path;
31  if (p.parseOptionalRSquare()) {
32  do {
33  StringAttr instName;
34  if (p.parseAttribute(instName))
35  return Attribute();
36  path.push_back(instName);
37  } while (!p.parseOptionalComma());
38  if (p.parseRSquare())
39  return Attribute();
40  }
41  return RootedInstancePathAttr::get(p.getContext(), root, path);
42 }

50 Attribute SwitchInstanceAttr::parse(AsmParser &p, Type type) {
51  if (p.parseLess())
52  return Attribute();
53  if (!p.parseOptionalGreater())
54  return SwitchInstanceAttr::get(p.getContext(), {});
55 
56  SmallVector<SwitchInstanceCaseAttr> instPairs;
57  do {
58  auto path =
59  parseRootedInstancePath(p).dyn_cast_or_null<RootedInstancePathAttr>();
60  if (!path)
61  return Attribute();
62 
63  Attribute attr;
64  if (p.parseEqual() || p.parseAttribute(attr))
65  return Attribute();
66  instPairs.push_back(
67  SwitchInstanceCaseAttr::get(p.getContext(), path, attr));
68  } while (!p.parseOptionalComma());
69  if (p.parseGreater())
70  return Attribute();
71 
72  return SwitchInstanceAttr::get(p.getContext(), instPairs);
73 }

162 Attribute MSFTDialect::parseAttribute(DialectAsmParser &p, Type type) const {
163  StringRef attrName;
164  Attribute attr;
165  if (p.parseKeyword(&attrName))
166  return Attribute();
167  auto parseResult = generatedAttributeParser(p, attrName, type, attr);
168  if (parseResult.hasValue())
169  return attr;
170  p.emitError(p.getNameLoc(), "Unexpected msft attribute '" + attrName + "'");
171  return {};
172 }
