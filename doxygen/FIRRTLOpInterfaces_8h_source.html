27 /// This holds the name and type that describes the module's ports.
28 struct PortInfo {
29  StringAttr name;
30  FIRRTLType type;
31  Direction direction;
32  StringAttr sym = {};
33  Location loc = UnknownLoc::get(type.getContext());
34  AnnotationSet annotations = AnnotationSet(type.getContext());
35 
36  StringRef getName() const { return name ? name.getValue() : ""; }
37 
38  /// Return true if this is a simple output-only port. If you want the
39  /// direction of the port, use the \p direction parameter.
40  bool isOutput() {
41  auto flags = type.getRecursiveTypeProperties();
42  return flags.isPassive && !flags.containsAnalog &&
43  direction == Direction::Out;
44  }
45 
46  /// Return true if this is a simple input-only port. If you want the
47  /// direction of the port, use the \p direction parameter.
48  bool isInput() {
49  auto flags = type.getRecursiveTypeProperties();
50  return flags.isPassive && !flags.containsAnalog &&
51  direction == Direction::In;
52  }
53 
54  /// Return true if this is an inout port. This will be true if the port
55  /// contains either bi-directional signals or analog types.
56  bool isInOut() { return !isOutput() && !isInput(); }
57 };
