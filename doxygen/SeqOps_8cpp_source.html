23 ParseResult parseCompRegOp(OpAsmParser &parser, OperationState &result) {
24  llvm::SMLoc loc = parser.getCurrentLocation();
25  SmallVector<OpAsmParser::OperandType, 4> operands;
26  if (parser.parseOperandList(operands))
27  return failure();
28  switch (operands.size()) {
29  case 0:
30  return parser.emitError(loc, "expected operands");
31  case 1:
32  return parser.emitError(loc, "expected clock operand");
33  case 2:
34  // No reset.
35  break;
36  case 3:
37  return parser.emitError(loc, "expected resetValue operand");
38  case 4:
39  // reset and reset value included.
40  break;
41  default:
42  return parser.emitError(loc, "too many operands");
43  }
44 
45  Type ty;
46  if (parser.parseOptionalAttrDict(result.attributes) || parser.parseColon() ||
47  parser.parseType(ty))
48  return failure();
49  Type i1 = IntegerType::get(result.getContext(), 1);
50 
51  // If there was no name specified, check to see if there was a useful name
52  // specified in the asm file.
53  if (!result.attributes.getNamed("name")) {
54  // If there is no explicit name attribute, get it from the SSA result name.
55  // If numeric, just use an empty name.
56  StringRef resultName = parser.getResultName(0).first;
57  if (!resultName.empty() && isdigit(resultName[0]))
58  resultName = "";
59  result.addAttribute("name", parser.getBuilder().getStringAttr(resultName));
60  }
61 
62  result.addTypes({ty});
63  if (operands.size() == 2)
64  return parser.resolveOperands(operands, {ty, i1}, loc, result.operands);
65  else
66  return parser.resolveOperands(operands, {ty, i1, i1, ty}, loc,
67  result.operands);
68 }

70 static void printCompRegOp(::mlir::OpAsmPrinter &p, CompRegOp reg) {
71  p << ' ' << reg.input() << ", " << reg.clk();
72  if (reg.reset())
73  p << ", " << reg.reset() << ", " << reg.resetValue() << ' ';
74 
75  SmallVector<StringRef> elidedAttrs;
76  // Determine if 'name' can be elided.
77  if (reg.name().empty()) {
78  elidedAttrs.push_back("name");
79  } else {
80  SmallString<32> resultNameStr;
81  llvm::raw_svector_ostream tmpStream(resultNameStr);
82  p.printOperand(reg.data(), tmpStream);
83  auto actualName = tmpStream.str().drop_front();
84  if (actualName == reg.name())
85  elidedAttrs.push_back("name");
86  }
87 
88  p.printOptionalAttrDict(reg->getAttrs(), elidedAttrs);
89  p << " : " << reg.input().getType();
90 }
