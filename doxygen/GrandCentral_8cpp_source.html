119 /// Conversion from a `DescribedSignal` to YAML. This is
120 /// implemented using YAML normalization to first convert this to an internal
121 /// `Field` structure which has a one-to-one mapping to the YAML represntation.
122 template <>
123 struct MappingContextTraits<DescribedSignal, Context> {
124  /// A one-to-one representation with a YAML representation of a signal/field.
125  struct Field {
126  /// The name of the field.
127  StringRef name;
128 
129  /// An optional, textual description of what the field is.
130  Optional<StringRef> description;
131 
132  /// The dimensions of the field.
133  SmallVector<unsigned, 2> dimensions;
134 
135  /// The width of the underlying type.
136  unsigned width;
137 
138  /// Construct a `Field` from a `DescribedSignal` (an `sv::InterfaceSignalOp`
139  /// with an optional description).
140  Field(IO &io, DescribedSignal &op)
141  : name(op.signal.sym_nameAttr().getValue()) {
142 
143  // Convert the description from a `StringAttr` (which may be null) to an
144  // `Optional<StringRef>`. This aligns exactly with the YAML
145  // representation.
146  if (op.description) {
147  description = op.description.getValue();
148  description->consume_front("// ");
149  }
150 
151  // Unwrap the type of the field into an array of dimensions and a width.
152  // By example, this is going from the following hardware type:
153  //
154  // !hw.uarray<1xuarray<2xuarray<3xi8>>>
155  //
156  // To the following representation:
157  //
158  // dimensions: [ 3, 2, 1 ]
159  // width: 8
160  //
161  // Note that the above is equivalenet to the following Verilog
162  // specification.
163  //
164  // wire [7:0] foo [2:0][1:0][0:0]
165  //
166  // Do this by repeatedly unwrapping unpacked array types until you get to
167  // the underlying type. The dimensions need to be reversed as this
168  // unwrapping happens in reverse order of the final representation.
169  auto tpe = op.signal.type();
170  while (auto vector = tpe.dyn_cast<hw::UnpackedArrayType>()) {
171  dimensions.push_back(vector.getSize());
172  tpe = vector.getElementType();
173  }
174  dimensions = SmallVector<unsigned>(llvm::reverse(dimensions));
175 
176  // The final non-array type must be an integer. Leave this as an assert
177  // with a blind cast because we generated this type in this pass (and we
178  // therefore cannot fail this cast).
179  assert(tpe.isa<IntegerType>());
180  width = tpe.cast<IntegerType>().getWidth();
181  }
182 
183  /// A no-argument constructor is necessary to work with LLVM's YAML library.
184  Field(IO &io){UNIMPLEMENTED_DEFAULT("Field")}
185 
186  /// This cannot be denomralized back to an interface op.
187  DescribedSignal denormalize(IO &) {
188  UNIMPLEMENTED_DENORM("DescribedSignal")
189  }
190  };

254 /// Conversion from an `sv::InterfaceOp` to YAML. This is implemented using
255 /// YAML normalization to first convert the interface to an internal `Interface`
256 /// which reformats the Grand Central-generated interface into the YAML format.
257 template <>
258 struct MappingContextTraits<sv::InterfaceOp, Context> {
259  /// A YAML-serializable representation of an interface. This consists of
260  /// fields (vector or ground types) and nested interfaces.
261  struct Interface {
262  /// The name of the interface.
263  StringRef name;
264 
265  /// All ground or vectors that make up the interface.
266  std::vector<DescribedSignal> fields;
267 
268  /// Instantiations of _other_ interfaces.
269  std::vector<DescribedInstance> instances;
270 
271  /// Construct an `Interface` from an `sv::InterfaceOp`. This is tuned to
272  /// "parse" the structure of an interface that the Grand Central pass
273  /// generates. The structure of `Field`s and `Instance`s is documented
274  /// below.
275  ///
276  /// A field will look like the following. The verbatim description is
277  /// optional:
278  ///
279  /// sv.verbatim "// <description>" {
280  /// firrtl.grandcentral.yaml.type = "description",
281  /// symbols = []}
282  /// sv.interface.signal @<name> : <type>
283  ///
284  /// An interface instanctiation will look like the following. The verbatim
285  /// description is optional.
286  ///
287  /// sv.verbatim "// <description>" {
288  /// firrtl.grandcentral.type = "description",
289  /// symbols = []}
290  /// sv.verbatim "<name> <symbol>();" {
291  /// firrtl.grandcentral.yaml.name = "<name>",
292  /// firrtl.grandcentral.yaml.dimensions = [<first dimension>, ...],
293  /// firrtl.grandcentral.yaml.symbol = @<symbol>,
294  /// firrtl.grandcentral.yaml.type = "instance",
295  /// symbols = []}
296  ///
297  Interface(IO &io, sv::InterfaceOp &op) : name(op.getName()) {
298  // A mutable store of the description. This occurs in the op _before_ the
299  // field or instance, so we need someplace to put it until we use it.
300  StringAttr description = {};
301 
302  for (auto &op : op.getBodyBlock()->getOperations()) {
303  TypeSwitch<Operation *>(&op)
304  // A verbatim op is either a description or an interface
305  // instantiation.
306  .Case<sv::VerbatimOp>([&](sv::VerbatimOp op) {
307  auto tpe = op->getAttrOfType<StringAttr>(
308  "firrtl.grandcentral.yaml.type");
309 
310  // This is a descripton. Update the mutable description and
311  // continue;
312  if (tpe.getValue() == "description") {
313  description = op.stringAttr();
314  return;
315  }
316 
317  // This is an unsupported construct. Just drop it.
318  if (tpe.getValue() == "unsupported") {
319  description = {};
320  return;
321  }
322 
323  // This is an instance of another interface. Add the symbol to
324  // the vector of instances.
325  auto name = op->getAttrOfType<StringAttr>(
326  "firrtl.grandcentral.yaml.name");
327  auto dimensions = op->getAttrOfType<ArrayAttr>(
328  "firrtl.grandcentral.yaml.dimensions");
329  auto symbol = op->getAttrOfType<FlatSymbolRefAttr>(
330  "firrtl.grandcentral.yaml.symbol");
331  instances.push_back(
332  DescribedInstance({name, description, dimensions, symbol}));
333  })
334  // An interface signal op is a field.
335  .Case<sv::InterfaceSignalOp>([&](sv::InterfaceSignalOp op) {
336  fields.push_back(DescribedSignal({description, op}));
337  description = {};
338  });
339  }
340  }
341 
342  /// A no-argument constructor is necessary to work with LLVM's YAML library.
343  Interface(IO &io){UNIMPLEMENTED_DEFAULT("Interface")}
344 
345  /// This cannot be denomralized back to an interface op.
346  sv::InterfaceOp denormalize(IO &) {
347  UNIMPLEMENTED_DENORM("sv::InterfaceOp")
348  }
349  };

354  io.mapRequired("fields", keys->fields, ctx);
355  io.mapRequired("instances", keys->instances, ctx);
356  }

368 /// A helper to build verbatim strings with symbol placeholders. Provides a
369 /// mechanism to snapshot the current string and symbols and restore back to
370 /// this state after modifications. These snapshots are particularly useful when
371 /// the string is assembled through hierarchical traversal of some sort, which
372 /// populates the string with a prefix common to all children of a hierarchy
373 /// (like the interface field traversal in the `GrandCentralPass`).
374 ///
375 /// The intended use is as follows:
376 ///
377 /// void baz(VerbatimBuilder &v) {
378 /// foo(v.snapshot().append("bar"));
379 /// }
380 ///
381 /// The function `baz` takes a snapshot of the current verbatim text `v`, adds
382 /// "bar" to it and calls `foo` with that appended verbatim text. After the call
383 /// to `foo` returns, any changes made by `foo` as well as the "bar" are dropped
384 /// from the verbatim text `v`, as the temporary snapshot goes out of scope.
385 struct VerbatimBuilder {
386  struct Base {
387  SmallString<128> string;
388  SmallVector<Attribute> symbols;
389  VerbatimBuilder builder() { return VerbatimBuilder(*this); }
390  operator VerbatimBuilder() { return builder(); }
391  };
392 
393  /// Constructing a builder will snapshot the `Base` which holds the actual
394  /// string and symbols.
395  VerbatimBuilder(Base &base)
396  : base(base), stringBaseSize(base.string.size()),
397  symbolsBaseSize(base.symbols.size()) {}
398 
399  /// Destroying a builder will reset the `Base` to the original string and
400  /// symbols.
401  ~VerbatimBuilder() {
402  base.string.resize(stringBaseSize);
403  base.symbols.resize(symbolsBaseSize);
404  }
405 
406  // Disallow copying.
407  VerbatimBuilder(const VerbatimBuilder &) = delete;
408  VerbatimBuilder &operator=(const VerbatimBuilder &) = delete;
409 
410  /// Take a snapshot of the current string and symbols. This returns a new
411  /// `VerbatimBuilder` that will reset to the current state of the string once
412  /// destroyed.
413  VerbatimBuilder snapshot() { return VerbatimBuilder(base); }
414 
415  /// Get the current string.
416  StringRef getString() const { return base.string; }
417  /// Get the current symbols;
418  ArrayRef<Attribute> getSymbols() const { return base.symbols; }
419 
420  /// Append to the string.
421  VerbatimBuilder &append(char c) {
422  base.string.push_back(c);
423  return *this;
424  }
425 
426  /// Append to the string.
427  VerbatimBuilder &append(const Twine &twine) {
428  twine.toVector(base.string);
429  return *this;
430  }
431 
432  /// Append a placeholder and symbol to the string.
433  VerbatimBuilder &append(Attribute symbol) {
434  unsigned id = base.symbols.size();
435  base.symbols.push_back(symbol);
436  append("{{" + Twine(id) + "}}");
437  return *this;
438  }
439 
440  VerbatimBuilder &operator+=(char c) { return append(c); }
441  VerbatimBuilder &operator+=(const Twine &twine) { return append(twine); }
442  VerbatimBuilder &operator+=(Attribute symbol) { return append(symbol); }
443 
444 private:
445  Base &base;
446  size_t stringBaseSize;
447  size_t symbolsBaseSize;
448 };

450 /// A wrapper around a string that is used to encode a type which cannot be
451 /// represented by an mlir::Type for some reason. This is currently used to
452 /// represent either an interface, a n-dimensional vector of interfaces, or a
453 /// tombstone for an actually unsupported type (e.g., an AugmentedBooleanType).
454 struct VerbatimType {
455  /// The textual representation of the type.
456  std::string str;
457 
458  /// True if this is a type which must be "instatiated" and requires a trailing
459  /// "()".
460  bool instantiation;
461 
462  /// A vector storing the width of each dimension of the type.
463  SmallVector<int32_t, 4> dimensions = {};
464 
465  /// Serialize this type to a string.
466  std::string toStr(StringRef name) {
467  SmallString<64> stringType(str);
468  stringType.append(" ");
469  stringType.append(name);
470  for (auto d : llvm::reverse(dimensions)) {
471  stringType.append("[");
472  stringType.append(Twine(d).str());
473  stringType.append("]");
474  }
475  if (instantiation)
476  stringType.append("()");
477  stringType.append(";");
478  return std::string(stringType);
479  }

678 Optional<Attribute> GrandCentralPass::fromAttr(Attribute attr) {
679  auto dict = attr.dyn_cast<DictionaryAttr>();
680  if (!dict) {
681  emitCircuitError() << "attribute is not a dictionary: " << attr << "\n";
682  return None;
683  }
684 
685  auto clazz = dict.getAs<StringAttr>("class");
686  if (!clazz) {
687  emitCircuitError() << "missing 'class' key in " << dict << "\n";
688  return None;
689  }
690 
691  auto classBase = clazz.getValue();
692  classBase.consume_front("sifive.enterprise.grandcentral.Augmented");
693 
694  if (classBase == "BundleType") {
695  if (dict.getAs<StringAttr>("defName") && dict.getAs<ArrayAttr>("elements"))
696  return AugmentedBundleTypeAttr::get(&getContext(), dict);
697  emitCircuitError() << "has an invalid AugmentedBundleType that does not "
698  "contain 'defName' and 'elements' fields: "
699  << dict;
700  } else if (classBase == "VectorType") {
701  if (dict.getAs<StringAttr>("name") && dict.getAs<ArrayAttr>("elements"))
702  return AugmentedVectorTypeAttr::get(&getContext(), dict);
703  emitCircuitError() << "has an invalid AugmentedVectorType that does not "
704  "contain 'name' and 'elements' fields: "
705  << dict;
706  } else if (classBase == "GroundType") {
707  auto id = dict.getAs<IntegerAttr>("id");
708  auto name = dict.getAs<StringAttr>("name");
709  if (id && leafMap.count(id) && name)
710  return AugmentedGroundTypeAttr::get(&getContext(), dict);
711  if (!id || !name)
712  emitCircuitError() << "has an invalid AugmentedGroundType that does not "
713  "contain 'id' and 'name' fields: "
714  << dict;
715  if (id && !leafMap.count(id))
716  emitCircuitError() << "has an AugmentedGroundType with 'id == "
717  << id.getValue().getZExtValue()
718  << "' that does not have a scattered leaf to connect "
719  "to in the circuit "
720  "(was the leaf deleted or constant prop'd away?)";
721  } else if (classBase == "StringType") {
722  if (auto name = dict.getAs<StringAttr>("name"))
723  return AugmentedStringTypeAttr::get(&getContext(), dict);
724  } else if (classBase == "BooleanType") {
725  if (auto name = dict.getAs<StringAttr>("name"))
726  return AugmentedBooleanTypeAttr::get(&getContext(), dict);
727  } else if (classBase == "IntegerType") {
728  if (auto name = dict.getAs<StringAttr>("name"))
729  return AugmentedIntegerTypeAttr::get(&getContext(), dict);
730  } else if (classBase == "DoubleType") {
731  if (auto name = dict.getAs<StringAttr>("name"))
732  return AugmentedDoubleTypeAttr::get(&getContext(), dict);
733  } else if (classBase == "LiteralType") {
734  if (auto name = dict.getAs<StringAttr>("name"))
735  return AugmentedLiteralTypeAttr::get(&getContext(), dict);
736  } else if (classBase == "DeletedType") {
737  if (auto name = dict.getAs<StringAttr>("name"))
738  return AugmentedDeletedTypeAttr::get(&getContext(), dict);
739  } else {
740  emitCircuitError() << "has an invalid AugmentedType";
741  }
742  return None;
743 }

745 bool GrandCentralPass::traverseField(Attribute field, IntegerAttr id,
746  VerbatimBuilder &path) {
747  return TypeSwitch<Attribute, bool>(field)
748  .Case<AugmentedGroundTypeAttr>([&](auto ground) {
749  Value leafValue = leafMap.lookup(ground.getID());
750  assert(leafValue && "leafValue not found");
751 
752  auto builder =
753  OpBuilder::atBlockEnd(companionIDMap.lookup(id).mapping.getBody());
754 
755  auto enclosing = getEnclosingModule(leafValue);
756  auto srcPaths = instancePaths->getAbsolutePaths(enclosing);
757  assert(srcPaths.size() == 1 &&
758  "Unable to handle multiply instantiated companions");
759 
760  // Add the root module.
761  path += " = ";
762  path += FlatSymbolRefAttr::get(SymbolTable::getSymbolName(
763  srcPaths[0].empty()
764  ? enclosing
765  : srcPaths[0][0]->getParentOfType<FModuleLike>()));
766 
767  // Add the source path.
768  for (auto inst : srcPaths[0]) {
769  path += '.';
770  path += getInnerRefTo(inst);
771  }
772 
773  // Add the leaf value to the path.
774  auto uloc = builder.getUnknownLoc();
775  path += '.';
776  if (auto blockArg = leafValue.dyn_cast<BlockArgument>()) {
777  auto module = cast<FModuleOp>(blockArg.getOwner()->getParentOp());
778  path += getInnerRefTo(module, blockArg.getArgNumber());
779  } else {
780  path += getInnerRefTo(leafValue.getDefiningOp());
781  }
782 
783  // Assemble the verbatim op.
784  builder.create<sv::VerbatimOp>(
785  uloc,
786  StringAttr::get(&getContext(), "assign " + path.getString() + ";"),
787  ValueRange{}, ArrayAttr::get(&getContext(), path.getSymbols()));
788  return true;
789  })
790  .Case<AugmentedVectorTypeAttr>([&](auto vector) {
791  bool notFailed = true;
792  auto elements = vector.getElements();
793  for (size_t i = 0, e = elements.size(); i != e; ++i) {
794  auto field = fromAttr(elements[i]);
795  if (!field)
796  return false;
797  notFailed &=
798  traverseField(field.getValue(), id,
799  path.snapshot().append("[" + Twine(i) + "]"));
800  }
801  return notFailed;
802  })
803  .Case<AugmentedBundleTypeAttr>([&](AugmentedBundleTypeAttr bundle) {
804  bool anyFailed = true;
805  for (auto element : bundle.getElements()) {
806  auto field = fromAttr(element);
807  if (!field)
808  return false;
809  auto name = element.cast<DictionaryAttr>().getAs<StringAttr>("name");
810  if (!name)
811  name = element.cast<DictionaryAttr>().getAs<StringAttr>("defName");
812  anyFailed &=
813  traverseField(field.getValue(), id,
814  path.snapshot().append("." + name.getValue()));
815  }
816 
817  return anyFailed;
818  })
819  .Case<AugmentedStringTypeAttr>([&](auto a) { return false; })
820  .Case<AugmentedBooleanTypeAttr>([&](auto a) { return false; })
821  .Case<AugmentedIntegerTypeAttr>([&](auto a) { return false; })
822  .Case<AugmentedDoubleTypeAttr>([&](auto a) { return false; })
823  .Case<AugmentedLiteralTypeAttr>([&](auto a) { return false; })
824  .Case<AugmentedDeletedTypeAttr>([&](auto a) { return false; })
825  .Default([](auto a) { return true; });
826 }

828 Optional<TypeSum> GrandCentralPass::computeField(Attribute field,
829  IntegerAttr id,
830  StringAttr prefix,
831  VerbatimBuilder &path) {
832 
833  auto unsupported = [&](StringRef name, StringRef kind) {
834  return VerbatimType({("// <unsupported " + kind + " type>").str(), false});
835  };
836 
837  return TypeSwitch<Attribute, Optional<TypeSum>>(field)
838  .Case<AugmentedGroundTypeAttr>(
839  [&](AugmentedGroundTypeAttr ground) -> Optional<TypeSum> {
840  // Traverse to generate mappings.
841  traverseField(field, id, path);
842  auto value = leafMap.lookup(ground.getID());
843  auto tpe = value.getType().cast<FIRRTLType>();
844  if (!tpe.isGround()) {
845  value.getDefiningOp()->emitOpError()
846  << "cannot be added to interface with id '"
847  << id.getValue().getZExtValue()
848  << "' because it is not a ground type";
849  return None;
850  }
851  return TypeSum(IntegerType::get(getOperation().getContext(),
852  tpe.getBitWidthOrSentinel()));
853  })
854  .Case<AugmentedVectorTypeAttr>(
855  [&](AugmentedVectorTypeAttr vector) -> Optional<TypeSum> {
856  bool notFailed = true;
857  auto elements = vector.getElements();
858  auto firstElement = fromAttr(elements[0]);
859  auto elementType =
860  computeField(firstElement.getValue(), id, prefix,
861  path.snapshot().append("[" + Twine(0) + "]"));
862  if (!elementType)
863  return None;
864 
865  for (size_t i = 1, e = elements.size(); i != e; ++i) {
866  auto subField = fromAttr(elements[i]);
867  if (!subField)
868  return None;
869  notFailed &=
870  traverseField(subField.getValue(), id,
871  path.snapshot().append("[" + Twine(i) + "]"));
872  }
873 
874  if (auto *tpe = std::get_if<Type>(&elementType.getValue()))
875  return TypeSum(
876  hw::UnpackedArrayType::get(*tpe, elements.getValue().size()));
877  auto str = std::get<VerbatimType>(elementType.getValue());
878  str.dimensions.push_back(elements.getValue().size());
879  return TypeSum(str);
880  })
881  .Case<AugmentedBundleTypeAttr>(
882  [&](AugmentedBundleTypeAttr bundle) -> TypeSum {
883  auto iface = traverseBundle(bundle, id, prefix, path);
884  assert(iface && iface.getValue());
885  return VerbatimType({getInterfaceName(prefix, bundle), true});
886  })
887  .Case<AugmentedStringTypeAttr>([&](auto field) -> TypeSum {
888  return unsupported(field.getName().getValue(), "string");
889  })
890  .Case<AugmentedBooleanTypeAttr>([&](auto field) -> TypeSum {
891  return unsupported(field.getName().getValue(), "boolean");
892  })
893  .Case<AugmentedIntegerTypeAttr>([&](auto field) -> TypeSum {
894  return unsupported(field.getName().getValue(), "integer");
895  })
896  .Case<AugmentedDoubleTypeAttr>([&](auto field) -> TypeSum {
897  return unsupported(field.getName().getValue(), "double");
898  })
899  .Case<AugmentedLiteralTypeAttr>([&](auto field) -> TypeSum {
900  return unsupported(field.getName().getValue(), "literal");
901  })
902  .Case<AugmentedDeletedTypeAttr>([&](auto field) -> TypeSum {
903  return unsupported(field.getName().getValue(), "deleted");
904  });
905 }

907 /// Traverse an Annotation that is an AugmentedBundleType. During
908 /// traversal, construct any discovered SystemVerilog interfaces. If this
909 /// is the root interface, instantiate that interface in the parent. Recurse
910 /// into fields of the AugmentedBundleType to construct nested interfaces
911 /// and generate stringy-typed SystemVerilog hierarchical references to
912 /// drive the interface. Returns false on any failure and true on success.
913 Optional<sv::InterfaceOp>
914 GrandCentralPass::traverseBundle(AugmentedBundleTypeAttr bundle, IntegerAttr id,
915  StringAttr prefix, VerbatimBuilder &path) {
916  auto builder = OpBuilder::atBlockEnd(getOperation().getBody());
917  sv::InterfaceOp iface;
918  builder.setInsertionPointToEnd(getOperation().getBody());
919  auto loc = getOperation().getLoc();
920  auto iFaceName = getNamespace().newName(getInterfaceName(prefix, bundle));
921  iface = builder.create<sv::InterfaceOp>(loc, iFaceName);
922  if (maybeExtractInfo)
923  iface->setAttr("output_file",
924  hw::OutputFileAttr::getFromDirectoryAndFilename(
925  &getContext(), getOutputDirectory().getValue(),
926  iFaceName + ".sv",
927  /*excludFromFileList=*/true));
928 
929  builder.setInsertionPointToEnd(cast<sv::InterfaceOp>(iface).getBody());
930 
931  for (auto element : bundle.getElements()) {
932  auto field = fromAttr(element);
933  if (!field)
934  return None;
935 
936  auto name = element.cast<DictionaryAttr>().getAs<StringAttr>("name");
937  // auto signalSym = hw::InnerRefAttr::get(iface.sym_nameAttr(), name);
938  // TODO: The `append(name.getValue())` in the following should actually be
939  // `append(signalSym)`, but this requires that `computeField` and the
940  // functions it calls always return a type for which we can construct an
941  // `InterfaceSignalOp`. Since nested interface instances are currently
942  // busted (due to the interface being a symbol table), this doesn't work at
943  // the moment. Passing a `name` works most of the time, but can be brittle
944  // if the interface field requires renaming in the output (e.g. due to
945  // naming conflicts).
946  auto elementType =
947  computeField(field.getValue(), id, prefix,
948  path.snapshot().append(".").append(name.getValue()));
949  if (!elementType)
950  return None;
951 
952  auto uloc = builder.getUnknownLoc();
953  auto description =
954  element.cast<DictionaryAttr>().getAs<StringAttr>("description");
955  if (description) {
956  auto descriptionOp = builder.create<sv::VerbatimOp>(
957  uloc, ("// " + cleanupDescription(description.getValue())));
958 
959  // If we need to generate a YAML representation of this interface, then
960  // add an attribute indicating that this `sv::VerbatimOp` is actually a
961  // description.
962  if (maybeHierarchyFileYAML)
963  descriptionOp->setAttr("firrtl.grandcentral.yaml.type",
964  builder.getStringAttr("description"));
965  }
966 
967  if (auto *str = std::get_if<VerbatimType>(&elementType.getValue())) {
968  auto instanceOp =
969  builder.create<sv::VerbatimOp>(uloc, str->toStr(name.getValue()));
970 
971  // If we need to generate a YAML representation of the interface, then add
972  // attirbutes that describe what this `sv::VerbatimOp` is.
973  if (maybeHierarchyFileYAML) {
974  if (str->instantiation)
975  instanceOp->setAttr("firrtl.grandcentral.yaml.type",
976  builder.getStringAttr("instance"));
977  else
978  instanceOp->setAttr("firrtl.grandcentral.yaml.type",
979  builder.getStringAttr("unsupported"));
980  instanceOp->setAttr("firrtl.grandcentral.yaml.name", name);
981  instanceOp->setAttr("firrtl.grandcentral.yaml.dimensions",
982  builder.getI32ArrayAttr(str->dimensions));
983  instanceOp->setAttr(
984  "firrtl.grandcentral.yaml.symbol",
985  FlatSymbolRefAttr::get(builder.getContext(), str->str));
986  }
987  continue;
988  }
989 
990  auto tpe = std::get<Type>(elementType.getValue());
991  builder.create<sv::InterfaceSignalOp>(uloc, name.getValue(), tpe);
992  }
993 
994  interfaceMap[FlatSymbolRefAttr::get(builder.getContext(), iFaceName)] = iface;
995  return iface;
996 }

1012 /// This method contains the business logic of this pass.
1013 void GrandCentralPass::runOnOperation() {
1014  LLVM_DEBUG(llvm::dbgs() << "===- Running Grand Central Views/Interface Pass "
1015  "-----------------------------===\n");
1016 
1017  CircuitOp circuitOp = getOperation();
1018 
1019  // Look at the circuit annotaitons to do two things:
1020  //
1021  // 1. Determine extraction information (directory and filename).
1022  // 2. Populate a worklist of all annotations that encode interfaces.
1023  //
1024  // Remove annotations encoding interfaces, but leave extraction information as
1025  // this may be needed by later passes.
1026  SmallVector<Annotation> worklist;
1027  bool removalError = false;
1028  AnnotationSet::removeAnnotations(circuitOp, [&](Annotation anno) {
1029  if (anno.isClass("sifive.enterprise.grandcentral.AugmentedBundleType")) {
1030  worklist.push_back(anno);
1031  return true;
1032  }
1033  if (anno.isClass(extractGrandCentralClass)) {
1034  if (maybeExtractInfo.hasValue()) {
1035  emitCircuitError("more than one 'ExtractGrandCentralAnnotation' was "
1036  "found, but exactly one must be provided");
1037  removalError = true;
1038  return false;
1039  }
1040 
1041  auto directory = anno.getMember<StringAttr>("directory");
1042  auto filename = anno.getMember<StringAttr>("filename");
1043  if (!directory || !filename) {
1044  emitCircuitError()
1045  << "contained an invalid 'ExtractGrandCentralAnnotation' that does "
1046  "not contain 'directory' and 'filename' fields: "
1047  << anno.getDict();
1048  removalError = true;
1049  return false;
1050  }
1051 
1052  maybeExtractInfo = {directory, filename};
1053  // Do not delete this annotation. Extraction info may be needed later.
1054  return false;
1055  }
1056  if (anno.isClass("sifive.enterprise.grandcentral."
1057  "GrandCentralHierarchyFileAnnotation")) {
1058  if (maybeHierarchyFileYAML.hasValue()) {
1059  emitCircuitError("more than one 'GrandCentralHierarchyFileAnnotation' "
1060  "was found, but zero or one may be provided");
1061  removalError = true;
1062  return false;
1063  }
1064 
1065  auto filename = anno.getMember<StringAttr>("filename");
1066  if (!filename) {
1067  emitCircuitError()
1068  << "contained an invalid 'GrandCentralHierarchyFileAnnotation' "
1069  "that does not contain 'directory' and 'filename' fields: "
1070  << anno.getDict();
1071  removalError = true;
1072  return false;
1073  }
1074 
1075  maybeHierarchyFileYAML = filename;
1076  return true;
1077  }
1078  if (anno.isClass(
1079  "sifive.enterprise.grandcentral.PrefixInterfacesAnnotation")) {
1080  if (!interfacePrefix.empty()) {
1081  emitCircuitError("more than one 'PrefixInterfacesAnnotation' was "
1082  "found, but zero or one may be provided");
1083  removalError = true;
1084  return false;
1085  }
1086

1089  emitCircuitError()

1094  return false;
1095  }

1099  }
1100  return false;
1101  });
1102

1104  return signalPassFailure();

1118  else

1129  llvm::dbgs() << "\n";

1183  /// companionIDMap, (2) the parentIDMap, and (3) the leafMap.
1184  /// Annotations are removed as they are discovered and if they are not

1201  })

1207  "sifive.enterprise.grandcentral.AugmentedGroundType"))

1214  });

1217  AnnotationSet::removeAnnotations(op, [&](Annotation annotation) {
1218  if (!annotation.isClass(
1219  "sifive.enterprise.grandcentral.AugmentedGroundType"))
1220  return false;
1221  op.emitOpError()
1222  << "is marked as an interface element, but this does not make "
1223  "sense (is there a scattering bug or do you have a "
1224  "malformed hand-crafted MLIR circuit?)";
1225  removalError = true;
1226  return false;
1227  });
1228  AnnotationSet::removePortAnnotations(
1229  op, [&](unsigned i, Annotation annotation) {
1230  if (!annotation.isClass(
1231  "sifive.enterprise.grandcentral.AugmentedGroundType"))
1232  return false;
1233  op.emitOpError()
1234  << "has port '" << i
1235  << "' marked as an interface element, but this does not "
1236  "make sense (is there a scattering bug or do you have a "
1237  "malformed hand-crafted MLIR circuit?)";
1238  removalError = true;
1239  return false;
1240  });
1241  })
1242  .Case<FModuleOp>([&](FModuleOp op) {
1243  // Handle annotations on the ports.
1244  AnnotationSet::removePortAnnotations(
1245  op, [&](unsigned i, Annotation annotation) {
1246  if (!annotation.isClass(
1247  "sifive.enterprise.grandcentral.AugmentedGroundType"))
1248  return false;
1249  auto maybeID = getID(op, annotation);
