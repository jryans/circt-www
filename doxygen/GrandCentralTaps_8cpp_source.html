112 /// Return a version of `path` that skips all front instances it has in common
113 /// with `other`.
114 static InstancePath stripCommonPrefix(InstancePath path, InstancePath other) {
115  while (!path.empty() && !other.empty() && path.front() == other.front()) {
116  path = path.drop_front();
117  other = other.drop_front();
118  }
119  return path;

166 /// Parse a FIRRTL `UInt`/`SInt` literal.
167 static FailureOr<Literal> parseIntegerLiteral(MLIRContext *context,
168  StringRef literal, Location loc) {
169  auto initial = literal; // used for error reporting
170  auto consumed = [&]() {
171  return initial.take_front(initial.size() - literal.size());
172  };
173  auto bail = [&](const Twine &message) {
174  mlir::emitError(loc, "expected ")
175  << message << " in literal after `" << consumed() << "`";
176  return failure();
177  };
178  auto consume = [&](StringRef text) {
179  if (!literal.consume_front(text)) {
180  (void)bail(Twine("`") + text + "`");
181  return false;
182  }
183  return true;
184  };
185 
186  // Parse the leading keyword.
187  bool isSigned;
188  if (literal.consume_front("UInt")) {
189  isSigned = false;
190  } else if (literal.consume_front("SInt")) {
191  isSigned = true;
192  } else {
193  mlir::emitError(loc, "expected leading `UInt` or `SInt` in literal");
194  return failure();
195  }
196 
197  // Parse the optional width.
198  Optional<APInt> width = {};
199  if (literal.consume_front("<")) {
200  auto widthLiteral = literal.take_while(llvm::isDigit);
201  APInt parsedWidth;
202  if (widthLiteral.getAsInteger(10, parsedWidth))
203  return bail("integer width");
204  literal = literal.drop_front(widthLiteral.size());
205  if (!literal.consume_front(">"))
206  return bail("closing `>`");
207  width = parsedWidth;
208  }
209 
210  // Parse the opening parenthesis.
211  if (!consume("("))
212  return failure();
213 
214  // Parse the opening quotes and base specifier.
215  unsigned base = 10;
216  bool hasQuotes = false;
217  if (literal.consume_front("\"")) {
218  hasQuotes = true;
219  if (literal.consume_front("h"))
220  base = 16;
221  else if (literal.consume_front("o"))
222  base = 8;
223  else if (literal.consume_front("b"))
224  base = 2;
225  else
226  return bail("base specifier (`h`, `o`, or `b`)");
227  }
228 
229  // Parse the optional sign.
230  bool isNegative = false;
231  if (literal.consume_front("-"))
232  isNegative = true;
233  else if (literal.consume_front("+"))
234  isNegative = false;
235 
236  // Parse the actual value.
237  APInt parsedValue;
238  auto valueLiteral = literal.take_while(llvm::isHexDigit);
239  if (valueLiteral.getAsInteger(base, parsedValue))
240  return bail("integer value");
241  literal = literal.drop_front(valueLiteral.size());
242 
243  // Parse the closing quotes.
244  if (hasQuotes && !literal.consume_front("\""))
245  return bail("closing quotes");
246 
247  // Parse the closing parenthesis.
248  if (!consume(")"))
249  return failure();
250 
251  // Ensure that there's no junk afterwards.
252  if (!literal.empty()) {
253  mlir::emitError(loc, "extraneous `")
254  << literal << "` in literal after `" << consumed() << "`";
255  return failure();
256  }
257 
258  // Ensure we have a 0 bit at the top to properly hold a negative value.
259  if (parsedValue.isNegative())
260  parsedValue = parsedValue.zext(parsedValue.getBitWidth() + 1);
261  if (isNegative)
262  parsedValue = -parsedValue;
263 
264  // Ensure the width is sound.
265  int32_t saneWidth = -1;
266  if (width) {
267  saneWidth = (int32_t)width->getLimitedValue(INT32_MAX);
268  if (saneWidth != *width) {
269  mlir::emitError(loc, "width of literal `")
270  << initial << "` is too big to handle";
271  return failure();
272  }
273  parsedValue = isSigned ? parsedValue.sextOrTrunc(saneWidth)
274  : parsedValue.zextOrTrunc(saneWidth);
275  }
276 
277  // Assemble the literal.
278  Literal lit;
279  lit.type = IntType::get(context, isSigned, saneWidth);
280  lit.value = IntegerAttr::get(
281  IntegerType::get(context, parsedValue.getBitWidth(),
282  isSigned ? IntegerType::Signed : IntegerType::Unsigned),
283  parsedValue);
284  return lit;
285 }

344 void GrandCentralTapsPass::runOnOperation() {
345  auto circuitOp = getOperation();
346  LLVM_DEBUG(llvm::dbgs() << "Running the GCT Data Taps pass\n");
347 
348  // Here's a rough idea of what the Scala code is doing:
349  // - Gather the `source` of all `keys` of all `DataTapsAnnotation`s throughout
350  // the design.
351  // - Convert the sources, which are specified on modules, to the absolute
352  // paths in all instances. E.g. module M tap x will produce a.x and b.x if
353  // there are two instances a and b of module M in the design.
354  // - All data tap keys are specified on black box ports.
355  // - The code then processes every DataTapsAnnotation separately as follows
356  // (with the targeted blackbox and keys):
357  // - Check for collisions between SV keywords and key port names (skip this).
358  // - Find all instances of the blackbox, but then just pick the first one (we
359  // should probably do this for each?)
360  // - Process each key independently as follows:
361  // - Look up the absolute path of the source in the map. Ensure it exists and
362  // is unambiguous. Make it relative to the blackbox instance path.
363  // - Look up the port on the black box.
364  // - Create a hierarchical SV name and store this as an assignment to the
365  // blackbox port.
366  // - DeletedDataTapKey: skip and don't create a wiring
367  // - LiteralDataTapKey: just the literal
368  // - ReferenceDataTapKey: relative path with "." + source name
369  // - DataTapModuleSignalKey: relative path with "." + internal path
370  // - Generate a body for the blackbox module with the signal mapping
371 
372  AnnotationSet circuitAnnotations(circuitOp);
373  if (auto extractAnno =
374  circuitAnnotations.getAnnotation(extractGrandCentralClass)) {
375  auto directory = extractAnno.getMember<StringAttr>("directory");
376  if (!directory) {
377  circuitOp->emitError()
378  << "contained an invalid 'ExtractGrandCentralAnnotation' that does "
379  "not contain a 'directory' field: "
380  << extractAnno.getDict();
381  return signalPassFailure();
382  }
383  maybeExtractDirectory = directory;
384  }
385 
386  // Gather a list of extmodules that have data or mem tap annotations to be
387  // expanded.
388  SmallVector<AnnotatedExtModule, 4> modules;
389  for (auto &op : *circuitOp.getBody()) {
390  auto extModule = dyn_cast<FExtModuleOp>(&op);
391  if (!extModule)
392  continue;
393 
394  // Go through the module ports and collect the annotated ones.
395  AnnotatedExtModule result{extModule, {}, {}, {}};
396  result.filteredPortAnnos.reserve(extModule.getNumPorts());
397  for (unsigned argNum = 0, e = extModule.getNumPorts(); argNum < e;
398  ++argNum) {
399  // Go through all annotations on this port and add the data tap key and
400  // mem tap ones to the list.
401  auto annos = AnnotationSet::forPort(extModule, argNum);
402  annos.removeAnnotations([&](Annotation anno) {
403  if (anno.isClass(memTapClass, deletedKeyClass, literalKeyClass,
404  internalKeyClass) ||
405  isReferenceDataTapPortName(anno)) {
406  result.portAnnos.push_back({argNum, anno});
407  return true;
408  }
409  return false;
410  });
411  result.filteredPortAnnos.push_back(annos);
412  }
413 
414  // If there are data tap annotations on the module, which is likely the
415  // case, create a filtered array of annotations with them removed.
416  AnnotationSet annos(extModule.getOperation());
417  annos.removeAnnotations(
418  [&](Annotation anno) { return anno.isClass(dataTapsClass); });
419  result.filteredModuleAnnos = annos.getArrayAttr();
420 
421  if (!result.portAnnos.empty())
422  modules.push_back(std::move(result));
423  }
424 
425  LLVM_DEBUG({
426  for (auto m : modules) {
427  llvm::dbgs() << "Extmodule " << m.extModule.getName() << " has "
428  << m.portAnnos.size() << " port annotations\n";
429  }
430  });
431 
432  // Fast path if there's nothing to do.
433  if (modules.empty()) {
434  markAllAnalysesPreserved();
435  return;
436  }
437 
438  // Build a generator for absolute module and instance paths in the design.
439  InstancePathCache instancePaths(getAnalysis<InstanceGraph>());
440 
441  // Gather the annotated ports and operations throughout the design that we are
442  // supposed to tap in one way or another.
443  tappedPorts.clear();
444  tappedOps.clear();
445  circuitOp.walk([&](Operation *op) { gatherAnnotations(op); });
446 
447  LLVM_DEBUG({
448  llvm::dbgs() << "Tapped ports:\n";
449  for (auto it : tappedPorts)
450  llvm::dbgs() << "- " << it.first << ": "
451  << it.second.first->getAttr(SymbolTable::getSymbolAttrName())
452  << " port #" << it.second.second << "\n";
453  llvm::dbgs() << "Tapped ops:\n";
454  for (auto it : tappedOps)
455  llvm::dbgs() << "- " << it.first << ": " << *it.second << "\n";
456  });
457 
458  // Process each black box independently.
459  for (auto blackBox : modules) {
460  LLVM_DEBUG(llvm::dbgs() << "Generating impls for "
461  << blackBox.extModule.getName() << "\n");
462 
463  // As a first step, gather a list of all absolute paths to instances of
464  // this black box.
465  auto paths = instancePaths.getAbsolutePaths(blackBox.extModule);
466  LLVM_DEBUG({
467  for (auto path : paths)
468  llvm::dbgs() << "- " << path << "\n";
469  });
470 
471  // Go through the port annotations of the tap module and generate a
472  // hierarchical path for each.
473  portWiring.clear();
474  portWiring.reserve(blackBox.portAnnos.size());
475  for (auto portAnno : blackBox.portAnnos) {
476  processAnnotation(portAnno, blackBox, instancePaths);
477  }
478 
479  LLVM_DEBUG({
480  llvm::dbgs() << "- Wire up as follows:\n";
481  for (auto wiring : portWiring) {
482  llvm::dbgs() << "- Port " << wiring.portNum << ":\n";
483  for (auto path : wiring.prefices) {
484  llvm::dbgs() << " - " << path << "." << wiring.suffix << "\n";
485  }
486  }
487  });
488 
489  // Now we have an awkward mapping problem. We have multiple data tap
490  // module instances, which reference things in modules that in turn have
491  // multiple instances. This is a side-effect of how Grand Central
492  // annotates things on modules rather than instances. (However in practice
493  // these will have a one-to-one correspondence due to CHIRRTL having fully
494  // uniquified instances.) To solve this issue, create a dedicated
495  // implementation for every data tap instance, and among the possible
496  // targets for the data taps choose the one with the shortest relative
497  // path to the data tap instance.
498  ImplicitLocOpBuilder builder(blackBox.extModule->getLoc(),
499  blackBox.extModule);
500  unsigned implIdx = 0;
501  for (auto path : paths) {
502  builder.setInsertionPointAfter(blackBox.extModule);
503 
504  // Get the list of ports from the original extmodule, and update the
505  // annotations such that they no longer contain any data/mem taps.
506  auto ports = blackBox.extModule.getPorts();
507  for (auto port : llvm::zip(ports, blackBox.filteredPortAnnos)) {
508  std::get<0>(port).annotations = std::get<1>(port);
509  }
510 
511  // Create a new firrtl.module that implements the data tap.
512  auto name =
513  StringAttr::get(&getContext(), Twine(blackBox.extModule.getName()) +
514  "_impl_" + Twine(implIdx++));
515  LLVM_DEBUG(llvm::dbgs()
516  << "Implementing " << name << " ("
517  << blackBox.extModule.getName() << " for " << path << ")\n");
518  auto impl =
519  builder.create<FModuleOp>(name, ports, blackBox.filteredModuleAnnos);
520  // If extraction information was provided via an
521  // `ExtractGrandCentralAnnotation`, put the created data or memory taps
522  // inside this directory.
523  if (maybeExtractDirectory)
524  impl->setAttr("output_file",
525  hw::OutputFileAttr::getFromDirectoryAndFilename(
526  &getContext(), maybeExtractDirectory.getValue(),
527  impl.getName() + ".sv"));
528  builder.setInsertionPointToEnd(impl.getBody());
529 
530  // Connect the output ports to the appropriate tapped object.
531  for (auto port : portWiring) {
532  LLVM_DEBUG(llvm::dbgs() << "- Wiring up port " << port.portNum << "\n");
533 
534  // Handle literals. We send the literal string off to the FIRParser to
535  // translate into whatever ops are necessary. This yields a handle on
536  // value we're supposed to drive.
537  if (port.literal) {
538  LLVM_DEBUG(llvm::dbgs() << " - Connecting literal "
539  << port.literal.value << "\n");
540  auto literal =
541  builder.create<ConstantOp>(port.literal.type, port.literal.value);
542  auto arg = impl.getArgument(port.portNum);
543  builder.create<ConnectOp>(arg, literal);
544  continue;
545  }
546 
547  // Determine the shortest hierarchical prefix from this black box
548  // instance to the tapped object.
549  Optional<InstancePath> shortestPrefix;
550  for (auto prefix : port.prefices) {
551  auto relative = stripCommonPrefix(prefix, path);
552  if (!shortestPrefix.hasValue() ||
553  relative.size() < shortestPrefix->size())
554  shortestPrefix = relative;
555  }
556  if (!shortestPrefix.hasValue()) {
557  LLVM_DEBUG(llvm::dbgs() << " - Has no prefix, skipping\n");
558  continue;
559  }
560  LLVM_DEBUG(llvm::dbgs()
561  << " - Shortest prefix " << *shortestPrefix << "\n");
562 
563  // Determine the module at which the hierarchical name should start.
564  Operation *opInRootModule =
565  shortestPrefix->empty() ? path.back() : shortestPrefix->front();
566  auto rootModule = opInRootModule->getParentOfType<FModuleLike>();
567 
568  SmallVector<Attribute> symbols;
569  SmallString<128> hname;
570  auto addSymbol = [&](Attribute symbol) {
571  auto id = symbols.size();
572  symbols.push_back(symbol);
573  if (!hname.empty())
574  hname += '.';
575  ("{{" + Twine(id) + "}}").toVector(hname);
576  };
577 
578  // Concatenate the prefix into a proper full hierarchical name.
579  addSymbol(
580  FlatSymbolRefAttr::get(SymbolTable::getSymbolName(rootModule)));
581  for (auto inst : shortestPrefix.getValue())
582  addSymbol(getInnerRefTo(inst));
583  if (port.target.getOp()) {
584  if (port.target.hasPort())
585  addSymbol(
586  getInnerRefTo(port.target.getOp(), port.target.getPort()));
587  else
588  addSymbol(getInnerRefTo(port.target.getOp()));
589  }
590  if (!port.suffix.empty()) {
591  hname += '.';
592  hname += port.suffix;
593  }
594  LLVM_DEBUG(llvm::dbgs() << " - Connecting as " << hname << "\n");
595 
596  // Add a verbatim op that assigns this module port.
597  auto arg = impl.getArgument(port.portNum);
598  auto hnameExpr = builder.create<VerbatimExprOp>(
599  arg.getType().cast<FIRRTLType>(), hname, ValueRange{}, symbols);
600  builder.create<ConnectOp>(arg, hnameExpr);
601  }
602 
603  // Switch the instance from the original extmodule to this
604  // implementation. CAVEAT: If the same black box data tap module is
605  // instantiated in a parent module that itself is instantiated in
606  // different locations, this will pretty arbitrarily pick one of those
607  // locations.
608  path.back()->setAttr("moduleName", SymbolRefAttr::get(name));
609  }
610 
611  // Drop the original black box module.
612  blackBox.extModule.erase();
613  }
614 }

616 /// Gather the annotations on ports and operations into the `tappedPorts` and
617 /// `tappedOps` maps.
618 void GrandCentralTapsPass::gatherAnnotations(Operation *op) {
619  if (isa<FModuleOp, FExtModuleOp>(op)) {
620  // Handle port annotations on module/extmodule ops.
621  auto gather = [&](unsigned argNum, Annotation anno) {
622  if (isReferenceDataTapSource(anno)) {
623  gatherTap(anno, Port{op, argNum});
624  return true;
625  }
626  return false;
627  };
628  AnnotationSet::removePortAnnotations(op, gather);
629 
630  // Handle internal data taps on extmodule ops.
631  if (isa<FExtModuleOp>(op)) {
632  auto gather = [&](Annotation anno) {
633  if (anno.isClass(internalKeyClass)) {
634  gatherTap(anno, op);
635  return true;
636  }
637  return false;
638  };
639  AnnotationSet::removeAnnotations(op, gather);
640  }
641 
642  return;
643  }
644 
645  // Go through all annotations on this op and extract the interesting
646  // ones. Note that the way tap annotations are scattered to their
647  // targets, we should never see multiple values or memories annotated
648  // with the exact same annotation (hence the asserts).
649  AnnotationSet::removeAnnotations(op, [&](Annotation anno) {
650  if (anno.isClass(memTapClass) || isReferenceDataTapSource(anno)) {
651  gatherTap(anno, op);
652  return true;
653  }
654  return false;
655  });
656 }

658 void GrandCentralTapsPass::processAnnotation(AnnotatedPort &portAnno,
659  AnnotatedExtModule &blackBox,
660  InstancePathCache &instancePaths) {
661  LLVM_DEBUG(llvm::dbgs() << "- Processing port " << portAnno.portNum
662  << " anno " << portAnno.anno.getDict() << "\n");
663  auto key = getKey(portAnno.anno);
664  auto portName = blackBox.extModule.getPortNameAttr(portAnno.portNum);
665  PortWiring wiring;
666  wiring.portNum = portAnno.portNum;
667 
668  // Lookup the sibling annotation no the target. This may not exist, e.g. in
669  // the case of a `LiteralDataTapKey`, in which use the annotation on the
670  // data tap module port again.
671  auto targetAnnoIt = annos.find(key);
672  auto targetAnno =
673  targetAnnoIt != annos.end() ? targetAnnoIt->second : portAnno.anno;
674 
675  // Handle data taps on signals and ports.
676  if (targetAnno.isClass(referenceKeyClass)) {
677  // Handle ports.
678  if (auto port = tappedPorts.lookup(key)) {
679  wiring.prefices = instancePaths.getAbsolutePaths(port.first);
680  wiring.target = PortWiring::Target(port.first, port.second);
681  portWiring.push_back(std::move(wiring));
682  return;
683  }
684 
685  // Handle operations.
686  if (auto op = tappedOps.lookup(key)) {
687  // We require the target to be a wire or node, such that it gets a name
688  // during Verilog emission.
689  if (!isa<WireOp, NodeOp, RegOp, RegResetOp>(op)) {
690  auto diag = blackBox.extModule.emitError("ReferenceDataTapKey on port ")
691  << portName << " must be a wire, node, or reg";
692  diag.attachNote(op->getLoc()) << "referenced operation is here:";
693  signalPassFailure();
694  return;
695  }
696 
697  wiring.prefices =
698  instancePaths.getAbsolutePaths(op->getParentOfType<FModuleOp>());
699  wiring.target = PortWiring::Target(op);
700  portWiring.push_back(std::move(wiring));
701  return;
702  }
703 
704  // The annotation scattering must have placed this annotation on some
705  // target operation or block argument, which we should have picked up in
706  // the tapped args or ops maps.
707  blackBox.extModule.emitOpError(
708  "ReferenceDataTapKey annotation was not scattered to "
709  "an operation: ")
710  << targetAnno.getDict();
711  signalPassFailure();
712  return;
713  }
714 
715  // Handle data taps on black boxes.
716  if (targetAnno.isClass(internalKeyClass)) {
717  auto op = tappedOps.lookup(key);
718  if (!op) {
719  blackBox.extModule.emitOpError(
720  "DataTapModuleSignalKey annotation was not scattered to "
721  "an operation: ")
722  << targetAnno.getDict();
723  signalPassFailure();
724  return;
725  }
726 
727  // Extract the internal path we're supposed to append.
728  auto internalPath = targetAnno.getMember<StringAttr>("internalPath");
729  if (!internalPath) {
730  blackBox.extModule.emitError("DataTapModuleSignalKey annotation on port ")
731  << portName << " missing \"internalPath\" attribute";
732  signalPassFailure();
733  return;
734  }
735 
736  wiring.prefices = instancePaths.getAbsolutePaths(op);
737  wiring.suffix = internalPath.getValue();
738  portWiring.push_back(std::move(wiring));
739  return;
740  }
741 
742  // Handle data taps with literals.
743  if (targetAnno.isClass(literalKeyClass)) {
744  auto literal = portAnno.anno.getMember<StringAttr>("literal");
745  if (!literal) {
746  blackBox.extModule.emitError("LiteralDataTapKey annotation on port ")
747  << portName << " missing \"literal\" attribute";
748  signalPassFailure();
749  return;
750  }
751 
752  // Parse the literal.
753  auto parsed =
754  parseIntegerLiteral(blackBox.extModule.getContext(), literal.getValue(),
755  blackBox.extModule.getLoc());
756  if (failed(parsed)) {
757  blackBox.extModule.emitError("LiteralDataTapKey annotation on port ")
758  << portName << " has invalid literal \"" << literal.getValue()
759  << "\"";
760  signalPassFailure();
761  return;
762  }
763 
764  wiring.literal = *parsed;
765  portWiring.push_back(std::move(wiring));
766  return;
767  }
768 
769  // Handle memory taps.
770  if (targetAnno.isClass(memTapClass)) {
771  auto op = tappedOps.lookup(key);
772  if (!op) {
773  blackBox.extModule.emitOpError(
774  "MemTapAnnotation annotation was not scattered to "
775  "an operation: ")
776  << targetAnno.getDict();
777  signalPassFailure();
778  return;
779  }
780 
781  // Extract the memory location we're supposed to access.
782  auto word = portAnno.anno.getMember<IntegerAttr>("word");
783  if (!word) {
784  blackBox.extModule.emitError("MemTapAnnotation annotation on port ")
785  << portName << " missing \"word\" attribute";
786  signalPassFailure();
787  return;
788  }
789 
790  // Formulate a hierarchical reference into the memory.
791  // CAVEAT: This just assumes that the memory will map to something that
792  // can be indexed in the final Verilog. If the memory gets turned into
793  // an instance of some sort, we lack the information necessary to go in
794  // and access individual elements of it. This will break horribly since
795  // we generate memory impls out-of-line already, and memories coming
796  // from an external generator are even worse. This needs a special node
797  // in the IR that can properly inject the memory array on emission.
798  wiring.prefices =
799  instancePaths.getAbsolutePaths(op->getParentOfType<FModuleOp>());
800  wiring.target = PortWiring::Target(op);
801  ("Memory[" + Twine(word.getValue().getLimitedValue()) + "]")
802  .toVector(wiring.suffix);
803  portWiring.push_back(std::move(wiring));
804  return;
805  }
806 
807  // We never arrive here since the above code that populates the portAnnos
808  // list only adds annotations that we handle in one of the if statements
809  // above.
810  llvm_unreachable("portAnnos is never populated with unsupported annos");
811 }
