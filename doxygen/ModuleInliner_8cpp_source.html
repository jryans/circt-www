29 /// If this operation or any child operation has a name, add the prefix to that
30 /// operation's name.
31 static void rename(StringRef prefix, Operation *op) {
32  llvm::TypeSwitch<Operation *>(op)
33  .Case<CombMemOp, InstanceOp, MemOp, MemoryPortOp, NodeOp, RegOp,
34  RegResetOp, SeqMemOp, WireOp>([&](auto op) {
35  op.nameAttr(
36  StringAttr::get(op.getContext(), (prefix + op.name()).str()));
37  });
38  // Recursively rename any child operations.
39  for (auto &region : op->getRegions())
40  for (auto &block : region)
41  for (auto &op : block)
42  rename(prefix, &op);
43 }

45 /// Clone an operation, mapping used values and results with the mapper, and
46 /// apply the prefix to the name of the operation. This will clone to the
47 /// insert point of the builder.
48 static void cloneAndRename(StringRef prefix, OpBuilder &b,
49  BlockAndValueMapping &mapper, Operation &op) {
50  auto newOp = b.clone(op, mapper);
51  rename(prefix, newOp);

54 /// This function is used before inlining a module, to handle the conversion
55 /// between module ports and instance results. For every port in the target
56 /// module, create a wire, and assign a mapping from each module port to the
57 /// wire. When the body of the module is cloned, the value of the wire will be
58 /// used instead of the module's ports.
59 static SmallVector<Value> mapPortsToWires(StringRef prefix, OpBuilder &b,
60  BlockAndValueMapping &mapper,
61  FModuleOp target) {
62  SmallVector<Value> wires;
63  auto portInfo = target.getPorts();
64  for (unsigned i = 0, e = target.getNumPorts(); i < e; ++i) {
65  auto arg = target.getArgument(i);
66  // Get the type of the wire.
67  auto type = arg.getType().cast<FIRRTLType>();
68  auto wire = b.create<WireOp>(target.getLoc(), type,
69  (prefix + portInfo[i].getName()).str());
70  wires.push_back(wire);
71  mapper.map(arg, wire.getResult());
72  }
73  return wires;
74 }

76 /// This function is used after inlining a module, to handle the conversion
77 /// between module ports and instance results. This maps each wire to the
78 /// result of the instance operation. When future operations are cloned from
79 /// the current block, they will use the value of the wire instead of the
80 /// instance results.
81 static void mapResultsToWires(BlockAndValueMapping &mapper,
82  SmallVectorImpl<Value> &wires,
83  InstanceOp instance) {
84  for (unsigned i = 0, e = instance.getNumResults(); i < e; ++i) {
85  auto result = instance.getResult(i);
86  auto wire = wires[i];
87  mapper.map(result, wire);
88  }
89 }

91 /// Inlines, flattens, and removes dead modules in a circuit.
92 ///
93 /// The inliner works in a top down fashion, starting from the top level module,
94 /// and inlines every possible instance. With this method of recursive top-down
95 /// inlining, each operation will be cloned directly to its final location.
96 ///
97 /// The inliner uses a worklist to track which modules need to be processed.
98 /// When an instance op is not inlined, the referenced module is added to the
99 /// worklist. When the inliner is complete, it deletes every un-processed
100 /// module: either all instances of the module were inlined, or it was not
101 /// reachable from the top level module.
102 ///
103 /// During the inlining process, every cloned operation with a name must be
104 /// prefixed with the instance's name. The top-down process means that we know
105 /// the entire desired prefix when we clone an operation, and can set the name
106 /// attribute once. This means that we will not create any intermediate name
107 /// attributes (which will be interned by the compiler), and helps keep down the
108 /// total memory usage.
109 namespace {
110 class Inliner {
111 public:
112  /// Initialize the inliner to run on this circuit.
113  Inliner(CircuitOp circuit);
114 
115  /// Run the inliner.
116  void run();
117 
118 private:
119  /// Returns true if the operation is annotated to be flattened.
120  bool shouldFlatten(Operation *op);
121 
122  /// Returns true if the operation is annotated to be inlined.
123  bool shouldInline(Operation *op);
124 
125  /// Flattens a target module in to the insertion point of the builder,
126  /// renaming all operations using the prefix. This clones all operations from
127  /// the target, and does not trigger inlining on the target itself.
128  void flattenInto(StringRef prefix, OpBuilder &b, BlockAndValueMapping &mapper,
129  FModuleOp target);
130 
131  /// Inlines a target module in to the location of the build, prefixing all
132  /// operations with prefix. This clones all operations from the target, and
133  /// does not trigger inlining on the target itself.
134  void inlineInto(StringRef prefix, OpBuilder &b, BlockAndValueMapping &mapper,
135  FModuleOp target);
136 
137  /// Recursively flatten all instances in a module.
138  void flattenInstances(FModuleOp module);
139 
140  /// Inline any instances in the module which were marked for inlining.
141  void inlineInstances(FModuleOp module);
142 
143  CircuitOp circuit;
144  MLIRContext *context;
145 
146  // A symbol table with references to each module in a circuit.
147  SymbolTable symbolTable;
148 
149  /// The set of live modules. Anything not recorded in this set will be
150  /// removed by dead code elimination.
151  DenseSet<Operation *> liveModules;
152 
153  /// Worklist of modules to process for inlining or flattening.
154  SmallVector<FModuleOp, 16> worklist;
155 };
156 } // namespace

166 void Inliner::flattenInto(StringRef prefix, OpBuilder &b,
167  BlockAndValueMapping &mapper, FModuleOp target) {
168  for (auto &op : *target.getBody()) {
169  // If its not an instance op, clone it and continue.
170  auto instance = dyn_cast<InstanceOp>(op);
171  if (!instance) {
172  cloneAndRename(prefix, b, mapper, op);
173  continue;
174  }
175 
176  // If its not a regular module we can't inline it. Mark is as live.
177  auto module = symbolTable.lookup(instance.moduleName());
178  auto target = dyn_cast<FModuleOp>(module);
179  if (!target) {
180  liveModules.insert(module);
181  cloneAndRename(prefix, b, mapper, op);
182  continue;
183  }
184 
185  // Create the wire mapping for results + ports.
186  auto nestedPrefix = (prefix + instance.name() + "_").str();
187  auto wires = mapPortsToWires(nestedPrefix, b, mapper, target);
188  mapResultsToWires(mapper, wires, instance);
189 
190  // Unconditionally flatten all instance operations.
191  flattenInto(nestedPrefix, b, mapper, target);
192  }
193 }

195 void Inliner::flattenInstances(FModuleOp module) {
196  for (auto &op : llvm::make_early_inc_range(*module.getBody())) {
197  // If its not an instance op, skip it.
198  auto instance = dyn_cast<InstanceOp>(op);
199  if (!instance)
200  continue;
201 
202  // If its not a regular module we can't inline it. Mark is as live.
203  auto module = symbolTable.lookup(instance.moduleName());
204  auto target = dyn_cast<FModuleOp>(module);
205  if (!target) {
206  liveModules.insert(module);
207  continue;
208  }
209 
210  // Create the wire mapping for results + ports. We RAUW the results instead
211  // of mapping them.
212  BlockAndValueMapping mapper;
213  OpBuilder b(instance);
214  auto nestedPrefix = (instance.name() + "_").str();
215  auto wires = mapPortsToWires(nestedPrefix, b, mapper, target);
216  for (unsigned i = 0, e = instance.getNumResults(); i < e; ++i)
217  instance.getResult(i).replaceAllUsesWith(wires[i]);
218 
219  // Recursively flatten the target module.
220  flattenInto(nestedPrefix, b, mapper, target);
221 
222  // Erase the replaced instance.
223  instance.erase();
224  }

227 void Inliner::inlineInto(StringRef prefix, OpBuilder &b,
228  BlockAndValueMapping &mapper, FModuleOp target) {
229  for (auto &op : *target.getBody()) {
230  // If its not an instance op, clone it and continue.
231  auto instance = dyn_cast<InstanceOp>(op);
232  if (!instance) {
233  cloneAndRename(prefix, b, mapper, op);
234  continue;
235  }
236 
237  // If its not a regular module we can't inline it. Mark is as live.
238  auto module = symbolTable.lookup(instance.moduleName());
239  auto target = dyn_cast<FModuleOp>(module);
240  if (!target) {
241  liveModules.insert(module);
242  cloneAndRename(prefix, b, mapper, op);
243  continue;
244  }
245 
246  // If we aren't inlining the target, add it to the work list.
247  if (!shouldInline(target)) {
248  if (liveModules.insert(target).second) {
249  worklist.push_back(target);
250  }
251  cloneAndRename(prefix, b, mapper, op);
252  continue;
253  }
254 
255  // Create the wire mapping for results + ports.
256  auto nestedPrefix = (prefix + instance.name() + "_").str();
257  auto wires = mapPortsToWires(nestedPrefix, b, mapper, target);
258  mapResultsToWires(mapper, wires, instance);
259 
260  // Inline the module, it can be marked as flatten and inline.
261  if (shouldFlatten(target)) {
262  flattenInto(nestedPrefix, b, mapper, target);
263  } else {
264  inlineInto(nestedPrefix, b, mapper, target);
265  }
266  }

316 void Inliner::run() {
317  auto topModule = circuit.getMainModule();
318  // Mark the top module as live, so it doesn't get deleted.
319  liveModules.insert(topModule);
320 
321  // If the top module is not a regular module, there is nothing to do.
322  if (auto fmodule = dyn_cast<FModuleOp>(topModule))
323  worklist.push_back(fmodule);
324 
325  // If the module is marked for flattening, flatten it. Otherwise, inline
326  // every instance marked to be inlined.
327  while (!worklist.empty()) {
328  auto module = worklist.pop_back_val();
329  if (shouldFlatten(module)) {
330  flattenInstances(module);
331  continue;
332  }
333  inlineInstances(module);
334 
335  // Delete the flatten annotations. Any module with the inline annotation
336  // will be deleted, as there won't be any remaining instances of it.
337  AnnotationSet(module).removeAnnotationsWithClass(
338  "firrtl.transforms.FlattenAnnotation");
339  }
340 
341  // Delete all unreferenced modules.
342  for (auto &op : llvm::make_early_inc_range(*circuit.getBody())) {
343  if (isa<FExtModuleOp, FModuleOp>(op) && !liveModules.count(&op))
344  op.erase();
345  }
346 }
