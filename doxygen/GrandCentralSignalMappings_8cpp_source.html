31 namespace {
32 enum class MappingDirection {
33  /// The local signal forces the remote signal.
34  DriveRemote,
35  /// The local signal is driven to the value of the remote signal through a
36  /// connect.
37  ProbeRemote,
38 };
39 
40 /// The information necessary to connect a local signal in a module to a remote
41 /// value (in a different module and/or circuit).
42 struct SignalMapping {
43  /// Whether we force or read from the target.
44  MappingDirection dir;
45  /// The reference target of the thing we are forcing or probing.
46  StringAttr remoteTarget;
47  /// The type of the signal being mapped.
48  FIRRTLType type;
49  /// The block argument or result that interacts with the remote target, either
50  /// by forcing it or by reading from it through a connect.
51  Value localValue;
52  /// The name of the local value, for reuse in the generated signal mappings
53  /// module.
54  StringAttr localName;
55 };
56 
57 /// A helper structure that collects the data necessary to generate the signal
58 /// mappings module for an existing `FModuleOp` in the IR.
59 struct ModuleSignalMappings {
60  ModuleSignalMappings(FModuleOp module) : module(module) {}
61  void run();
62  void addTarget(Value value, Annotation anno);
63  FModuleOp emitMappingsModule();
64  void instantiateMappingsModule(FModuleOp mappingsModule);
65 
66  FModuleOp module;
67  bool allAnalysesPreserved = false;
68  SmallVector<SignalMapping> mappings;
69  SmallString<64> mappingsModuleName;
70 };

83 /// Analyze the `module` of this `ModuleSignalMappings` and generate the
84 /// corresponding auxiliary `FModuleOp` with the necessary cross-module
85 /// references and `ForceOp`s to probe and drive remote signals. This is
86 /// dictated by the presence of `SignalDriverAnnotation` on the module and
87 /// individual operations inside it.
88 void ModuleSignalMappings::run() {
89  // Check whether this module has any `SignalDriverAnnotation`s. These indicate
90  // whether the module contains any operations with such annotations and
91  // requires processing.
92  if (!AnnotationSet::removeAnnotations(module, signalDriverAnnoClass)) {
93  LLVM_DEBUG(llvm::dbgs() << "Skipping `" << module.getName()
94  << "` (has no annotations)\n");
95  allAnalysesPreserved = true;
96  return;
97  }
98  LLVM_DEBUG(llvm::dbgs() << "Running on module `" << module.getName()
99  << "`\n");
100 
101  // Gather the signal driver annotations on the ports of this module.
102  LLVM_DEBUG(llvm::dbgs() << "- Gather port annotations\n");
103  AnnotationSet::removePortAnnotations(
104  module, [&](unsigned i, Annotation anno) {
105  if (!anno.isClass(signalDriverAnnoClass))
106  return false;
107  addTarget(module.getArgument(i), anno);
108  return true;
109  });
110 
111  // Gather the signal driver annotations of the operations within this module.
112  LLVM_DEBUG(llvm::dbgs() << "- Gather operation annotations\n");
113  module.walk([&](Operation *op) {
114  AnnotationSet::removeAnnotations(op, [&](Annotation anno) {
115  if (!anno.isClass(signalDriverAnnoClass))
116  return false;
117  for (auto result : op->getResults())
118  addTarget(result, anno);
119  return true;
120  });
121  });
122 
123  // Pick a name for the module that implements the signal mappings.
124  CircuitNamespace circuitNamespace(module->getParentOfType<CircuitOp>());
125  mappingsModuleName =
126  circuitNamespace.newName(Twine(module.getName()) + "_signal_mappings");
127 
128  // Generate the mappings module.
129  auto mappingsModule = emitMappingsModule();
130 
131  // Instantiate the mappings module.
132  instantiateMappingsModule(mappingsModule);
133 }

135 /// Mark a `value` inside the `module` as being the target of the
136 /// `SignalDriverAnnotation` `anno`. This generates the necessary
137 /// `SignalMapping` information and adds an entry to the `mappings` array, to be
138 /// later consumed when the mappings module is constructed.
139 void ModuleSignalMappings::addTarget(Value value, Annotation anno) {
140  // We're emitting code for the "local" side of these annotations, which
141  // sits in the sub-circuit and interacts with the main circuit on the
142  // "remote" side.
143  if (anno.getMember<StringAttr>("side").getValue() != "local")
144  return;
145 
146  SignalMapping mapping;
147  mapping.dir = anno.getMember<StringAttr>("dir").getValue() == "source"
148  ? MappingDirection::ProbeRemote
149  : MappingDirection::DriveRemote;
150  mapping.remoteTarget = anno.getMember<StringAttr>("peer");
151  mapping.localValue = value;
152  mapping.type = value.getType().cast<FIRRTLType>();
153 
154  // Guess a name for the local value. This is only for readability's sake,
155  // giving the pass a hint for picking the names of the generated module ports.
156  if (auto blockArg = value.dyn_cast<BlockArgument>()) {
157  mapping.localName = module.getPortNameAttr(blockArg.getArgNumber());
158  } else if (auto op = value.getDefiningOp()) {
159  mapping.localName = op->getAttrOfType<StringAttr>("name");
160  }
161 
162  LLVM_DEBUG(llvm::dbgs() << " - " << mapping << "\n");
163  mappings.push_back(std::move(mapping));
164 }

166 /// Create a separate mappings module that contains cross-module references and
167 /// `ForceOp`s for each entry in the `mappings` array.
168 FModuleOp ModuleSignalMappings::emitMappingsModule() {
169  LLVM_DEBUG(llvm::dbgs() << "- Generating `" << mappingsModuleName << "`\n");
170 
171  // Determine what ports this module will have, given the signal mappings we
172  // are supposed to emit.
173  SmallVector<PortInfo> ports;
174  for (auto &mapping : mappings) {
175  ports.push_back(PortInfo{mapping.localName,
176  mapping.type,
177  mapping.dir == MappingDirection::DriveRemote
178  ? Direction::In
179  : Direction::Out,
180  {},
181  module.getLoc()});
182  LLVM_DEBUG(llvm::dbgs() << " - Adding port " << mapping.localName << "\n");
183  }
184 
185  // Create the actual module.
186  ImplicitLocOpBuilder builder(module.getLoc(), module);
187  auto mappingsModule = builder.create<FModuleOp>(
188  StringAttr::get(module.getContext(), mappingsModuleName), ports);
189 
190  // Generate the connect and force statements inside the module.
191  builder.setInsertionPointToStart(mappingsModule.getBody());
192  unsigned portIdx = 0;
193  for (auto &mapping : mappings) {
194  // TODO: Actually generate a proper XMR here. For now just do some textual
195  // replacements. Generating a real IR node (like a proper XMR op) would be
196  // much better, but the modules that `EmitSignalMappings` interacts with
197  // generally live in a separate circuit. Multiple circuits are not fully
198  // supported at the moment.
199  auto circuitSplit = mapping.remoteTarget.getValue().split('|').second;
200  auto moduleSplit = circuitSplit.split('>');
201  SmallString<32> remoteXmrName(moduleSplit.first.split(':').first);
202  remoteXmrName.push_back('.');
203  for (auto c : moduleSplit.second) {
204  if (c == '[' || c == '.')
205  remoteXmrName.push_back('_');
206  else if (c != ']')
207  remoteXmrName.push_back(c);
208  }
209  if (mapping.dir == MappingDirection::DriveRemote) {
210  auto xmr = builder.create<VerbatimWireOp>(mapping.type, remoteXmrName);
211  builder.create<ForceOp>(xmr, mappingsModule.getArgument(portIdx++));
212  } else {
213  auto xmr = builder.create<VerbatimWireOp>(mapping.type, remoteXmrName);
214  builder.create<ConnectOp>(mappingsModule.getArgument(portIdx++), xmr);
215  }
216  }
217  return mappingsModule;
218 }

220 /// Instantiate the generated mappings module inside the `module` we are working
221 /// on, and generated the necessary connections to local signals.
222 void ModuleSignalMappings::instantiateMappingsModule(FModuleOp mappingsModule) {
223  LLVM_DEBUG(llvm::dbgs() << "- Instantiating `" << mappingsModuleName
224  << "`\n");
225  // Create the actual module.
226  auto builder =
227  ImplicitLocOpBuilder::atBlockEnd(module.getLoc(), module.getBody());
228  auto inst = builder.create<InstanceOp>(mappingsModule, "signal_mappings");
229 
230  // Generate the connections to and from the instance.
231  unsigned portIdx = 0;
232  for (auto &mapping : mappings) {
233  Value dst = inst.getResult(portIdx++);
234  Value src = mapping.localValue;
235  if (mapping.dir == MappingDirection::ProbeRemote)
236  std::swap(src, dst);
237  builder.create<ConnectOp>(dst, src);
238  }
239 }
