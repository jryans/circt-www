39 /// This describes the bit in a pattern, 0/1/x.
40 enum class CaseZPatternBit { Zero = 0, One = 1, Any = 2 };

42 /// Return the letter for the specified pattern bit, e.g. "0", "1", "?" or "x".
43 /// isVerilog indicates whether we should use "?" (verilog syntax) or "x" (mlir
44 /// operation syntax.
45 char getLetter(CaseZPatternBit bit, bool isVerilog);

47 // This is provides convenient access to encode and decode a pattern.
48 struct CaseZPattern {
49  IntegerAttr attr;
50 
51  // Return the number of bits in the pattern.
52  size_t getWidth() const { return attr.getValue().getBitWidth() / 2; }
53 
54  /// Return the specified bit, bit 0 is the least significant bit.
55  CaseZPatternBit getBit(size_t bitNumber) const;
56 
57  /// Return true if this pattern always matches.
58  bool isDefault() const;
59 
60  /// Get a CaseZPattern from a specified list of CaseZPatternBit. Bits are
61  /// specified in most least significant order - element zero is the least
62  /// significant bit.
63  CaseZPattern(ArrayRef<CaseZPatternBit> bits, MLIRContext *context);
64 
65  /// Get a CaseZPattern for the specified constant value.
66  CaseZPattern(const APInt &value, MLIRContext *context);
67 
68  /// Get a CaseZPattern with a correctly encoded attribute.
69  CaseZPattern(IntegerAttr attr) : attr(attr) {}
70 
71  static CaseZPattern getDefault(unsigned width, MLIRContext *context);
