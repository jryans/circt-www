26 LogicalResult Problem::insertDependence(Dependence dep) {
27  Operation *src = dep.getSource();
28  Operation *dst = dep.getDestination();
29 
30  // Fail early on invalid dependences (src == dst == null), and def-use
31  // dependences that cannot be added because the source value is not the result
32  // of an operation (e.g., a BlockArgument).
33  if (!src || !dst)
34  return failure();
35 
36  // record auxiliary dependences explicitly
37  if (dep.isAuxiliary())
38  auxDependences[dst].insert(src);
39 
40  // auto-register the endpoints
41  operations.insert(src);
42  operations.insert(dst);
43 
44  return success();
45 }

92 LogicalResult Problem::verifyPrecedence(Dependence dep) {
93  Operation *i = dep.getSource();
94  Operation *j = dep.getDestination();
95 
96  unsigned stI = *getStartTime(i);
97  unsigned latI = *getLatency(*getLinkedOperatorType(i));
98  unsigned stJ = *getStartTime(j);
99 
100  // check if i's result is available before j starts
101  if (!(stI + latI <= stJ))
102  return getContainingOp()->emitError()
103  << "Precedence violated for dependence."
104  << "\n from: " << *i << ", result available in t=" << (stI + latI)
105  << "\n to: " << *j << ", starts in t=" << stJ;
106 
107  return success();
108 }

110 LogicalResult Problem::verify() {
111  for (auto *op : getOperations())

113  return failure();

117  if (failed(verifyPrecedence(dep)))
118  return failure();
119 
120  return success();

127 LogicalResult CyclicProblem::verifyPrecedence(Dependence dep) {
128  Operation *i = dep.getSource();
129  Operation *j = dep.getDestination();
130 
131  unsigned stI = *getStartTime(i);
132  unsigned latI = *getLatency(*getLinkedOperatorType(i));
133  unsigned stJ = *getStartTime(j);
134  unsigned dist = getDistance(dep).getValueOr(0); // optional property
135  unsigned ii = *getInitiationInterval();
136 
137  // check if i's result is available before j starts (dist iterations later)
138  if (!(stI + latI <= stJ + dist * ii))
139  return getContainingOp()->emitError()
140  << "Precedence violated for dependence."
141  << "\n from: " << *i << ", result available in t=" << (stI + latI)
142  << "\n to: " << *j << ", starts in t=" << stJ
143  << "\n dist: " << dist << ", II=" << ii;
144 
145  return success();
146 }

164 LogicalResult SharedOperatorsProblem::checkLatency(OperatorType opr) {
165  if (failed(Problem::checkLatency(opr)))
166  return failure();
167 
168  auto limit = getLimit(opr);
169  if (limit && *limit > 0 && *getLatency(opr) == 0)
170  return getContainingOp()->emitError()
171  << "Limited operator type '" << opr.getValue()
172  << "' has zero latency.";
173  return success();
174 }

176 LogicalResult SharedOperatorsProblem::verifyUtilization(OperatorType opr) {
177  auto limit = getLimit(opr);
178  if (!limit)
179  return success();
180 
181  llvm::SmallDenseMap<unsigned, unsigned> nOpsPerTimeStep;
182  for (auto *op : getOperations())
183  if (opr == *getLinkedOperatorType(op))
184  ++nOpsPerTimeStep[*getStartTime(op)];
185 
186  for (auto &kv : nOpsPerTimeStep)
187  if (kv.second > *limit)
188  return getContainingOp()->emitError()
189  << "Operator type '" << opr.getValue() << "' is oversubscribed."
190  << "\n time step: " << kv.first
191  << "\n #operations: " << kv.second << "\n limit: " << *limit;
192 
193  return success();
194 }

211 LogicalResult ModuloProblem::verifyUtilization(OperatorType opr) {
212  auto limit = getLimit(opr);
213  if (!limit)
214  return success();
215 
216  unsigned ii = *getInitiationInterval();
217  llvm::SmallDenseMap<unsigned, unsigned> nOpsPerCongruenceClass;
218  for (auto *op : getOperations())
219  if (opr == *getLinkedOperatorType(op))
220  ++nOpsPerCongruenceClass[*getStartTime(op) % ii];
221 
222  for (auto &kv : nOpsPerCongruenceClass)
223  if (kv.second > *limit)
224  return getContainingOp()->emitError()
225  << "Operator type '" << opr.getValue() << "' is oversubscribed."
226  << "\n congruence class: " << kv.first
227  << "\n #operations: " << kv.second << "\n limit: " << *limit;
228 
229  return success();
230 }

284 DependenceIterator::DependenceIterator(Problem &problem, Operation *op,
285  bool end)
286  : problem(problem), op(op), operandIdx(0), auxPredIdx(0), auxPreds(nullptr),
287  dep() {
288  if (!end) {
289  if (problem.auxDependences.count(op))
290  auxPreds = &problem.auxDependences[op];
291 
292  findNextDependence();
293  }
294 }

296 void DependenceIterator::findNextDependence() {
297  // Yield dependences corresponding to values used by `op`'s operands...
298  while (operandIdx < op->getNumOperands()) {
299  dep = Dependence(&op->getOpOperand(operandIdx++));
300  Operation *src = dep.getSource();
301 
302  // ... but only if they are outgoing from operations that are registered in
303  // the scheduling problem.
304  if (src && problem.hasOperation(src))
305  return;
306  }
307 
308  // Then, yield auxiliary dependences, if present.
309  if (auxPreds && auxPredIdx < auxPreds->size()) {
310  dep = Dependence((*auxPreds)[auxPredIdx++], op);
311  return;
312  }
313 
314  // An invalid dependence signals the end of iteration.
315  dep = Dependence();
316 }
