48 /// The (local) target of an annotation, resolved.
49 struct AnnoTarget {
50  Operation *op;
51  size_t portNum;
52  unsigned fieldIdx = 0;
53  AnnoTarget(Operation *op) : op(op), portNum(~0UL) {}
54  AnnoTarget(Operation *mod, size_t portNum) : op(mod), portNum(portNum) {}
55  AnnoTarget() : op(nullptr), portNum(~0UL) {}
56  operator bool() const { return op != nullptr; }
57 
58  bool isPort() const { return op && portNum != ~0UL; }
59  bool isInstance() const { return op && isa<InstanceOp>(op); }
60  FModuleOp getModule() const {
61  if (auto mod = dyn_cast<FModuleOp>(op))
62  return mod;
63  return op->getParentOfType<FModuleOp>();
64  }
65  FIRRTLType getType() const {
66  if (!op)
67  return FIRRTLType();
68  if (portNum != ~0UL) {
69  if (auto mod = dyn_cast<FModuleLike>(op))
70  return mod.getPortType(portNum).getSubTypeByFieldID(fieldIdx);
71  if (isa<MemOp, InstanceOp>(op))
72  return op->getResult(portNum)
73  .getType()
74  .cast<FIRRTLType>()
75  .getSubTypeByFieldID(fieldIdx);
76  llvm_unreachable("Unknown port instruction");
77  }
78  if (op->getNumResults() == 0)
79  return FIRRTLType();
80  return op->getResult(0).getType().cast<FIRRTLType>().getSubTypeByFieldID(
81  fieldIdx);
82  }
83 };

135 /// Find a matching name in an operation (usually FModuleOp). This walk could
136 /// be cached in the future. This finds a port or operation for a given name.
137 static AnnoTarget findNamedThing(StringRef name, Operation *op) {
138  AnnoTarget retval;
139  auto nameChecker = [name, &retval](Operation *op) -> WalkResult {
140  if (auto mod = dyn_cast<FModuleLike>(op)) {
141  // Check the ports.
142  auto ports = mod.getPorts();
143  for (size_t i = 0, e = ports.size(); i != e; ++i)
144  if (ports[i].name.getValue() == name) {
145  retval = AnnoTarget{op, i};
146  return WalkResult::interrupt();
147  }
148  return WalkResult::advance();
149  }
150  if (hasName(name, op)) {
151  retval = AnnoTarget{op};
152  return WalkResult::interrupt();
153  }
154  return WalkResult::advance();
155  };
156  op->walk(nameChecker);
157  return retval;
158 }

184 /// Apply a new annotation to a resolved target. This handles ports,
185 /// aggregates, modules, wires, etc.
186 static void addAnnotation(AnnoTarget ref, ArrayRef<NamedAttribute> anno) {
187  DictionaryAttr annotation;
188  if (ref.fieldIdx) {
189  SmallVector<NamedAttribute> annoField(anno.begin(), anno.end());
190  annoField.emplace_back(
191  StringAttr::get("circt.fieldID", ref.op->getContext()),
192  IntegerAttr::get(
193  IntegerType::get(ref.op->getContext(), 32, IntegerType::Signless),
194  ref.fieldIdx));
195  annotation = DictionaryAttr::get(ref.op->getContext(), annoField);
196  } else {
197  annotation = DictionaryAttr::get(ref.op->getContext(), anno);
198  }
199 
200  if (!ref.isPort()) {
201  auto newAnno = appendArrayAttr(getAnnotationsFrom(ref.op), annotation);
202  ref.op->setAttr(getAnnotationAttrName(), newAnno);
203  return;
204  }
205 
206  auto portAnnoRaw = ref.op->getAttr(getPortAnnotationAttrName());
207  ArrayAttr portAnno = portAnnoRaw.dyn_cast_or_null<ArrayAttr>();
208  if (!portAnno || portAnno.size() != getNumPorts(ref.op)) {
209  SmallVector<Attribute> emptyPortAttr(
210  getNumPorts(ref.op), ArrayAttr::get(ref.op->getContext(), {}));
211  portAnno = ArrayAttr::get(ref.op->getContext(), emptyPortAttr);
212  }
213  portAnno = replaceArrayAttrElement(
214  portAnno, ref.portNum,
215  appendArrayAttr(portAnno[ref.portNum].dyn_cast<ArrayAttr>(), annotation));
216  ref.op->setAttr("portAnnotations", portAnno);
217 }

219 // Some types have been expanded so the first layer of aggregate path is
220 // a return value.
221 static LogicalResult updateExpandedPort(StringRef field, AnnoTarget &entity) {
222  if (auto mem = dyn_cast<MemOp>(entity.op))
223  for (size_t p = 0, pe = mem.portNames().size(); p < pe; ++p)
224  if (mem.getPortNameStr(p) == field) {
225  entity.portNum = p;
226  return success();
227  }
228  if (auto inst = dyn_cast<InstanceOp>(entity.op))
229  for (size_t p = 0, pe = inst.getNumResults(); p < pe; ++p)
230  if (inst.getPortNameStr(p) == field) {
231  entity.portNum = p;
232  return success();
233  }
234  entity.op->emitError("Cannot find port with name ") << field;
235  return failure();
236 }

238 /// Try to resolve an non-array aggregate name from a target given the type and
239 /// operation of the resolved target. This needs to deal with places where we
240 /// represent bundle returns as split into constituent parts.
241 static LogicalResult updateStruct(StringRef field, AnnoTarget &entity) {
242  // The first field for some ops refers to expanded return values.
243  if (isa<MemOp, InstanceOp>(entity.op) && entity.portNum == ~0UL)
244  return updateExpandedPort(field, entity);
245 
246  auto bundle = entity.getType().dyn_cast<BundleType>();
247  if (!bundle)
248  return entity.op->emitError("field access '")
249  << field << "' into non-bundle type '" << bundle << "'";
250  if (auto idx = bundle.getElementIndex(field)) {
251  entity.fieldIdx += bundle.getFieldID(*idx);
252  return success();
253  }
254  return entity.op->emitError("cannot resolve field '")
255  << field << "' in subtype '" << bundle << "'";
256 }

258 /// Try to resolve an array index from a target given the type of the resolved
259 /// target.
260 static LogicalResult updateArray(StringRef indexStr, AnnoTarget &entity) {
261  size_t index;
262  if (indexStr.getAsInteger(10, index)) {
263  entity.op->emitError("Cannot convert '") << indexStr << "' to an integer";
264  return failure();
265  }
266 
267  auto vec = entity.getType().dyn_cast<FVectorType>();
268  if (!vec)
269  return entity.op->emitError("index access '")
270  << index << "' into non-vector type '" << vec << "'";
271  entity.fieldIdx += vec.getFieldID(index);
272  return success();
273 }

275 /// Convert a parsed target string to a resolved target structure. This
276 /// resolves all names and aggregates from a parsed target.
277 Optional<AnnoPathValue> resolveEntities(TokenAnnoTarget path,
278  ApplyState state) {
279  // Validate circuit name.
280  if (!path.circuit.empty() && state.circuit.name() != path.circuit) {
281  state.circuit->emitError("circuit name doesn't match annotation '")
282  << path.circuit << '\'';
283  return {};
284  }
285  // Circuit only target.
286  if (path.module.empty()) {
287  assert(path.name.empty() && path.instances.empty() &&
288  path.component.empty());
289  return AnnoPathValue(state.circuit);
290  }
291 
292  // Resolve all instances for non-local paths.
293  SmallVector<InstanceOp> instances;
294  for (auto p : path.instances) {
295  auto mod = state.symTbl.lookup<FModuleOp>(p.first);
296  if (!mod) {
297  state.circuit->emitError("module doesn't exist '") << p.first << '\'';
298  return {};
299  }
300  auto resolved = findNamedThing(p.second, mod);
301  if (!resolved.isInstance()) {
302  state.circuit.emitError("cannot find instance '")
303  << p.second << "' in '" << mod.getName() << "'";
304  return {};
305  }
306  instances.push_back(cast<InstanceOp>(resolved.op));
307  }
308  // The final module is where the named target is (or is the named target).
309  auto mod = state.symTbl.lookup<FModuleOp>(path.module);
310  if (!mod) {
311  state.circuit->emitError("module doesn't exist '") << path.module << '\'';
312  return {};
313  }
314  AnnoTarget ref;
315  if (path.name.empty()) {
316  assert(path.component.empty());
317  ref = AnnoTarget(mod);
318  } else {
319  ref = findNamedThing(path.name, mod);
320  if (!ref) {
321  state.circuit->emitError("cannot find name '")
322  << path.name << "' in " << mod.getName();
323  return {};
324  }
325  }
326  // If we have aggregate specifiers, resolve those now.
327  for (auto agg : path.component) {
328  if (agg.isIndex) {
329  if (failed(updateArray(agg.name, ref)))
330  return {};
331  } else {
332  if (failed(updateStruct(agg.name, ref)))
333  return {};
334  }
335  }
336  return AnnoPathValue(instances, ref);
337 }

339 /// Return an input \p target string in canonical form. This converts a Legacy
340 /// Annotation (e.g., A.B.C) into a modern annotation (e.g., ~A|B>C). Trailing
341 /// subfield/subindex references are preserved.
342 static std::string canonicalizeTarget(StringRef target) {
343 
344  if (target.empty())
345  return target.str();
346 
347  // If this is a normal Target (not a Named), erase that field in the JSON
348  // object and return that Target.
349  if (target[0] == '~')
350  return target.str();
351 
352  // This is a legacy target using the firrtl.annotations.Named type. This
353  // can be trivially canonicalized to a non-legacy target, so we do it with
354  // the following three mappings:
355  // 1. CircuitName => CircuitTarget, e.g., A -> ~A
356  // 2. ModuleName => ModuleTarget, e.g., A.B -> ~A|B
357  // 3. ComponentName => ReferenceTarget, e.g., A.B.C -> ~A|B>C
358  std::string newTarget = ("~" + target).str();
359  auto n = newTarget.find('.');
360  if (n != std::string::npos)
361  newTarget[n] = '|';
362  n = newTarget.find('.');
363  if (n != std::string::npos)
364  newTarget[n] = '>';
365  return newTarget;
366 }

368 /// split a target string into it constituent parts. This is the primary parser
369 /// for targets.
370 static Optional<TokenAnnoTarget> tokenizePath(StringRef origTarget) {
371  StringRef target = origTarget;
372  TokenAnnoTarget retval;
373  std::tie(retval.circuit, target) = target.split('|');
374  if (!retval.circuit.empty() && retval.circuit[0] == '~')
375  retval.circuit = retval.circuit.drop_front();
376  while (target.count(':')) {
377  StringRef nla;
378  std::tie(nla, target) = target.split(':');
379  StringRef inst, mod;
380  std::tie(mod, inst) = nla.split('/');
381  retval.instances.emplace_back(mod, inst);
382  }
383  // remove aggregate
384  auto targetBase =
385  target.take_until([](char c) { return c == '.' || c == '['; });
386  auto aggBase = target.drop_front(targetBase.size());
387  std::tie(retval.module, retval.name) = targetBase.split('>');
388  while (!aggBase.empty()) {
389  if (aggBase[0] == '.') {
390  aggBase = aggBase.drop_front();
391  StringRef field = aggBase.take_front(aggBase.find_first_of("[."));
392  aggBase = aggBase.drop_front(field.size());
393  retval.component.push_back({field, false});
394  } else if (aggBase[0] == '[') {
395  aggBase = aggBase.drop_front();
396  StringRef index = aggBase.take_front(aggBase.find_first_of(']'));
397  aggBase = aggBase.drop_front(index.size() + 1);
398  retval.component.push_back({index, true});
399  } else {
400  return {};
401  }
402  }
403 
404  return retval;
405 }

407 /// Make an anchor for a non-local annotation. Use the expanded path to build
408 /// the module and name list in the anchor.
409 static FlatSymbolRefAttr buildNLA(AnnoPathValue target, ApplyState state) {
410  OpBuilder b(state.circuit.getBodyRegion());
411  SmallVector<Attribute> mods;
412  SmallVector<Attribute> insts;
413  for (auto inst : target.instances) {
414  mods.push_back(FlatSymbolRefAttr::get(inst->getParentOfType<FModuleOp>()));
415  insts.push_back(StringAttr::get(state.circuit.getContext(), inst.name()));
416  }
417  mods.push_back(FlatSymbolRefAttr::get(target.ref.getModule()));
418  insts.push_back(
419  StringAttr::get(state.circuit.getContext(), getName(target.ref.op)));
420  auto modAttr = ArrayAttr::get(state.circuit.getContext(), mods);
421  auto instAttr = ArrayAttr::get(state.circuit.getContext(), insts);
422  auto nla = b.create<NonLocalAnchor>(state.circuit.getLoc(), "nla", modAttr,
423  instAttr);
424  state.symTbl.insert(nla);
425  return FlatSymbolRefAttr::get(nla);
426 }

428 /// Scatter breadcrumb annotations corresponding to non-local annotations
429 /// along the instance path. Returns symbol name used to anchor annotations to
430 /// path.
431 // FIXME: uniq annotation chain links
432 static FlatSymbolRefAttr scatterNonLocalPath(AnnoPathValue target,
433  ApplyState state) {
434 
435  FlatSymbolRefAttr sym = buildNLA(target, state);
436 
437  NamedAttrList pathmetadata;
438  pathmetadata.append("circt.nonlocal", sym);
439  pathmetadata.append(
440  "class", StringAttr::get(state.circuit.getContext(), "circt.nonlocal"));
441  for (auto item : target.instances)
442  addAnnotation(AnnoTarget(item), pathmetadata);

444  return sym;
445 }

457 /// Implementation of standard resolution. First parses the target path, then
458 /// resolves it.
459 static Optional<AnnoPathValue> stdResolveImpl(StringRef rawPath,
460  ApplyState state) {
461  auto pathStr = canonicalizeTarget(rawPath);
462  StringRef path{pathStr};
463 
464  auto tokens = tokenizePath(path);
465  if (!tokens) {
466  state.circuit->emitError("Cannot tokenize annotation path ") << rawPath;
467  return {};
468  }
469 
470  return resolveEntities(*tokens, state);
471 }

473 /// (SFC) FIRRTL SingleTargetAnnotation resolver. Uses the 'target' field of
474 /// the annotation with standard parsing to resolve the path. This requires
475 /// 'target' to exist and be normalized (per docs/FIRRTLAnnotations.md).
476 static Optional<AnnoPathValue> stdResolve(DictionaryAttr anno,
477  ApplyState state) {
478  auto target = anno.getNamed("target");
479  if (!target) {
480  state.circuit.emitError("No target field in annotation ") << anno;
481  return {};

483  if (!target->getValue().isa<StringAttr>()) {
484  state.circuit.emitError(
485  "Target field in annotation doesn't contain string ")

487  return {};
488  }
489  return stdResolveImpl(target->getValue().cast<StringAttr>().getValue(),
490  state);

507 /// An applier which puts the annotation on the target and drops the 'target'
508 /// field from the annotaiton. Optionally handles non-local annotations.
509 static LogicalResult applyWithoutTargetImpl(AnnoPathValue target,
510  DictionaryAttr anno,
511  ApplyState state,
512  bool allowNonLocal) {
513  if (!allowNonLocal && !target.isLocal())
514  return failure();
515  SmallVector<NamedAttribute> newAnnoAttrs;
516  for (auto &na : anno) {
517  if (na.getName() != "target") {
518  newAnnoAttrs.push_back(na);
519  } else if (!target.isLocal()) {
520  auto sym = scatterNonLocalPath(target, state);
521  newAnnoAttrs.push_back(
522  {StringAttr::get("circt.nonlocal", anno.getContext()), sym});
523  }
524  }
525  addAnnotation(target.ref, newAnnoAttrs);
526  return success();
527 }

529 /// An applier which puts the annotation on the target and drops the 'target'
530 /// field from the annotaiton. Optionally handles non-local annotations.
531 /// Ensures the target resolves to an expected type of operation.
532 template <bool allowNonLocal, typename T, typename... Tr>
533 static LogicalResult applyWithoutTarget(AnnoPathValue target,
534  DictionaryAttr anno, ApplyState state) {
535  if (!target.isOpOfType<T, Tr...>())
536  return failure();
537  return applyWithoutTargetImpl(target, anno, state, allowNonLocal);

598 LogicalResult LowerAnnotationsPass::applyAnnotation(DictionaryAttr anno,
599  ApplyState state) {
600  // Lookup the class
601  StringRef annoClassVal;
602  if (auto annoClass = anno.getNamed("class"))
603  annoClassVal = annoClass->getValue().cast<StringAttr>().getValue();
604  else if (ignoreClasslessAnno)
605  annoClassVal = "circt.missing";
606  else
607  return state.circuit.emitError("Annotation without a class: ") << anno;
608 
609  // See if we handle the class
610  auto record = getAnnotationHandler(annoClassVal, ignoreUnhandledAnno);
611  if (!record)
612  return state.circuit.emitWarning("Unhandled annotation: ") << anno;
613 
614  // Try to apply the annotation
615  auto target = record->resolver(anno, state);
616  if (!target)
617  return state.circuit.emitError("Unable to resolve target of annotation: ")
618  << anno;
619  if (record->applier(*target, anno, state).failed())
620  return state.circuit.emitError("Unable to apply annotation: ") << anno;
621  return success();
622 }

624 // This is the main entrypoint for the lowering pass.
625 void LowerAnnotationsPass::runOnOperation() {
626  CircuitOp circuit = getOperation();
627  SymbolTable modules(circuit);
