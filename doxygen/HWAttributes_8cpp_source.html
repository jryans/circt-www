41 Attribute HWDialect::parseAttribute(DialectAsmParser &p, Type type) const {
42  StringRef attrName;
43  Attribute attr;
44  if (p.parseKeyword(&attrName))
45  return Attribute();
46  auto parseResult = generatedAttributeParser(p, attrName, type, attr);
47  if (parseResult.hasValue())
48  return attr;
49 
50  // Parse "#hw.param.expr.add" as ParamExprAttr.
51  if (attrName.startswith(ParamExprAttr::getMnemonic())) {
52  auto string = attrName.drop_front(ParamExprAttr::getMnemonic().size());
53  if (string.front() == '.')
54  return parseParamExprWithOpcode(string.drop_front(), p, type);
55  }
56 
57  p.emitError(p.getNameLoc(), "Unexpected hw attribute '" + attrName + "'");
58  return {};
59 }

71 static std::string canonicalizeFilename(const Twine &directory,
72  const Twine &filename) {
73  SmallString<128> fullPath;
74 
75  // If the filename is an absolute path, we don't need the directory.
76  if (llvm::sys::path::is_absolute(filename))
77  filename.toVector(fullPath);
78  else
79  llvm::sys::path::append(fullPath, directory, filename);
80 
81  // If this is a directory target, we need to ensure it ends with a `/`
82  if (filename.isTriviallyEmpty() && !fullPath.endswith("/"))
83  fullPath += "/";
84 
85  return std::string(fullPath);
86 }

117 /// Option ::= 'excludeFromFileList' | 'includeReplicatedOp'
118 /// OutputFileAttr ::= 'output_file<' directory ',' name (',' Option)* '>'
119 Attribute OutputFileAttr::parse(AsmParser &p, Type type) {
120  StringAttr filename;
121  if (p.parseLess() || p.parseAttribute<StringAttr>(filename))
122  return Attribute();
123 
124  // Parse the additional keyword attributes. Its easier to let people specify
125  // these more than once than to detect the problem and do something about it.
126  bool excludeFromFileList = false;
127  bool includeReplicatedOps = false;
128  while (true) {
129  if (p.parseOptionalComma())
130  break;
131  if (!p.parseOptionalKeyword("excludeFromFileList"))
132  excludeFromFileList = true;
133  else if (!p.parseKeyword("includeReplicatedOps",
134  "or 'excludeFromFileList'"))
135  includeReplicatedOps = true;
136  else
137  return Attribute();
138  }
139 
140  if (p.parseGreater())
141  return Attribute();
142 
143  auto *context = p.getContext();
144 
145  return OutputFileAttr::get(context, filename,
146  BoolAttr::get(context, excludeFromFileList),
147  BoolAttr::get(context, includeReplicatedOps));
148 }

260 /// Given a binary function, if the two operands are known constant integers,
261 /// use the specified fold function to compute the result.
262 static Attribute foldBinaryOp(
263  ArrayRef<Attribute> operands,
264  llvm::function_ref<APInt(const APInt &, const APInt &)> calculate) {
265  assert(operands.size() == 2 && "binary operator always has two operands");
266  if (auto lhs = operands[0].dyn_cast<IntegerAttr>())
267  if (auto rhs = operands[1].dyn_cast<IntegerAttr>())
268  return IntegerAttr::get(lhs.getType(),
269  calculate(lhs.getValue(), rhs.getValue()));
270  return {};
271 }

282 /// This implements a < comparison for two operands to an associative operation
283 /// imposing an ordering upon them.
284 ///
285 /// The ordering provided puts more complex things to the start of the list,
286 /// from left to right:
287 /// expressions :: verbatims :: decl.refs :: constant
288 ///
289 static bool paramExprOperandSortPredicate(Attribute lhs, Attribute rhs) {
290  // Simplify the code below - we never have to care about exactly equal values.
291  if (lhs == rhs)
292  return false;
293 
294  // All expressions are "less than" a constant, since they appear on the right.
295  if (rhs.isa<IntegerAttr>()) {
296  // We don't bother to order constants w.r.t. each other since they will be
297  // folded - they can all compare equal.
298  return !lhs.isa<IntegerAttr>();
299  }
300  if (lhs.isa<IntegerAttr>())
301  return false;
302 
303  // Next up are named parameters.
304  if (auto rhsParam = rhs.dyn_cast<ParamDeclRefAttr>()) {
305  // Parameters are sorted lexically w.r.t. each other.
306  if (auto lhsParam = lhs.dyn_cast<ParamDeclRefAttr>())
307  return lhsParam.getName().getValue() < rhsParam.getName().getValue();
308  // They otherwise appear on the right of other things.
309  return true;
310  }
311  if (lhs.isa<ParamDeclRefAttr>())
312  return false;
313 
314  // Next up are verbatim parameters.
315  if (auto rhsParam = rhs.dyn_cast<ParamVerbatimAttr>()) {
316  // Verbatims are sorted lexically w.r.t. each other.
317  if (auto lhsParam = lhs.dyn_cast<ParamVerbatimAttr>())
318  return lhsParam.getValue().getValue() < rhsParam.getValue().getValue();
319  // They otherwise appear on the right of other things.
320  return true;
321  }
322  if (lhs.isa<ParamVerbatimAttr>())
323  return false;
324 
325  // The only thing left are nested expressions.
326  auto lhsExpr = lhs.cast<ParamExprAttr>(), rhsExpr = rhs.cast<ParamExprAttr>();
327  // Sort by the string form of the opcode, e.g. add, .. mul,... then xor.
328  if (lhsExpr.getOpcode() != rhsExpr.getOpcode())
329  return stringifyPEO(lhsExpr.getOpcode()) <
330  stringifyPEO(rhsExpr.getOpcode());
331 
332  // If they are the same opcode, then sort by arity: more complex to the left.
333  ArrayRef<Attribute> lhsOperands = lhsExpr.getOperands(),
334  rhsOperands = rhsExpr.getOperands();
335  if (lhsOperands.size() != rhsOperands.size())
336  return lhsOperands.size() > rhsOperands.size();
337 
338  // We know the two subexpressions are different (they'd otherwise be pointer
339  // equivalent) so just go compare all of the elements.
340  for (size_t i = 0, e = lhsOperands.size(); i != e; ++i) {
341  if (paramExprOperandSortPredicate(lhsOperands[i], rhsOperands[i]))
342  return true;
343  if (paramExprOperandSortPredicate(rhsOperands[i], lhsOperands[i]))
344  return false;
345  }
346 
347  llvm_unreachable("expressions should never be equivalent");
348  return false;
349 }

351 /// Given a fully associative variadic integer operation, constant fold any
352 /// constant operands and move them to the right. If the whole expression is
353 /// constant, then return that, otherwise update the operands list.
354 static Attribute simplifyAssocOp(
355  PEO opcode, SmallVector<Attribute, 4> &operands,
356  llvm::function_ref<APInt(const APInt &, const APInt &)> calculateFn,
357  llvm::function_ref<bool(const APInt &)> identityConstantFn,
358  llvm::function_ref<bool(const APInt &)> destructiveConstantFn = {}) {
359  auto type = operands[0].getType();
360  assert(isHWIntegerType(type));
361  if (operands.size() == 1)
362  return operands[0];
363 
364  // Flatten any of the same operation into the operand list:
365  // `(add x, (add y, z))` => `(add x, y, z)`.
366  for (size_t i = 0, e = operands.size(); i != e; ++i) {
367  if (auto subexpr = dyn_castPE(opcode, operands[i])) {
368  std::swap(operands[i], operands.back());
369  operands.pop_back();
370  --e;
371  --i;
372  operands.append(subexpr.getOperands().begin(),
373  subexpr.getOperands().end());
374  }
375  }
376 
377  // Impose an ordering on the operands, pushing subexpressions to the left and
378  // constants to the right, with verbatims and parameters in the middle - but
379  // predictably ordered w.r.t. each other.
380  llvm::stable_sort(operands, paramExprOperandSortPredicate);
381 
382  // Merge any constants, they will appear at the back of the operand list now.
383  if (operands.back().isa<IntegerAttr>()) {
384  while (operands.size() >= 2 &&
385  operands[operands.size() - 2].isa<IntegerAttr>()) {
386  APInt c1 = operands.pop_back_val().cast<IntegerAttr>().getValue();
387  APInt c2 = operands.pop_back_val().cast<IntegerAttr>().getValue();
388  auto resultConstant = IntegerAttr::get(type, calculateFn(c1, c2));
389  operands.push_back(resultConstant);
390  }
391 
392  auto resultCst = operands.back().cast<IntegerAttr>();
393 
394  // If the resulting constant is the destructive constant (e.g. `x*0`), then
395  // return it.
396  if (destructiveConstantFn && destructiveConstantFn(resultCst.getValue()))
397  return resultCst;
398 
399  // Remove the constant back to our operand list if it is the identity
400  // constant for this operator (e.g. `x*1`) and there are other operands.
401  if (identityConstantFn(resultCst.getValue()) && operands.size() != 1)
402  operands.pop_back();
403  }
404 
405  return operands.size() == 1 ? operands[0] : Attribute();
406 }

408 /// Analyze an operand to an add. If it is a multiplication by a constant (e.g.
409 /// `(a*b*42)` then split it into the non-constant and the constant portions
410 /// (e.g. `a*b` and `42`). Otherwise return the operand as the first value and
411 /// null as the second (standin for "multiplication by 1").
412 static std::pair<Attribute, Attribute> decomposeAddend(Attribute operand) {
413  if (auto mul = dyn_castPE(PEO::Mul, operand))
414  if (auto cst = mul.getOperands().back().dyn_cast<IntegerAttr>()) {
415  auto nonCst = ParamExprAttr::get(PEO::Mul, mul.getOperands().drop_back());
416  return {nonCst, cst};
417  }
418  return {operand, Attribute()};
419 }

422  return IntegerAttr::get(type, APInt(type.getIntOrFloatBitWidth(), 1));

425 static Attribute simplifyAdd(SmallVector<Attribute, 4> &operands) {
426  if (auto result = simplifyAssocOp(
427  PEO::Add, operands, [](auto a, auto b) { return a + b; },
428  /*identityCst*/ [](auto cst) { return cst.isZero(); }))
429  return result;
430 
431  // Canonicalize the add by splitting all addends into their variable and
432  // constant factors.
433  SmallVector<std::pair<Attribute, Attribute>> decomposedOperands;
434  llvm::SmallDenseSet<Attribute> nonConstantParts;
435  for (auto &op : operands) {
436  decomposedOperands.push_back(decomposeAddend(op));
437 
438  // Keep track of non-constant parts we've already seen. If we see multiple
439  // uses of the same value, then we can fold them together with a multiply.
440  // This handles things like `(a+b+a)` => `(a*2 + b)` and `(a*2 + b + a)` =>
441  // `(a*3 + b)`.
442  if (!nonConstantParts.insert(decomposedOperands.back().first).second) {
443  // The thing we multiply will be the common expression.
444  Attribute mulOperand = decomposedOperands.back().first;
445 
446  // Find the index of the first occurrence.
447  size_t i = 0;
448  while (decomposedOperands[i].first != mulOperand)
449  ++i;
450  // Remove both occurrences from the operand list.
451  operands.erase(operands.begin() + (&op - &operands[0]));
452  operands.erase(operands.begin() + i);
453 
454  auto type = mulOperand.getType();
455  auto c1 = decomposedOperands[i].second,
456  c2 = decomposedOperands.back().second;
457  // Fill in missing constant multiplicands with 1.
458  if (!c1)
459  c1 = getOneOfType(type);
460  if (!c2)
461  c2 = getOneOfType(type);
462  // Re-add the "a"*(c1+c2) expression to the operand list and
463  // re-canonicalize.
464  auto constant = ParamExprAttr::get(PEO::Add, c1, c2);
465  auto mulCst = ParamExprAttr::get(PEO::Mul, mulOperand, constant);
466  operands.push_back(mulCst);
467  return ParamExprAttr::get(PEO::Add, operands);
468  }
469  }
470 
471  return {};
472 }

474 static Attribute simplifyMul(SmallVector<Attribute, 4> &operands) {
475  if (auto result = simplifyAssocOp(
476  PEO::Mul, operands, [](auto a, auto b) { return a * b; },
477  /*identityCst*/ [](auto cst) { return cst.isOne(); },
478  /*destructiveCst*/ [](auto cst) { return cst.isZero(); }))
479  return result;
480 
481  // We always build a sum-of-products representation, so if we see an addition
482  // as a subexpr, we need to pull it out: (a+b)*c*d ==> (a*c*d + b*c*d).
483  for (size_t i = 0, e = operands.size(); i != e; ++i) {
484  if (auto subexpr = dyn_castPE(PEO::Add, operands[i])) {
485  // Pull the `c*d` operands out - it is whatever operands remain after
486  // removing the `(a+b)` term.
487  SmallVector<Attribute> mulOperands(operands.begin(), operands.end());
488  mulOperands.erase(mulOperands.begin() + i);
489 
490  // Build each add operand.
491  SmallVector<Attribute> addOperands;
492  for (auto addOperand : subexpr.getOperands()) {
493  mulOperands.push_back(addOperand);
494  addOperands.push_back(ParamExprAttr::get(PEO::Mul, mulOperands));
495  mulOperands.pop_back();
496  }
497  // Canonicalize and form the add expression.
498  return ParamExprAttr::get(PEO::Add, addOperands);
499  }
500  }
501 
502  return {};
503 }

524 static Attribute simplifyShl(SmallVector<Attribute, 4> &operands) {
525  assert(isHWIntegerType(operands[0].getType()));
526 
527  if (auto rhs = operands[1].dyn_cast<IntegerAttr>()) {
528  // Constant fold simple integers.
529  if (auto lhs = operands[0].dyn_cast<IntegerAttr>())
530  return IntegerAttr::get(lhs.getType(),
531  lhs.getValue().shl(rhs.getValue()));
532

534  // add/mul canonicalization.
535  auto rhsCst = APInt::getOneBitSet(rhs.getValue().getBitWidth(),
536  rhs.getValue().getZExtValue());
537  return ParamExprAttr::get(PEO::Mul, operands[0],
538  IntegerAttr::get(rhs.getType(), rhsCst));
539  }
540  return {};

579 /// Build a parameter expression. This automatically canonicalizes and
580 /// folds, so it may not necessarily return a ParamExprAttr.
581 Attribute ParamExprAttr::get(PEO opcode, ArrayRef<Attribute> operandsIn) {
582  assert(!operandsIn.empty() && "Cannot have expr with no operands");
583  // All operands must have the same type, which is the type of the result.
584  auto type = operandsIn.front().getType();
585  assert(llvm::all_of(operandsIn.drop_front(),
586  [&](auto op) { return op.getType() == type; }));
587 
588  SmallVector<Attribute, 4> operands(operandsIn.begin(), operandsIn.end());
589 
590  // Verify and canonicalize parameter expressions.
591  Attribute result;
592  switch (opcode) {
593  case PEO::Add:
594  result = simplifyAdd(operands);
595  break;
596  case PEO::Mul:
597  result = simplifyMul(operands);
598  break;
599  case PEO::And:
