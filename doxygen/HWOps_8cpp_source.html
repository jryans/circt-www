42 /// Check parameter specified by `value` to see if it is valid within the scope
43 /// of the specified module `module`. If not, emit an error at the location of
44 /// `usingOp` and return failure, otherwise return success. If `usingOp` is
45 /// null, then no diagnostic is generated.
46 ///
47 /// If `disallowParamRefs` is true, then parameter references are not allowed.
48 LogicalResult hw::checkParameterInContext(Attribute value, Operation *module,
49  Operation *usingOp,
50  bool disallowParamRefs) {
51  // Literals are always ok. Their types are already known to match
52  // expectations.
53  if (value.isa<IntegerAttr>() || value.isa<FloatAttr>() ||
54  value.isa<StringAttr>() || value.isa<ParamVerbatimAttr>())
55  return success();
56 
57  // Check both subexpressions of an expression.
58  if (auto expr = value.dyn_cast<ParamExprAttr>()) {
59  for (auto op : expr.getOperands())
60  if (failed(
61  checkParameterInContext(op, module, usingOp, disallowParamRefs)))
62  return failure();
63  return success();
64  }
65 
66  // Parameter references need more analysis to make sure they are valid within
67  // this module.
68  if (auto parameterRef = value.dyn_cast<ParamDeclRefAttr>()) {
69  auto nameAttr = parameterRef.getName();
70 
71  // Don't allow references to parameters from the default values of a
72  // parameter list.
73  if (disallowParamRefs) {
74  if (usingOp)
75  usingOp->emitOpError("parameter ")
76  << nameAttr << " cannot be used as a default value for a parameter";
77  return failure();
78  }
79 
80  // Find the corresponding attribute in the module.
81  for (auto param : module->getAttrOfType<ArrayAttr>("parameters")) {
82  auto paramAttr = param.cast<ParamDeclAttr>();
83  if (paramAttr.getName() != nameAttr)
84  continue;
85 
86  // If the types match then the reference is ok.
87  if (paramAttr.getType().getValue() == parameterRef.getType())
88  return success();
89 
90  if (usingOp) {
91  auto diag = usingOp->emitOpError("parameter ")
92  << nameAttr << " used with type " << parameterRef.getType()
93  << "; should have type " << paramAttr.getType().getValue();
94  diag.attachNote(module->getLoc()) << "module declared here";
95  }
96  return failure();
97  }
98 
99  if (usingOp) {
100  auto diag = usingOp->emitOpError("use of unknown parameter ") << nameAttr;
101  diag.attachNote(module->getLoc()) << "module declared here";
102  }
103  return failure();
104  }
105 
106  if (usingOp)
107  usingOp->emitOpError("invalid parameter value ") << value;
108  return failure();
109 }

301 static void buildModule(OpBuilder &builder, OperationState &result,
302  StringAttr name, const ModulePortInfo &ports,
303  ArrayAttr parameters,
304  ArrayRef<NamedAttribute> attributes,
305  StringAttr comment) {
306  using namespace mlir::function_like_impl;
307 
308  // Add an attribute for the name.
309  result.addAttribute(SymbolTable::getSymbolAttrName(), name);
310 
311  SmallVector<Attribute> argNames, resultNames;
312  SmallVector<Type, 4> argTypes, resultTypes;
313  SmallVector<Attribute> argAttrs, resultAttrs;
314  auto exportPortIdent = StringAttr::get("hw.exportPort", builder.getContext());
315 
316  for (auto elt : ports.inputs) {
317  if (elt.direction == PortDirection::INOUT && !elt.type.isa<hw::InOutType>())
318  elt.type = hw::InOutType::get(elt.type);
319  argTypes.push_back(elt.type);
320  argNames.push_back(elt.name);
321  Attribute attr;
322  if (elt.sym && !elt.sym.getValue().empty())
323  attr = builder.getDictionaryAttr(
324  {{exportPortIdent, FlatSymbolRefAttr::get(elt.sym)}});
325  else
326  attr = builder.getDictionaryAttr({});
327  argAttrs.push_back(attr);
328  }
329 
330  for (auto elt : ports.outputs) {
331  resultTypes.push_back(elt.type);
332  resultNames.push_back(elt.name);
333  Attribute attr;
334  if (elt.sym && !elt.sym.getValue().empty())
335  attr = builder.getDictionaryAttr(
336  {{exportPortIdent, FlatSymbolRefAttr::get(elt.sym)}});
337  else
338  attr = builder.getDictionaryAttr({});
339  resultAttrs.push_back(attr);
340  }
341 
342  // Allow clients to pass in null for the parameters list.
343  if (!parameters)
344  parameters = builder.getArrayAttr({});
345 
346  // Record the argument and result types as an attribute.
347  auto type = builder.getFunctionType(argTypes, resultTypes);
348  result.addAttribute(getTypeAttrName(), TypeAttr::get(type));
349  result.addAttribute("argNames", builder.getArrayAttr(argNames));
350  result.addAttribute("resultNames", builder.getArrayAttr(resultNames));
351  result.addAttribute(mlir::function_like_impl::getArgDictAttrName(),
352  builder.getArrayAttr(argAttrs));
353  result.addAttribute(mlir::function_like_impl::getResultDictAttrName(),
354  builder.getArrayAttr(resultAttrs));
355  result.addAttribute("parameters", parameters);
356  if (!comment)
357  comment = builder.getStringAttr("");
358  result.addAttribute("comment", comment);
359  result.addAttributes(attributes);
360  result.addRegion();
361 }

546 /// module-parameter-list ::= `<` parameter-decl (`,` parameter-decl)* `>`
547 /// parameter-decl ::= identifier `:` type
548 /// parameter-decl ::= identifier `:` type `=` attribute

550 static ParseResult parseOptionalParameters(OpAsmParser &parser,
551  SmallVector<Attribute> &parameters) {
552 
553  return parser.parseCommaSeparatedList(
554  OpAsmParser::Delimiter::OptionalLessGreater, [&]() {
555  std::string name;
556  Type type;
557  Attribute value;
558 
559  if (parser.parseKeywordOrString(&name) || parser.parseColonType(type))
560  return failure();

564  if (parser.parseAttribute(value, type))
565  return failure();
566  }

569  parameters.push_back(ParamDeclAttr::get(builder.getContext(),
570  builder.getStringAttr(name),

573  });

576 static ParseResult parseHWModuleOp(OpAsmParser &parser, OperationState &result,
577  ExternModKind modKind = PlainMod) {
578  using namespace mlir::function_like_impl;
579 
580  auto loc = parser.getCurrentLocation();
581 
582  SmallVector<OpAsmParser::OperandType, 4> entryArgs;
583  SmallVector<NamedAttrList, 4> argAttrs;
584  SmallVector<NamedAttrList, 4> resultAttrs;
585  SmallVector<Type, 4> argTypes;
586  SmallVector<Type, 4> resultTypes;
587  SmallVector<Attribute> parameters;
588  auto &builder = parser.getBuilder();
589 
590  // Parse the name as a symbol.
591  StringAttr nameAttr;
592  if (parser.parseSymbolName(nameAttr, SymbolTable::getSymbolAttrName(),
593  result.attributes))
594  return failure();
595 
596  FlatSymbolRefAttr kindAttr;
597  if (modKind == GenMod) {
598  if (parser.parseComma() ||
599  parser.parseAttribute(kindAttr, "generatorKind", result.attributes)) {
600  return failure();
601  }
602  }
603 
604  // Parse the function signature.
605  bool isVariadic = false;
606  SmallVector<Attribute> resultNames;
607  if (parseOptionalParameters(parser, parameters) ||
608  module_like_impl::parseModuleFunctionSignature(
609  parser, entryArgs, argTypes, argAttrs, isVariadic, resultTypes,
610  resultAttrs, resultNames) ||
611  // If function attributes are present, parse them.
612  parser.parseOptionalAttrDictWithKeyword(result.attributes))
613  return failure();
614 
615  // Record the argument and result types as an attribute. This is necessary
616  // for external modules.
617  auto type = builder.getFunctionType(argTypes, resultTypes);
618  result.addAttribute(getTypeAttrName(), TypeAttr::get(type));
619 
620  auto *context = result.getContext();
621 
622  if (hasAttribute("resultNames", result.attributes) ||
623  hasAttribute("parameters", result.attributes)) {
624  parser.emitError(
625  loc, "explicit `resultNames` / `parameters` attributes not allowed");
626  return failure();
627  }
628 
629  // Use the argument and result names if not already specified.
630  SmallVector<Attribute> argNames;
631  if (!entryArgs.empty()) {
632  for (auto &arg : entryArgs)
633  argNames.push_back(module_like_impl::getPortNameAttr(context, arg.name));
634  } else if (!argTypes.empty()) {
635  // The parser returns empty names in a special way.
636  argNames.assign(argTypes.size(), StringAttr::get(context, ""));
637  }
638 
639  // An explicit `argNames` attribute overrides the MLIR names. This is how

754 static LogicalResult verifyModuleCommon(Operation *module) {
755  assert(isAnyModule(module) &&
756  "verifier hook should only be called on modules");
757 
758  auto moduleType = getModuleType(module);
759  auto argNames = module->getAttrOfType<ArrayAttr>("argNames");
760  auto resultNames = module->getAttrOfType<ArrayAttr>("resultNames");
761  if (argNames.size() != moduleType.getNumInputs())
762  return module->emitOpError("incorrect number of argument names");
763  if (resultNames.size() != moduleType.getNumResults())
764  return module->emitOpError("incorrect number of result names");
765 
766  SmallPtrSet<Attribute, 4> paramNames;
767 
768  // Check parameter default values are sensible.
769  for (auto param : module->getAttrOfType<ArrayAttr>("parameters")) {
770  auto paramAttr = param.cast<ParamDeclAttr>();
771 
772  // Check that we don't have any redundant parameter names. These are
773  // resolved by string name: reuse of the same name would cause ambiguities.
774  if (!paramNames.insert(paramAttr.getName()).second)
775  return module->emitOpError("parameter ")
776  << paramAttr << " has the same name as a previous parameter";
777 
778  // Default values are allowed to be missing, check them if present.
779  auto value = paramAttr.getValue();
780  if (!value)
781  continue;
782 
783  if (value.getType() != paramAttr.getType().getValue())
784  return module->emitOpError("parameter ")
785  << paramAttr << " should have type "
786  << paramAttr.getType().getValue() << "; has type "
787  << value.getType();

789  // Verify that this is a valid parameter value, disallowing parameter
790  // references. We could allow parameters to refer to each other in the
791  // future with lexical ordering if there is a need.
792  if (failed(checkParameterInContext(value, module, module,
793  /*disallowParamRefs=*/true)))
794  return failure();
795  }
796  return success();
797 }

874 LogicalResult InstanceOp::verifySymbolUses(SymbolTableCollection &symbolTable) {
875  auto *module = symbolTable.lookupNearestSymbolFrom(*this, moduleNameAttr());
876  if (module == nullptr)
877  return emitError("Cannot find module definition '") << moduleName() << "'";
878 
879  // It must be some sort of module.
880  if (!isAnyModule(module))
881  return emitError("symbol reference '")
882  << moduleName() << "' isn't a module";
883 
884  // Check that input and result types are consistent with the referenced
885  // module.
886  // Emit an error message on the instance, with a note indicating which module
887  // is being referenced.
888  auto emitError =
889  [&](std::function<void(InFlightDiagnostic & diag)> fn) -> LogicalResult {
890  auto diag = emitOpError();
891  fn(diag);
892  diag.attachNote(module->getLoc()) << "module declared here";
893  return failure();
894  };
895 
896  // Make sure our port and result names match.
897  ArrayAttr argNames = argNamesAttr();
898  ArrayAttr modArgNames = module->getAttrOfType<ArrayAttr>("argNames");
899 
900  // Check operand types first.
901  auto numOperands = getOperation()->getNumOperands();
902  auto expectedOperandTypes = getModuleType(module).getInputs();
903 
904  if (expectedOperandTypes.size() != numOperands)
905  return emitError([&](auto &diag) {
906  diag << "has a wrong number of operands; expected "
907  << expectedOperandTypes.size() << " but got " << numOperands;
908  });
909 
910  if (argNames.size() != numOperands)
911  return emitError([&](auto &diag) {
912  diag << "has a wrong number of input port names; expected " << numOperands
913  << " but got " << argNames.size();
914  });
915 
916  for (size_t i = 0; i != numOperands; ++i) {
917  auto expectedType = expectedOperandTypes[i];
918  auto operandType = getOperand(i).getType();
919  if (operandType != expectedType)
920  return emitError([&](auto &diag) {
921  diag << "operand type #" << i << " must be " << expectedType
922  << ", but got " << operandType;
923  });
924 
925  if (argNames[i] != modArgNames[i])
926  return emitError([&](auto &diag) {
927  diag << "input label #" << i << " must be " << modArgNames[i]
928  << ", but got " << argNames[i];
929  });
930  }
931 
932  // Check result types and labels.
933  auto numResults = getOperation()->getNumResults();
934  auto expectedResultTypes = getModuleType(module).getResults();
935  ArrayAttr resultNames = resultNamesAttr();
936  ArrayAttr modResultNames = module->getAttrOfType<ArrayAttr>("resultNames");
937 
938  if (expectedResultTypes.size() != numResults)
939  return emitError([&](auto &diag) {
940  diag << "has a wrong number of results; expected "
941  << expectedResultTypes.size() << " but got " << numResults;
942  });
943  if (resultNames.size() != numResults)
944  return emitError([&](auto &diag) {
945  diag << "has a wrong number of results port labels; expected "
946  << numResults << " but got " << resultNames.size();
947  });
948 
949  for (size_t i = 0; i != numResults; ++i) {
950  auto expectedType = expectedResultTypes[i];
951  auto resultType = getResult(i).getType();
952  if (resultType != expectedType)
953  return emitError([&](auto &diag) {
954  diag << "result type #" << i << " must be " << expectedType
955  << ", but got " << resultType;
956  });
957 
958  if (resultNames[i] != modResultNames[i])
959  return emitError([&](auto &diag) {
960  diag << "input label #" << i << " must be " << modResultNames[i]
961  << ", but got " << resultNames[i];
962  });
963  }
964 
965  // Check parameters match up.
966  ArrayAttr parameters = this->parameters();
967  ArrayAttr modParameters = module->getAttrOfType<ArrayAttr>("parameters");
968  auto numParameters = parameters.size();
969  if (numParameters != modParameters.size())
970  return emitError([&](auto &diag) {
971  diag << "expected " << modParameters.size() << " parameters but had "
972  << numParameters;
973  });
974 
975  for (size_t i = 0; i != numParameters; ++i) {
976  auto param = parameters[i].cast<ParamDeclAttr>();
977  auto modParam = modParameters[i].cast<ParamDeclAttr>();
978 
979  auto paramName = param.getName();
980  if (paramName != modParam.getName())
981  return emitError([&](auto &diag) {
982  diag << "parameter #" << i << " should have name " << modParam.getName()
983  << " but has name " << paramName;
984  });
985 
986  if (param.getType() != modParam.getType())
987  return emitError([&](auto &diag) {
988  diag << "parameter " << paramName << " should have type "
989  << modParam.getType() << " but has type " << param.getType();
990  });
991 
992  // All instance parameters must have a value. Specify the same value as
993  // a module's default value if you want the default.
994  if (!param.getValue())
995  return emitOpError("parameter ") << paramName << " must have a value";
996  }
997 
998  return success();
999 }

1022  OperationState &result) {
1023  auto *context = result.getContext();
1024  StringAttr instanceNameAttr;
1025  StringAttr sym_nameAttr;
1026  FlatSymbolRefAttr moduleNameAttr;
1027  SmallVector<OpAsmParser::OperandType, 4> inputsOperands;
1028  SmallVector<Type> inputsTypes;
1029  SmallVector<Type> allResultTypes;
1030  SmallVector<Attribute> argNames, resultNames, parameters;
1031  auto noneType = parser.getBuilder().getType<NoneType>();
1032 
1033  if (parser.parseAttribute(instanceNameAttr, noneType, "instanceName",
1034  result.attributes))
1035  return failure();
1036 
1037  if (succeeded(parser.parseOptionalKeyword("sym"))) {
1038  // Parsing an optional symbol name doesn't fail, so no need to check the
1039  // result.
1040  (void)parser.parseOptionalSymbolName(
1041  sym_nameAttr, InnerName::getInnerNameAttrName(), result.attributes);
1042  }
1043 
1044  auto parseInputPort = [&]() -> ParseResult {
1045  std::string portName;
1046  if (parser.parseKeywordOrString(&portName))
1047  return failure();
1048  argNames.push_back(StringAttr::get(context, portName));
1049  inputsOperands.push_back({});
1050  inputsTypes.push_back({});
1051  return failure(parser.parseColon() ||
1052  parser.parseOperand(inputsOperands.back()) ||
1053  parser.parseColon() || parser.parseType(inputsTypes.back()));
1054  };
1055 
1056  auto parseResultPort = [&]() -> ParseResult {
1057  std::string portName;
1058  if (parser.parseKeywordOrString(&portName))
1059  return failure();
1060  resultNames.push_back(StringAttr::get(parser.getContext(), portName));
1061  allResultTypes.push_back({});
1062  return parser.parseColonType(allResultTypes.back());
1063  };
1064 
1065  llvm::SMLoc parametersLoc, inputsOperandsLoc;
1066  if (parser.parseAttribute(moduleNameAttr, noneType, "moduleName",
1067  result.attributes) ||
1068  parser.getCurrentLocation(&parametersLoc) ||
1069  parseOptionalParameters(parser, parameters) ||
1070  parser.getCurrentLocation(&inputsOperandsLoc) ||
1071  parser.parseCommaSeparatedList(OpAsmParser::Delimiter::Paren,
1072  parseInputPort) ||
1073  parser.resolveOperands(inputsOperands, inputsTypes, inputsOperandsLoc,
1074  result.operands) ||
1075  parser.parseArrow() ||
1076  parser.parseCommaSeparatedList(OpAsmParser::Delimiter::Paren,
1077  parseResultPort) ||
1078  parser.parseOptionalAttrDict(result.attributes)) {
1079  return failure();
1080  }
1081 
1082  result.addAttribute("argNames", parser.getBuilder().getArrayAttr(argNames));
1083  result.addAttribute("resultNames",
1084  parser.getBuilder().getArrayAttr(resultNames));
1085  result.addAttribute("parameters",
1086  parser.getBuilder().getArrayAttr(parameters));
1087  result.addTypes(allResultTypes);
1088  return success();
1089 }
