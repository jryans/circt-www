37 /// This holds the name, type, direction of a module's ports
38 struct PortInfo {
39  StringAttr name;
40  PortDirection direction;
41  Type type;
42 
43  /// This is the argument index or the result index depending on the direction.
44  /// "0" for an output means the first output, "0" for a in/inout means the
45  /// first argument.
46  size_t argNum = ~0U;
47 
48  /// The optional symbol for this port.
49  StringAttr sym = {};
50 
51  StringRef getName() const { return name.getValue(); }
52  bool isOutput() const { return direction == OUTPUT; }
53 
54  /// Return a unique numeric identifier for this port.
55  ssize_t getId() const { return isOutput() ? argNum : (-1 - argNum); };
56 };

58 /// This holds a decoded list of input/inout and output ports for a module or
59 /// instance.
60 struct ModulePortInfo {
61  explicit ModulePortInfo(ArrayRef<PortInfo> inputs, ArrayRef<PortInfo> outputs)
62  : inputs(inputs.begin(), inputs.end()),
63  outputs(outputs.begin(), outputs.end()) {}
64 
65  explicit ModulePortInfo(ArrayRef<PortInfo> mergedPorts) {
66  inputs.reserve(mergedPorts.size());
67  outputs.reserve(mergedPorts.size());
68  for (auto port : mergedPorts) {
69  if (port.isOutput())
70  outputs.push_back(port);
71  else
72  inputs.push_back(port);
73  }
74  }
75 
76  /// This contains a list of the input and inout ports.
77  SmallVector<PortInfo> inputs;
78  /// This is a list of the output ports.
79  SmallVector<PortInfo> outputs;
80 };

165 /// This stores lookup tables to make manipulating and working with the IR more
166 /// efficient. There are two phases to this object: the "building" phase in
167 /// which it is "write only" and then the "using" phase which is read-only (and
168 /// thus can be used by multiple threads). The
169 /// "freeze" method transitions between the two states.
170 class SymbolCache {
171 public:
172  class Item {
173  public:
174  Item(Operation *op) : op(op), port(~0ULL) {}
175  Item(Operation *op, size_t port) : op(op), port(port) {}
176  bool hasPort() const { return port != ~0ULL; }
177  Operation *getOp() const { return op; }
178  size_t getPort() const { return port; }
179 
180  private:
181  Operation *op;
182  size_t port;
183  };
184 
185  /// In the building phase, add symbols.
186  void addDefinition(StringAttr symbol, Operation *op) {
187  assert(!isFrozen && "cannot mutate a frozen cache");
188  symbolCache.try_emplace(symbol.getValue(), op, ~0ULL);
189  }
190 
191  // Add inner names, which might be ports
192  void addDefinition(StringAttr symbol, StringRef name, Operation *op,
193  size_t port = ~0ULL) {
194  assert(!isFrozen && "cannot mutate a frozen cache");
195  auto key = mkInnerKey(symbol.getValue(), name);
196  symbolCache.try_emplace(StringRef(key.data(), key.size()), op, port);
197  }
198 
199  /// Mark the cache as frozen, which allows it to be shared across threads.
200  void freeze() { isFrozen = true; }
201 
202  Operation *getDefinition(StringRef symbol) const {
203  assert(isFrozen && "cannot read from this cache until it is frozen");
204  auto it = symbolCache.find(symbol);
205  if (it == symbolCache.end())
206  return nullptr;
207  assert(!it->second.hasPort() && "Module names should never be ports");
208  return it->second.getOp();
209  }
210 
211  Operation *getDefinition(StringAttr symbol) const {
212  return getDefinition(symbol.getValue());
213  }
214 
215  Operation *getDefinition(FlatSymbolRefAttr symbol) const {
216  return getDefinition(symbol.getValue());
217  }
218 
219  Item getDefinition(StringRef symbol, StringRef name) const {
220  assert(isFrozen && "cannot read from this cache until it is frozen");
221  auto key = mkInnerKey(symbol, name);
222  auto it = symbolCache.find(StringRef(key.data(), key.size()));
223  return it == symbolCache.end() ? Item{nullptr, ~0ULL} : it->second;
224  }
225 
226  Item getDefinition(StringAttr symbol, StringAttr name) const {
227  return getDefinition(symbol.getValue(), name.getValue());
228  }
229 
230 private:
231  bool isFrozen = false;
232 
233  /// This stores a lookup table from symbol attribute to the operation
234  /// (hw.module, hw.instance, etc) that defines it.
235  /// TODO: It is super annoying that symbols are *defined* as StringAttr, but
236  /// are then referenced as FlatSymbolRefAttr. Why can't we have nice
237  /// pointer uniqued things?? :-(
238  llvm::StringMap<Item> symbolCache;
239 
240  // Construct a string key with embedded null. StringMapImpl::FindKey uses
241  // explicit lengths and stores keylength, rather than relying on null
242  // characters.
243  SmallVector<char> mkInnerKey(StringRef mod, StringRef name) const {
244  assert(!mod.contains(0) && !name.contains(0) &&
245  "Null character in identifier");
246  SmallVector<char> key;
247  key.append(mod.begin(), mod.end());
248  key.push_back(0);
249  key.append(name.begin(), name.end());
250  return key;
251  }
252 };
