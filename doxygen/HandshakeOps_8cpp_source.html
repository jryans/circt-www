46 // Sized Operation with Single Type (SOST).
47 // These are operation on the format:
48 // opname operands optAttrDict : dataType
49 // containing a 'size' (=operands.size()) and 'dataType' attribute.
50 // if 'explicitSize' is set, the operation is parsed as follows:
51 // opname [$size] operands opAttrDict : dataType
52 // If the datatype of the operation is "None", the operation is also added a
53 // {control = true} attribute. if 'alwaysControl' is set, the control attribute
54 // is always set.

74  SmallVectorImpl<OpAsmParser::OperandType> &operands,
75  OperationState &result, int &size, Type &type, bool explicitSize,
76  bool alwaysControl = false) {
77  if (explicitSize)

142 struct EliminateUnusedForkResultsPattern : mlir::OpRewritePattern<ForkOp> {
143  using mlir::OpRewritePattern<ForkOp>::OpRewritePattern;

145  LogicalResult matchAndRewrite(ForkOp op,
146  PatternRewriter &rewriter) const override {
147  std::set<unsigned> unusedIndexes;
148 
149  for (auto res : llvm::enumerate(op.getResults()))
150  if (res.value().getUses().empty())
151  unusedIndexes.insert(res.index());
152 
153  if (unusedIndexes.size() == 0)
154  return failure();
155 
156  // Create a new fork op, dropping the unused results.
157  rewriter.setInsertionPoint(op);
158  auto newFork =
159  rewriter.create<ForkOp>(op.getLoc(), op.getOperand(),
160  op.getNumResults() - unusedIndexes.size());
161  rewriter.updateRootInPlace(op, [&] {
162  unsigned i = 0;
163  for (auto oldRes : llvm::enumerate(op.getResults()))
164  if (unusedIndexes.count(oldRes.index()) == 0)
165  oldRes.value().replaceAllUsesWith(newFork.getResult(i++));
166  });
167  rewriter.eraseOp(op);
168  return success();
169  }
170 };

301 static LogicalResult verify(MuxOp op) {
302  unsigned numDataOperands = static_cast<int>(op.dataOperands().size());
303  if (numDataOperands < 2)
304  return op.emitError("need at least two inputs to mux");
305 
306  auto selectType = op.selectOperand().getType();
307 
308  unsigned selectBits;
309  if (auto integerType = selectType.dyn_cast<IntegerType>())
310  selectBits = integerType.getWidth();
311  else if (selectType.isIndex())
312  selectBits = IndexType::kInternalStorageBitWidth;
313  else
314  return op.emitError("unsupported type for select operand: ") << selectType;
315 
316  double maxDataOperands = std::pow(2, selectBits);
317  if (numDataOperands > maxDataOperands)
318  return op.emitError("select bitwidth was ")
319  << selectBits << ", which can mux "
320  << static_cast<int64_t>(maxDataOperands) << " operands, but found "
321  << numDataOperands << " operands";
322 
323  return success();
324 }

376 static ParseResult verifyFuncOp(handshake::FuncOp op) {
377  // If this function is external there is nothing to do.
378  if (op.isExternal())
379  return success();
380 
381  // Verify that the argument list of the function and the arg list of the
382  // entry block line up. The trait already verified that the number of
383  // arguments is the same between the signature and the block.
384  auto fnInputTypes = op.getType().getInputs();
385  Block &entryBlock = op.front();
386 
387  for (unsigned i = 0, e = entryBlock.getNumArguments(); i != e; ++i)
388  if (fnInputTypes[i] != entryBlock.getArgument(i).getType())
389  return op.emitOpError("type of entry block argument #")
390  << i << '(' << entryBlock.getArgument(i).getType()
391  << ") must match the type of the corresponding argument in "
392  << "function signature(" << fnInputTypes[i] << ')';
393 
394  // Verify that we have a name for each argument and result of this function.
395  auto verifyPortNameAttr = [&](StringRef attrName,
396  unsigned numIOs) -> LogicalResult {
397  auto portNamesAttr = op->getAttrOfType<ArrayAttr>(attrName);
398 
399  if (!portNamesAttr)
400  return op.emitOpError() << "expected attribute '" << attrName << "'.";
401 
402  auto portNames = portNamesAttr.getValue();
403  if (portNames.size() != numIOs)
404  return op.emitOpError()
405  << "attribute '" << attrName << "' has " << portNames.size()
406  << " entries but is expected to have " << numIOs << ".";
407 
408  if (llvm::any_of(portNames,
409  [&](Attribute attr) { return !attr.isa<StringAttr>(); }))
410  return op.emitOpError() << "expected all entries in attribute '"
411  << attrName << "' to be strings.";
412 
413  return success();
414  };
415  if (failed(verifyPortNameAttr("argNames", op.getNumArguments())))
416  return failure();
417  if (failed(verifyPortNameAttr("resNames", op.getNumResults())))
418  return failure();
419 
420  return success();
421 }

495 void handshake::FuncOp::resolveArgAndResNames() {
496  auto type = getType();
497  Builder builder(getContext());
498 
499  /// Generate a set of fallback names. These are used in case names are
500  /// missing from the currently set arg- and res name attributes.
501  auto fallbackArgNames = getFuncOpNames(builder, type.getInputs(), "in");
502  auto fallbackResNames = getFuncOpNames(builder, type.getResults(), "out");
503  auto argNames = getArgNames().getValue();
504  auto resNames = getResNames().getValue();
505 
506  /// Use fallback names where actual names are missing.
507  auto resolveNames = [&](auto &fallbackNames, auto &actualNames,
508  StringRef attrName) {
509  for (auto fallbackName : llvm::enumerate(fallbackNames)) {
510  if (actualNames.size() <= fallbackName.index())
511  addStringToStringArrayAttr(
512  builder, this->getOperation(), attrName,
513  fallbackName.value().template cast<StringAttr>());
514  }
515  };
516  resolveNames(fallbackArgNames, argNames, "argNames");
517  resolveNames(fallbackResNames, resNames, "resNames");
518 }

520 static ParseResult parseFuncOp(OpAsmParser &parser, OperationState &result) {
521  auto &builder = parser.getBuilder();
522  StringAttr nameAttr;
523  SmallVector<OpAsmParser::OperandType, 4> args;
524  SmallVector<Type, 4> argTypes, resTypes;
525  SmallVector<NamedAttrList, 4> argAttributes, resAttributes;
526  SmallVector<Attribute> argNames;
527 
528  // Parse signature
529  if (parser.parseSymbolName(nameAttr, SymbolTable::getSymbolAttrName(),
530  result.attributes) ||
531  parseFuncOpArgs(parser, args, argTypes, argNames, argAttributes, resTypes,
532  resAttributes))
533  return failure();
534  mlir::function_like_impl::addArgAndResultAttrs(builder, result, argAttributes,
535  resAttributes);
536 
537  // Set function type
538  result.addAttribute(
539  handshake::FuncOp::getTypeAttrName(),
540  TypeAttr::get(builder.getFunctionType(argTypes, resTypes)));
541 
542  // Parse attributes
543  if (failed(parser.parseOptionalAttrDictWithKeyword(result.attributes)))
544  return failure();
545 
546  // If argNames and resNames wasn't provided manually, infer argNames attribute
547  // from the parsed SSA names and resNames from our naming convention.
548  if (!result.attributes.get("argNames"))
549  result.addAttribute("argNames", builder.getArrayAttr(argNames));
550  if (!result.attributes.get("resNames")) {
551  auto resNames = getFuncOpNames(builder, resTypes, "out");
552  result.addAttribute("resNames", builder.getArrayAttr(resNames));
553  }
554 
555  // Parse region
556  auto *body = result.addRegion();
557  return parser.parseRegion(*body, args, argTypes);
558 }

577 LogicalResult EliminateSimpleControlMergesPattern::matchAndRewrite(
578  ControlMergeOp op, PatternRewriter &rewriter) const {
579  auto dataResult = op.getResult(0);
580  auto choiceResult = op.getResult(1);
581  auto choiceUnused = choiceResult.use_empty();
582  if (!choiceUnused && !choiceResult.hasOneUse())
583  return failure();
584 
585  Operation *choiceUser;
586  if (choiceResult.hasOneUse()) {
587  choiceUser = choiceResult.getUses().begin().getUser();
588  if (!isa<SinkOp>(choiceUser))
589  return failure();
590  }
591 
592  auto merge = rewriter.create<MergeOp>(op.getLoc(), op.dataOperands());
593 
594  for (auto &use : dataResult.getUses()) {
595  auto *user = use.getOwner();
596  rewriter.updateRootInPlace(
597  user, [&]() { user->setOperand(use.getOperandNumber(), merge); });
598  }
599 
600  if (choiceUnused) {
601  rewriter.eraseOp(op);
602  return success();
603  }
604 
605  rewriter.eraseOp(choiceUser);
606  rewriter.eraseOp(op);
607  return success();
608 }

853 static LogicalResult verifyMemoryOp(handshake::MemoryOp op) {
854  auto memrefType = op.memRefType();
855 
856  if (memrefType.getNumDynamicDims() != 0)
857  return op.emitOpError()
858  << "memref dimensions for handshake.memory must be static.";
859  if (memrefType.getShape().size() != 1)
860  return op.emitOpError() << "memref must have only a single dimension.";
861 
862  unsigned stCount = op.stCount();
863  unsigned ldCount = op.ldCount();
864  int addressCount = memrefType.getShape().size();
865 
866  auto inputType = op.inputs().getType();
867  auto outputType = op.outputs().getType();
868  Type dataType = memrefType.getElementType();
869 
870  unsigned numOperands = static_cast<int>(op.inputs().size());
871  unsigned numResults = static_cast<int>(op.outputs().size());
872  if (numOperands != (1 + addressCount) * stCount + addressCount * ldCount)
873  return op.emitOpError("number of operands ")
874  << numOperands << " does not match number expected of "
875  << 2 * stCount + ldCount << " with " << addressCount
876  << " address inputs per port";
877 
878  if (numResults != stCount + 2 * ldCount)
879  return op.emitOpError("number of results ")
880  << numResults << " does not match number expected of "
881  << stCount + 2 * ldCount << " with " << addressCount
882  << " address inputs per port";
883 
884  Type addressType = stCount > 0 ? inputType[1] : inputType[0];
885 
886  for (unsigned i = 0; i < stCount; i++) {
887  if (inputType[2 * i] != dataType)
888  return op.emitOpError("data type for store port ")
889  << i << ":" << inputType[2 * i] << " doesn't match memory type "
890  << dataType;
891  if (inputType[2 * i + 1] != addressType)
892  return op.emitOpError("address type for store port ")
893  << i << ":" << inputType[2 * i + 1]
894  << " doesn't match address type " << addressType;
895  }
896  for (unsigned i = 0; i < ldCount; i++) {
897  Type ldAddressType = inputType[2 * stCount + i];
898  if (ldAddressType != addressType)
899  return op.emitOpError("address type for load port ")
900  << i << ":" << ldAddressType << " doesn't match address type "
901  << addressType;
902  }
903  for (unsigned i = 0; i < ldCount; i++) {
904  if (outputType[i] != dataType)
905  return op.emitOpError("data type for load port ")
906  << i << ":" << outputType[i] << " doesn't match memory type "
907  << dataType;
908  }
909  for (unsigned i = 0; i < stCount; i++) {
910  Type syncType = outputType[ldCount + i];
911  if (!syncType.isa<NoneType>())
912  return op.emitOpError("data type for sync port for store port ")
913  << i << ":" << syncType << " is not 'none'";
914  }
915  for (unsigned i = 0; i < ldCount; i++) {
916  Type syncType = outputType[ldCount + stCount + i];
917  if (!syncType.isa<NoneType>())
918  return op.emitOpError("data type for sync port for load port ")
919  << i << ":" << syncType << " is not 'none'";
920  }
921 
922  return success();
923 }

985 bool handshake::MemoryOp::allocateMemory(
986  llvm::DenseMap<unsigned, unsigned> &memoryMap,
987  std::vector<std::vector<llvm::Any>> &store,
988  std::vector<double> &storeTimes) {
989  if (memoryMap.count(id()))
990  return false;
991 
992  auto type = memRefType();
993  std::vector<llvm::Any> in;
994 
995  ArrayRef<int64_t> shape = type.getShape();
996  int allocationSize = 1;
997  unsigned count = 0;
998  for (int64_t dim : shape) {
999  if (dim > 0)
1000  allocationSize *= dim;
1001  else {
1002  assert(count < in.size());
1003  allocationSize *= llvm::any_cast<APInt>(in[count++]).getSExtValue();
1004  }
1005  }
1006  unsigned ptr = store.size();
1007  store.resize(ptr + 1);
1008  storeTimes.resize(ptr + 1);
1009  store[ptr].resize(allocationSize);
1010  storeTimes[ptr] = 0.0;
1011  mlir::Type elementType = type.getElementType();
1012  int width = elementType.getIntOrFloatBitWidth();
1013  for (int i = 0; i < allocationSize; i++) {
1014  if (elementType.isa<mlir::IntegerType>()) {
1015  store[ptr][i] = APInt(width, 0);
1016  } else if (elementType.isa<mlir::FloatType>()) {
1017  store[ptr][i] = APFloat(0.0);
1018  } else {
1019  llvm_unreachable("Unknown result type!\n");
1020  }
1021  }
1022 
1023  memoryMap[id()] = ptr;
1024  return true;
1025 }
