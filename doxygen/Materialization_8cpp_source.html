33 static void replaceFirstUse(Operation *op, Value oldVal, Value newVal) {
34  for (int i = 0, e = op->getNumOperands(); i < e; ++i)
35  if (op->getOperand(i) == oldVal) {
36  op->setOperand(i, newVal);
37  break;
38  }
39  return;
40 }

45 void insertFork(Value result, bool isLazy, OpBuilder &rewriter) {
46  // Get successor operations
47  std::vector<Operation *> opsToProcess;
48  for (auto &u : result.getUses())
49  opsToProcess.push_back(u.getOwner());
50 
51  // Insert fork after op
52  rewriter.setInsertionPointAfterValue(result);
53  Operation *newOp;
54  if (isLazy)
55  newOp = rewriter.create<LazyForkOp>(result.getLoc(), result,
56  opsToProcess.size());
57  else
58  newOp =
59  rewriter.create<ForkOp>(result.getLoc(), result, opsToProcess.size());
60 
61  // Modify operands of successor
62  // opsToProcess may have multiple instances of same operand
63  // Replace uses one by one to assign different fork outputs to them
64  for (int i = 0, e = opsToProcess.size(); i < e; ++i)
65  replaceFirstUse(opsToProcess[i], result, newOp->getResult(i));
66 }

68 // Insert Fork Operation for every operation and function argument with more
69 // than one successor.
70 LogicalResult addForkOps(handshake::FuncOp f, OpBuilder &rewriter) {
71  for (Operation &op : f.getOps()) {
72  // Ignore terminators, and don't add Forks to Forks.
73  if (op.getNumSuccessors() == 0 && !isa<ForkOp>(op)) {
74  for (auto result : op.getResults()) {
75  // If there is a result and it is used more than once
76  if (!result.use_empty() && !result.hasOneUse())
77  insertFork(result, false, rewriter);
78  }
79  }
80  }
81 
82  for (auto barg : f.front().getArguments())
83  if (!barg.use_empty() && !barg.hasOneUse())
84  insertFork(barg, false, rewriter);
85 
86  return success();
87 }

89 // Create sink for every unused result
90 LogicalResult addSinkOps(handshake::FuncOp f, OpBuilder &rewriter) {
91  BlockValues liveOuts;
92 
93  for (Block &block : f) {
94  for (Operation &op : block) {
95  // Do not add sinks for unused MLIR operations which the rewriter will
96  // later remove We have already replaced these ops with their handshake
97  // equivalents
98  // TODO: should we use other indicator for op that has been erased?
99  if (isa<mlir::CondBranchOp, mlir::BranchOp, memref::LoadOp,
100  mlir::AffineReadOpInterface, mlir::AffineForOp>(op))
101  continue;
102 
103  if (op.getNumResults() == 0)
104  continue;
105 
106  for (auto result : op.getResults())
107  if (result.use_empty()) {
108  rewriter.setInsertionPointAfter(&op);
109  auto sinkOp = rewriter.create<SinkOp>(op.getLoc(), result);
110  if (result.getType().isa<NoneType>())
111  sinkOp->setAttr("control", rewriter.getBoolAttr(true));
112  }
113  }
114  }
115  return success();
116 }
