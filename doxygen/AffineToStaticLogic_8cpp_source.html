53 void AffineToStaticLogic::runOnFunction() {
54  // Get scheduling analysis for the whole function.
55  schedulingAnalysis = &getAnalysis<CyclicSchedulingAnalysis>();
56 
57  // Collect perfectly nested loops and work on them.
58  auto outerLoops = getOperation().getOps<AffineForOp>();
59  for (auto root : llvm::make_early_inc_range(outerLoops)) {
60  SmallVector<AffineForOp> nestedLoops;
61  getPerfectlyNestedLoops(nestedLoops, root);
62 
63  // Restrict to single loops to simplify things for now.
64  if (nestedLoops.size() != 1)
65  continue;
66 
67  // Populate the target operator types.
68  if (failed(populateOperatorTypes(nestedLoops)))
69  return signalPassFailure();
70 
71  // Solve the scheduling problem computed by the analysis.
72  if (failed(solveSchedulingProblem(nestedLoops)))
73  return signalPassFailure();
74 
75  // Convert the IR.
76  if (failed(createStaticLogicPipeline(nestedLoops)))
77  return signalPassFailure();
78  }
79 }

81 /// Populate the schedling problem operator types for the dialect we are
82 /// targetting. Right now, we assume Calyx, which has a standard library with
83 /// well-defined operator latencies. Ultimately, we should move this to a
84 /// dialect interface in the Scheduling dialect.
85 LogicalResult AffineToStaticLogic::populateOperatorTypes(
86  SmallVectorImpl<AffineForOp> &loopNest) {
87  // Scheduling analyis only considers the innermost loop nest for now.
88  auto forOp = loopNest.back();
89 
90  // Retrieve the cyclic scheduling problem for this loop.
91  CyclicProblem &problem = schedulingAnalysis->getProblem(forOp);
92 
93  // Load the Calyx operator library into the problem. This is a very minimal
94  // set of arithmetic and memory operators for now. This should ultimately be
95  // pulled out into some sort of dialect interface.
96  Problem::OperatorType combOpr = problem.getOrInsertOperatorType("comb");
97  problem.setLatency(combOpr, 0);
98  Problem::OperatorType seqOpr = problem.getOrInsertOperatorType("seq");
99  problem.setLatency(seqOpr, 1);
100  Problem::OperatorType mcOpr = problem.getOrInsertOperatorType("multicycle");
101  problem.setLatency(mcOpr, 3);
102 
103  Operation *unsupported;
104  WalkResult result = forOp.getBody()->walk([&](Operation *op) {
105  return TypeSwitch<Operation *, WalkResult>(op)
106  .Case<AddIOp, AffineIfOp, AffineYieldOp, mlir::ConstantOp, IndexCastOp,
107  memref::AllocaOp>([&](Operation *combOp) {
108  // Some known combinational ops.
109  problem.setLinkedOperatorType(combOp, combOpr);
110  return WalkResult::advance();
111  })
112  .Case<AffineReadOpInterface, AffineWriteOpInterface>(
113  [&](Operation *seqOp) {
114  // Some known sequential ops. In certain cases, reads may be
115  // combinational in Calyx, but taking advantage of that is left as
116  // a future enhancement.
117  problem.setLinkedOperatorType(seqOp, seqOpr);
118  return WalkResult::advance();
119  })
120  .Case<MulIOp>([&](Operation *mcOp) {
121  // Some known multi-cycle ops.
122  problem.setLinkedOperatorType(mcOp, mcOpr);
123  return WalkResult::advance();
124  })
125  .Default([&](Operation *badOp) {
126  unsupported = op;
127  return WalkResult::interrupt();
128  });
129  });
130 
131  if (result.wasInterrupted())
132  return forOp.emitError("unsupported operation ") << *unsupported;
133 
134  return success();
135 }

137 /// Solve the pre-computed scheduling problem.
138 LogicalResult AffineToStaticLogic::solveSchedulingProblem(
139  SmallVectorImpl<AffineForOp> &loopNest) {
140  // Scheduling analyis only considers the innermost loop nest for now.
141  auto forOp = loopNest.back();
142 
143  // Retrieve the cyclic scheduling problem for this loop.
144  CyclicProblem &problem = schedulingAnalysis->getProblem(forOp);
145 
146  // Optionally debug problem inputs.
147  LLVM_DEBUG(forOp.getBody()->walk<WalkOrder::PreOrder>([&](Operation *op) {
148  llvm::dbgs() << "Scheduling inputs for " << *op;
149  auto opr = problem.getLinkedOperatorType(op);
150  llvm::dbgs() << "\n opr = " << opr;
151  llvm::dbgs() << "\n latency = " << problem.getLatency(*opr);
152  for (auto dep : problem.getDependences(op))
153  if (dep.isAuxiliary())
154  llvm::dbgs() << "\n dep = { distance = " << problem.getDistance(dep)
155  << ", source = " << *dep.getSource() << " }";
156  llvm::dbgs() << "\n\n";
157  }));
158 
159  // Verify and solve the problem.
160  if (failed(problem.check()))
161  return failure();
162 
163  auto *anchor = forOp.getBody()->getTerminator();
164  if (failed(scheduleSimplex(problem, anchor)))
165  return failure();
166 
167  // Optionally debug problem outputs.
168  LLVM_DEBUG({
169  llvm::dbgs() << "Scheduled initiation interval = "
170  << problem.getInitiationInterval() << "\n\n";
171  forOp.getBody()->walk<WalkOrder::PreOrder>([&](Operation *op) {
172  llvm::dbgs() << "Scheduling outputs for " << *op;
173  llvm::dbgs() << "\n start = " << problem.getStartTime(op);
174  llvm::dbgs() << "\n\n";
175  });
176  });
177 
178  return success();
179 }

181 /// Create the pipeline op for a loop nest.
182 LogicalResult AffineToStaticLogic::createStaticLogicPipeline(
183  SmallVectorImpl<AffineForOp> &loopNest) {
184  // Scheduling analyis only considers the innermost loop nest for now.
185  auto forOp = loopNest.back();
186 
187  // Retrieve the cyclic scheduling problem for this loop.
188  CyclicProblem &problem = schedulingAnalysis->getProblem(forOp);
189 
190  auto outerLoop = loopNest.front();
191  auto innerLoop = loopNest.back();
192  ImplicitLocOpBuilder builder(outerLoop.getLoc(), outerLoop);
193 
194  // Create constants for the loop's lower and upper bounds.
195  int64_t lbValue = innerLoop.getConstantLowerBound();
196  auto lowerBound = builder.create<arith::ConstantOp>(
197  IntegerAttr::get(builder.getI64Type(), lbValue));
198  int64_t ubValue = innerLoop.getConstantUpperBound();
199  auto upperBound = builder.create<arith::ConstantOp>(
200  IntegerAttr::get(builder.getI64Type(), ubValue));
201  int64_t stepValue = innerLoop.getStep();
202  auto step = builder.create<arith::ConstantOp>(
203  IntegerAttr::get(builder.getI64Type(), stepValue));
204 
205  // Create the pipeline op, with the same result types as the inner loop. An
206  // iter arg is created for the induction variable.
207  TypeRange resultTypes = innerLoop.getResultTypes();
208 
209  auto ii =
210  builder.getI64IntegerAttr(problem.getInitiationInterval().getValue());
211 
212  SmallVector<Value> iterArgs;
213  iterArgs.push_back(lowerBound);
214  iterArgs.append(innerLoop.getIterOperands().begin(),
215  innerLoop.getIterOperands().end());
216 
217  auto pipeline = builder.create<PipelineWhileOp>(resultTypes, ii, iterArgs);
218 
219  // Create the condition, which currently just compares the induction variable
220  // to the upper bound.
221  Block &condBlock = pipeline.getCondBlock();
222  builder.setInsertionPointToStart(&condBlock);
223  auto cmpResult = builder.create<arith::CmpIOp>(
224  builder.getI1Type(), arith::CmpIPredicate::ult, condBlock.getArgument(0),
225  upperBound);
226  condBlock.getTerminator()->insertOperands(0, {cmpResult});
227 
228  // Create the first stage.
229  Block &stagesBlock = pipeline.getStagesBlock();
230  builder.setInsertionPointToStart(&stagesBlock);
231  auto stage = builder.create<PipelineStageOp>(lowerBound.getType());
232  auto &stageBlock = stage.getBodyBlock();
233  builder.setInsertionPointToStart(&stageBlock);
234 
235  // Add the induction variable increment to the first stage.
236  auto incResult =
237  builder.create<arith::AddIOp>(stagesBlock.getArgument(0), step);
238  stageBlock.getTerminator()->insertOperands(0, {incResult});
239 
240  // Add the induction variable result to the terminator iter args.
241  auto stagesTerminator =
242  cast<PipelineTerminatorOp>(stagesBlock.getTerminator());
243  stagesTerminator.iter_argsMutable().append({stage.getResult(0)});
244 
245  // Remove the loop nest from the IR.
246  for (auto loop : llvm::reverse(loopNest))
247  loop.erase();
248 
249  return success();
250 }
