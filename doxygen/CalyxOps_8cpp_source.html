59 /// Verify that the value is not a "complex" value. For example, the source
60 /// of an AssignOp should be a constant or port, e.g.
61 /// %and = comb.and %a, %b : i1
62 /// calyx.assign %port = %and, %c1_i1 ? : i1 // Incorrect
63 /// calyx.assign %port = %c1_i1, %and ? : i1 // Correct
64 /// TODO(Calyx): This is useful to verify current MLIR can be lowered to the
65 /// native compiler. Remove this when Calyx supports wire declarations.
66 /// See: https://github.com/llvm/circt/pull/1774 for context.
67 template <typename Op>
68 static LogicalResult verifyNotComplexSource(Op op) {
69  Operation *definingOp = op.src().getDefiningOp();
70  if (definingOp == nullptr)
71  // This is a port of the parent component.
72  return success();
73 
74  // Currently, we use the Combinational dialect to perform logical operations
75  // on wires, i.e. comb::AndOp, comb::OrOp, comb::XorOp.
76  if (auto dialect = definingOp->getDialect(); isa<comb::CombDialect>(dialect))
77  return op->emitOpError("has source that is not a port or constant. "
78  "Complex logic should be conducted in the guard.");
79 
80  return success();
81 }

114 /// Verifies the body of a ControlLikeOp.
115 static LogicalResult verifyControlBody(Operation *op) {
116  if (isa<SeqOp, ParOp>(op))
117  // This does not apply to sequential and parallel regions.
118  return success();
119 
120  // Some ControlLike operations have (possibly) multiple regions, e.g. IfOp.
121  for (auto &region : op->getRegions()) {
122  auto opsIt = region.getOps();
123  size_t numOperations = std::distance(opsIt.begin(), opsIt.end());
124  // A body of a ControlLike operation may have a single EnableOp within it.
125  // However, that must be the only operation.
126  // E.g. Allowed: calyx.control { calyx.enable @A }
127  // Not Allowed: calyx.control { calyx.enable @A calyx.seq { ... } }
128  bool usesEnableAsCompositionOperator =
129  numOperations > 1 && llvm::any_of(region.front(), [](auto &&bodyOp) {
130  return isa<EnableOp>(bodyOp);
131  });
132  if (usesEnableAsCompositionOperator)
133  return op->emitOpError(
134  "EnableOp is not a composition operator. It should be nested "
135  "in a control flow operation, such as \"calyx.seq\"");
136 
137  // Verify that multiple control flow operations are nested inside a single
138  // control operator. See: https://github.com/llvm/circt/issues/1723
139  size_t numControlFlowRegions =
140  llvm::count_if(opsIt, [](auto &&op) { return hasControlRegion(&op); });
141  if (numControlFlowRegions > 1)
142  return op->emitOpError(
143  "has an invalid control sequence. Multiple control flow operations "
144  "must all be nested in a single calyx.seq or calyx.par");
145  }
146  return success();
147 }

161  auto parent = op->getParentOp();

186 // Helper function for parsing a group port operation, i.e. GroupDoneOp and
187 // GroupPortOp. These may take one of two different forms:
188 // (1) %<guard> ? %<src> : i1
189 // (2) %<src> : i1
190 static ParseResult parseGroupPort(OpAsmParser &parser, OperationState &result) {
191  SmallVector<OpAsmParser::OperandType, 2> operandInfos;
192  OpAsmParser::OperandType guardOrSource;
193  if (parser.parseOperand(guardOrSource))
194  return failure();
195 
196  if (succeeded(parser.parseOptionalQuestion())) {
197  OpAsmParser::OperandType source;
198  // The guard exists.
199  if (parser.parseOperand(source))
200  return failure();
201  operandInfos.push_back(source);
202  }
203  // No matter if this is the source or guard, it should be last.
204  operandInfos.push_back(guardOrSource);
205 
206  Type type;
207  // Resolving the operands with the same type works here since the source and
208  // guard of a group port is always i1.
209  if (parser.parseColonType(type) ||
210  parser.resolveOperands(operandInfos, type, result.operands))
211  return failure();
212 
213  return success();
214 }

231 // Collapse nested control of the same type for SeqOp and ParOp, e.g.
232 // calyx.seq { calyx.seq { ... } } -> calyx.seq { ... }

234 static LogicalResult collapseControl(OpTy controlOp,
235  PatternRewriter &rewriter) {
236  static_assert(std::is_same<SeqOp, OpTy>() || std::is_same<ParOp, OpTy>(),
237  "Should be a SeqOp or ParOp.");

240  Block *controlBody = controlOp.getBody();

246  }

248  return failure();

251 /// A helper function to check whether the conditional and group (if it exists)
252 /// needs to be erased to maintain a valid state of a Calyx program. If these
253 /// have no more uses, they will be erased.
254 template <typename OpTy>
255 static void eraseControlWithGroupAndConditional(OpTy op,
256  PatternRewriter &rewriter) {
257  static_assert(std::is_same<OpTy, IfOp>() || std::is_same<OpTy, WhileOp>(),
258  "This is only applicable to WhileOp and IfOp.");
259 
260  // Save information about the operation, and erase it.
261  Value cond = op.cond();
262  Optional<StringRef> groupName = op.groupName();
263  auto component = op->template getParentOfType<ComponentOp>();
264  rewriter.eraseOp(op);
265 
266  // Clean up the attached conditional and combinational group (if it exists).
267  if (groupName.hasValue()) {
268  auto group = component.getWiresOp().template lookupSymbol<GroupInterface>(
269  *groupName);
270  if (SymbolTable::symbolKnownUseEmpty(group, component.getRegion()))
271  rewriter.eraseOp(group);
272  }
273  // Check the conditional after the Group, since it will be driven within.
274  if (!cond.isa<BlockArgument>() && cond.getDefiningOp()->use_empty())
275  rewriter.eraseOp(cond.getDefiningOp());
276 }

402 /// Parses the ports of a Calyx component signature, and adds the corresponding
403 /// port names to `attrName`.
404 static ParseResult
405 parsePortDefList(OpAsmParser &parser, OperationState &result,
406  SmallVectorImpl<OpAsmParser::OperandType> &ports,
407  SmallVectorImpl<Type> &portTypes,
408  SmallVectorImpl<NamedAttrList> &portAttrs) {
409  auto parsePort = [&]() -> ParseResult {
410  OpAsmParser::OperandType port;
411  Type portType;
412  // Expect each port to have the form `%<ssa-name> : <type>`.
413  if (parser.parseRegionArgument(port) || parser.parseColon() ||
414  parser.parseType(portType))
415  return failure();
416  ports.push_back(port);
417  portTypes.push_back(portType);
418 
419  NamedAttrList portAttr;
420  portAttrs.push_back(succeeded(parser.parseOptionalAttrDict(portAttr))
421  ? portAttr
422  : NamedAttrList());
423  return success();
424  };
425 
426  return parser.parseCommaSeparatedList(OpAsmParser::Delimiter::Paren,
427  parsePort);
428 }

430 /// Parses the signature of a Calyx component.
431 static ParseResult
432 parseComponentSignature(OpAsmParser &parser, OperationState &result,
433  SmallVectorImpl<OpAsmParser::OperandType> &ports,
434  SmallVectorImpl<Type> &portTypes) {
435  SmallVector<OpAsmParser::OperandType> inPorts, outPorts;
436  SmallVector<Type> inPortTypes, outPortTypes;
437  SmallVector<NamedAttrList> portAttributes;
438 
439  if (parsePortDefList(parser, result, inPorts, inPortTypes, portAttributes))
440  return failure();
441 
442  if (parser.parseArrow() ||
443  parsePortDefList(parser, result, outPorts, outPortTypes, portAttributes))
444  return failure();
445 
446  auto *context = parser.getBuilder().getContext();
447  // Add attribute for port names; these are currently
448  // just inferred from the SSA names of the component.
449  SmallVector<Attribute> portNames;
450  auto getPortName = [context](const auto &port) -> StringAttr {
451  StringRef name = port.name;
452  if (name.startswith("%"))
453  name = name.drop_front();
454  return StringAttr::get(context, name);
455  };
456  llvm::transform(inPorts, std::back_inserter(portNames), getPortName);
457  llvm::transform(outPorts, std::back_inserter(portNames), getPortName);
458 
459  result.addAttribute("portNames", ArrayAttr::get(context, portNames));
460  result.addAttribute(
461  "portDirections",
462  direction::packAttribute(context, inPorts.size(), outPorts.size()));
463 
464  ports.append(inPorts);
465  ports.append(outPorts);
466  portTypes.append(inPortTypes);
467  portTypes.append(outPortTypes);
468 
469  SmallVector<Attribute> portAttrs;
470  llvm::transform(portAttributes, std::back_inserter(portAttrs),
471  [&](auto attr) { return attr.getDictionary(context); });
472  result.addAttribute("portAttributes", ArrayAttr::get(context, portAttrs));
473 
474  return success();
475 }

477 static ParseResult parseComponentOp(OpAsmParser &parser,
478  OperationState &result) {
479  using namespace mlir::function_like_impl;
480 
481  StringAttr componentName;
482  if (parser.parseSymbolName(componentName, SymbolTable::getSymbolAttrName(),
483  result.attributes))
484  return failure();
485 
486  SmallVector<OpAsmParser::OperandType> ports;
487  SmallVector<Type> portTypes;
488  if (parseComponentSignature(parser, result, ports, portTypes))
489  return failure();
490 
491  // Build the component's type for FunctionLike trait. All ports are listed
492  // as arguments so they may be accessed within the component.
493  auto type =
494  parser.getBuilder().getFunctionType(portTypes, /*resultTypes=*/{});
495  result.addAttribute(ComponentOp::getTypeAttrName(), TypeAttr::get(type));
496 
497  auto *body = result.addRegion();
498  if (parser.parseRegion(*body, ports, portTypes))
499  return failure();
500 
501  if (body->empty())
502  body->push_back(new Block());
503 
504  if (parser.parseOptionalAttrDict(result.attributes))
505  return failure();
506 
507  return success();
508 }

510 /// Determines whether the given ComponentOp has all the required ports.
511 static LogicalResult hasRequiredPorts(ComponentOp op) {
512  // Get all identifiers from the component ports.
513  llvm::SmallVector<StringRef, 4> identifiers;
514  for (PortInfo &port : op.getPortInfo()) {
515  auto portIds = port.getAllIdentifiers();
516  identifiers.append(portIds.begin(), portIds.end());
517  }
518  // Sort the identifiers: a pre-condition for std::set_intersection.
519  std::sort(identifiers.begin(), identifiers.end());
520 
521  llvm::SmallVector<StringRef, 4> intersection,
522  interfacePorts{"clk", "done", "go", "reset"};
523  // Find the intersection between all identifiers and required ports.
524  std::set_intersection(interfacePorts.begin(), interfacePorts.end(),
525  identifiers.begin(), identifiers.end(),
526  std::back_inserter(intersection));
527 
528  if (intersection.size() == interfacePorts.size())
529  return success();
530 
531  SmallVector<StringRef, 4> difference;
532  std::set_difference(interfacePorts.begin(), interfacePorts.end(),
533  intersection.begin(), intersection.end(),
534  std::back_inserter(difference));
535  return op->emitOpError()
536  << "is missing the following required port attribute identifiers: "
537  << difference;
538 }

540 static LogicalResult verifyComponentOp(ComponentOp op) {
541  // Verify there is exactly one of each the wires and control operations.
542  auto wIt = op.getBody()->getOps<WiresOp>();
543  auto cIt = op.getBody()->getOps<ControlOp>();
544  if (std::distance(wIt.begin(), wIt.end()) +
545  std::distance(cIt.begin(), cIt.end()) !=
546  2)
547  return op.emitOpError(
548  "requires exactly one of each: 'calyx.wires', 'calyx.control'.");
549 
550  if (failed(hasRequiredPorts(op)))
551  return failure();
552 
553  // Verify the component actually does something: has a non-empty Control
554  // region, or continuous assignments.
555  bool hasNoControlConstructs =
556  op.getControlOp().getBody()->getOperations().empty();
557  bool hasNoAssignments = op.getWiresOp().getBody()->getOps<AssignOp>().empty();
558  if (hasNoControlConstructs && hasNoAssignments)
559  return op->emitOpError(
560  "The component currently does nothing. It needs to either have "
561  "continuous assignments in the Wires region or control constructs in "
562  "the Control region.");
563 
564  return success();
565 }

577 void ComponentOp::build(OpBuilder &builder, OperationState &result,
578  StringAttr name, ArrayRef<PortInfo> ports) {
579  using namespace mlir::function_like_impl;
580 
581  result.addAttribute(::mlir::SymbolTable::getSymbolAttrName(), name);
582 
583  std::pair<SmallVector<Type, 8>, SmallVector<Type, 8>> portIOTypes;
584  std::pair<SmallVector<Attribute, 8>, SmallVector<Attribute, 8>> portIONames;
585  std::pair<SmallVector<Attribute, 8>, SmallVector<Attribute, 8>>
586  portIOAttributes;
587  SmallVector<Direction, 8> portDirections;
588  // Avoid using llvm::partition or llvm::sort to preserve relative ordering
589  // between individual inputs and outputs.
590  for (auto &&port : ports) {
591  bool isInput = port.direction == Direction::Input;
592  (isInput ? portIOTypes.first : portIOTypes.second).push_back(port.type);
593  (isInput ? portIONames.first : portIONames.second).push_back(port.name);
594  (isInput ? portIOAttributes.first : portIOAttributes.second)
595  .push_back(port.attributes);
596  }
597  auto portTypes = concat(portIOTypes.first, portIOTypes.second);
598  auto portNames = concat(portIONames.first, portIONames.second);
599  auto portAttributes = concat(portIOAttributes.first, portIOAttributes.second);
600 
601  // Build the function type of the component.
602  auto functionType = builder.getFunctionType(portTypes, {});
603  result.addAttribute(getTypeAttrName(), TypeAttr::get(functionType));
604 
605  // Record the port names and number of input ports of the component.
606  result.addAttribute("portNames", builder.getArrayAttr(portNames));
607  result.addAttribute("portDirections",
608  direction::packAttribute(builder.getContext(),
609  portIOTypes.first.size(),
610  portIOTypes.second.size()));
611  // Record the attributes of the ports.
612  result.addAttribute("portAttributes", builder.getArrayAttr(portAttributes));
613 
614  // Create a single-blocked region.
615  Region *region = result.addRegion();
616  Block *body = new Block();
617  region->push_back(body);
618 
619  // Add all ports to the body.

709 /// Verifies the defining operation of a value is combinational.
710 static LogicalResult isCombinational(Value value, GroupInterface group) {
711  Operation *definingOp = value.getDefiningOp();
712  if (definingOp == nullptr || definingOp->hasTrait<Combinational>())
713  // This is a port of the parent component or combinational.
714  return success();
715 
716  // For now, assumes all component instances are combinational. Once
717  // combinational components are supported, this can be strictly enforced.
718  if (isa<InstanceOp>(definingOp))
719  return success();
720 
721  // Constants and logical operations are OK.
722  if (isa<comb::CombDialect, hw::HWDialect>(definingOp->getDialect()))
723  return success();
724 
725  // Reads to MemoryOp and RegisterOp are combinational. Writes are not.
726  if (auto r = dyn_cast<RegisterOp>(definingOp)) {
727  return value == r.out()
728  ? success()
729  : group->emitOpError()
730  << "with register: \"" << r.instanceName()
731  << "\" is conducting a memory store. This is not "
732  "combinational.";
733  } else if (auto m = dyn_cast<MemoryOp>(definingOp)) {
734  auto writePorts = {m.writeData(), m.writeEn()};
735  return (llvm::none_of(writePorts, [&](Value p) { return p == value; }))
736  ? success()
737  : group->emitOpError()
738  << "with memory: \"" << m.instanceName()
739  << "\" is conducting a memory store. This "
740  "is not combinational.";
741  }
742 
743  std::string portName =
744  valueName(group->getParentOfType<ComponentOp>(), value);
745  return group->emitOpError() << "with port: " << portName
746  << ". This operation is not combinational.";
747 }

783 /// Determines whether the given port is used in the group. Its use depends on
784 /// the `isDriven` value; if true, then the port should be a destination in an
785 /// AssignOp. Otherwise, it should be the source, i.e. a read.
786 static bool portIsUsedInGroup(GroupInterface group, Value port, bool isDriven) {
787  return llvm::any_of(port.getUses(), [&](auto &&use) {
788  auto assignOp = dyn_cast<AssignOp>(use.getOwner());

790  return false;

794  // This is a continuous assignment.
795  return false;

798  // (1) it is a {source, destination} of an assignment.
799  // (2) that assignment is found in the provided group.
800 
801  // If not driven, then read.
802  Value expected = isDriven ? assignOp.dest() : assignOp.src();

804  });

878 /// Verifies that certain ports of primitives are either driven or read
879 /// together.
880 static LogicalResult verifyPrimitivePortDriving(AssignOp assign,
881  GroupInterface group) {
882  Operation *destDefiningOp = assign.dest().getDefiningOp();
883  if (destDefiningOp == nullptr)
884  return success();
885  auto destCell = dyn_cast<CellInterface>(destDefiningOp);
886  if (destCell == nullptr)
887  return success();
888 
889  LogicalResult verifyWrites =
890  TypeSwitch<Operation *, LogicalResult>(destCell)
891  .Case<RegisterOp>([&](auto op) {
892  // We only want to verify this is written to if the {write enable,
893  // in} port is driven.
894  return succeeded(group.drivesAnyPort({op.write_en(), op.in()}))
895  ? group.drivesAllPorts({op.write_en(), op.in()})
896  : success();
897  })
898  .Case<MemoryOp>([&](auto op) {
899  SmallVector<Value> requiredWritePorts;
900  // If writing to memory, write_en, write_data, and all address ports
901  // should be driven.
902  requiredWritePorts.push_back(op.writeEn());
903  requiredWritePorts.push_back(op.writeData());
904  for (Value address : op.addrPorts())
905  requiredWritePorts.push_back(address);
906 
907  // We only want to verify the write ports if either write_data or
908  // write_en is driven.
909  return succeeded(
910  group.drivesAnyPort({op.writeData(), op.writeEn()}))
911  ? group.drivesAllPorts(requiredWritePorts)
912  : success();
913  })
914  .Case<AndLibOp, OrLibOp, XorLibOp, AddLibOp, SubLibOp, GtLibOp,
915  LtLibOp, EqLibOp, NeqLibOp, GeLibOp, LeLibOp, LshLibOp,
916  RshLibOp, SgtLibOp, SltLibOp, SeqLibOp, SneqLibOp, SgeLibOp,
917  SleLibOp, SrshLibOp>([&](auto op) {
918  Value lhs = op.left(), rhs = op.right();
919  return succeeded(group.drivesAnyPort({lhs, rhs}))
920  ? group.drivesAllPorts({lhs, rhs})
921  : success();
922  })
923  .Default([&](auto op) { return success(); });
924 
925  if (failed(verifyWrites))
926  return group->emitOpError()
927  << "with cell: " << destCell->getName() << " \""
928  << destCell.instanceName()
929  << "\" is performing a write and failed to drive all necessary "
930  "ports.";
931 
932  Operation *srcDefiningOp = assign.src().getDefiningOp();
933  if (srcDefiningOp == nullptr)
934  return success();
935  auto srcCell = dyn_cast<CellInterface>(srcDefiningOp);
936  if (srcCell == nullptr)
937  return success();
938 
939  LogicalResult verifyReads =
940  TypeSwitch<Operation *, LogicalResult>(srcCell)
941  .Case<MemoryOp>([&](auto op) {
942  // If reading memory, all address ports should be driven. Note that
943  // we only want to verify the read ports if read_data is used in the
944  // group.
945  return succeeded(group.readsAnyPort({op.readData()}))
946  ? group.drivesAllPorts(op.addrPorts())
947  : success();
948  })
949  .Default([&](auto op) { return success(); });
950 
951  if (failed(verifyReads))
952  return group->emitOpError() << "with cell: " << srcCell->getName() << " \""
953  << srcCell.instanceName()
954  << "\" is having a read performed upon it, and "
955  "failed to drive all necessary ports.";
956 
957  return success();
958 }

998 /// `isDestination` boolean is used to distinguish whether the value is a source
999 /// or a destination.

1001  bool isDestination) {

1056 static ParseResult parseAssignOp(OpAsmParser &parser, OperationState &result) {
1057  OpAsmParser::OperandType destination;
1058  if (parser.parseOperand(destination) || parser.parseEqual())
1059  return failure();
1060 
1061  // An AssignOp takes one of the two following forms:
1062  // (1) %<dest> = %<src> : <type>
1063  // (2) %<dest> = %<guard> ? %<src> : <type>
1064  OpAsmParser::OperandType guardOrSource;
1065  if (parser.parseOperand(guardOrSource))
1066  return failure();
1067 
1068  // Since the guard is optional, we need to check if there is an accompanying
1069  // `?` symbol.
1070  OpAsmParser::OperandType source;
1071  bool hasGuard = succeeded(parser.parseOptionalQuestion());
1072  if (hasGuard) {
1073  // The guard exists. Parse the source.
1074  if (parser.parseOperand(source))
1075  return failure();
1076  }
1077 
1078  Type type;
1079  if (parser.parseColonType(type) ||
1080  parser.resolveOperand(destination, type, result.operands))
1081  return failure();
1082 
1083  if (hasGuard) {
1084  Type i1Type = parser.getBuilder().getI1Type();
1085  // Since the guard is optional, it is listed last in the arguments of the
1086  // AssignOp. Therefore, we must parse the source first.
1087  if (parser.resolveOperand(source, type, result.operands) ||
1088  parser.resolveOperand(guardOrSource, i1Type, result.operands))
1089  return failure();
1090  } else {
1091  // This is actually a source.
1092  if (parser.resolveOperand(guardOrSource, type, result.operands))
1093  return failure();
1094  }
1095 
1096  return success();
1097 }

1493 /// Checks preconditions for the common tail pattern. This canonicalization is
1494 /// stringent about not entering nested control operations, as this may cause
1495 /// unintentional changes in behavior.
1496 /// We only look for two cases: (1) both regions are ParOps, and
1497 /// (2) both regions are SeqOps. The case when these are different, e.g. ParOp
1498 /// and SeqOp, will only produce less optimal code, or even worse, change the
1499 /// behavior.
1500 template <typename OpTy>
1501 static bool hasCommonTailPatternPreConditions(IfOp op) {
1502  static_assert(std::is_same<SeqOp, OpTy>() || std::is_same<ParOp, OpTy>(),
1503  "Should be a SeqOp or ParOp.");
1504 
1505  if (!op.thenBodyExists() || !op.elseBodyExists())
1506  return false;
1507  if (op.getThenBody()->empty() || op.getElseBody()->empty())
1508  return false;
1509 
1510  Block *thenBody = op.getThenBody(), *elseBody = op.getElseBody();
1511  return isa<OpTy>(thenBody->front()) && isa<OpTy>(elseBody->front());
1512 }

1514 /// seq {
1515 /// if %a with @G { if %a with @G {
1516 /// seq { ... calyx.enable @A } seq { ... }
1517 /// else { -> } else {
1518 /// seq { ... calyx.enable @A } seq { ... }
1519 /// } }
1520 /// calyx.enable @A
1521 /// }
1522 struct CommonTailPatternWithSeq : mlir::OpRewritePattern<IfOp> {
1523  using mlir::OpRewritePattern<IfOp>::OpRewritePattern;
1524 
1525  LogicalResult matchAndRewrite(IfOp ifOp,
1526  PatternRewriter &rewriter) const override {
1527  if (!hasCommonTailPatternPreConditions<SeqOp>(ifOp))
1528  return failure();
1529 
1530  auto thenControl = cast<SeqOp>(ifOp.getThenBody()->front()),
1531  elseControl = cast<SeqOp>(ifOp.getElseBody()->front());
1532  Optional<EnableOp> lastThenEnableOp = getLastEnableOp(thenControl),
1533  lastElseEnableOp = getLastEnableOp(elseControl);
1534 
1535  if (!lastThenEnableOp.hasValue() || !lastElseEnableOp.hasValue())
1536  return failure();
1537  if (lastThenEnableOp->groupName() != lastElseEnableOp->groupName())
1538  return failure();
1539 
1540  // Place the IfOp and pulled EnableOp inside a sequential region, in case
1541  // this IfOp is nested in a ParOp. This avoids unintentionally
1542  // parallelizing the pulled out EnableOps.
1543  rewriter.setInsertionPointAfter(ifOp);
1544  SeqOp seqOp = rewriter.create<SeqOp>(ifOp.getLoc());
1545  Block *body = seqOp.getBody();
1546  ifOp->remove();
1547  body->push_back(ifOp);
1548  rewriter.setInsertionPointToEnd(body);
1549  rewriter.create<EnableOp>(seqOp.getLoc(), lastThenEnableOp->groupName());
1550 
1551  // Erase the common EnableOp from the Then and Else regions.
1552  rewriter.eraseOp(*lastThenEnableOp);
1553  rewriter.eraseOp(*lastElseEnableOp);
1554  return success();
1555  }
1556 };

1558 /// if %a with @G { par {
1559 /// par { if %a with @G {
1560 /// ... par { ... }
1561 /// calyx.enable @A } else {
1562 /// calyx.enable @B -> par { ... }
1563 /// } }
1564 /// } else { calyx.enable @A
1565 /// par { calyx.enable @B
1566 /// ... }
1567 /// calyx.enable @A
1568 /// calyx.enable @B
1569 /// }
1570 /// }
1571 struct CommonTailPatternWithPar : mlir::OpRewritePattern<IfOp> {
1572  using mlir::OpRewritePattern<IfOp>::OpRewritePattern;
1573 
1574  LogicalResult matchAndRewrite(IfOp ifOp,
1575  PatternRewriter &rewriter) const override {
1576  if (!hasCommonTailPatternPreConditions<ParOp>(ifOp))
1577  return failure();
1578  auto thenControl = cast<ParOp>(ifOp.getThenBody()->front()),
1579  elseControl = cast<ParOp>(ifOp.getElseBody()->front());
1580 
1581  llvm::StringMap<EnableOp> A = getAllEnableOpsInImmediateBody(thenControl),
1582  B = getAllEnableOpsInImmediateBody(elseControl);
1583 
1584  // Compute the intersection between `A` and `B`.
1585  SmallVector<StringRef> groupNames;
1586  for (auto a = A.begin(); a != A.end(); ++a) {
1587  StringRef groupName = a->getKey();
1588  auto b = B.find(groupName);
1589  if (b == B.end())
1590  continue;
1591  // This is also an element in B.
1592  groupNames.push_back(groupName);
1593  // Since these are being pulled out, erase them.
1594  rewriter.eraseOp(a->getValue());
1595  rewriter.eraseOp(b->getValue());
1596  }
1597  // Place the IfOp and EnableOp(s) inside a parallel region, in case this
1598  // IfOp is nested in a SeqOp. This avoids unintentionally sequentializing
1599  // the pulled out EnableOps.
1600  rewriter.setInsertionPointAfter(ifOp);
1601  ParOp parOp = rewriter.create<ParOp>(ifOp.getLoc());
1602  Block *body = parOp.getBody();
1603  ifOp->remove();
1604  body->push_back(ifOp);
1605 
1606  // Pull out the intersection between these two sets, and erase their
1607  // counterparts in the Then and Else regions.
1608  rewriter.setInsertionPointToEnd(body);
1609  for (StringRef groupName : groupNames)
1610  rewriter.create<EnableOp>(parOp.getLoc(), groupName);
1611 
1612  return success();
1613  }
1614 };
