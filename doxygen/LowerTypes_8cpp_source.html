9 // This file defines the LowerTypes pass. This pass replaces aggregate types
10 // with expanded values.
11 //
12 // This pass walks the operations in reverse order. This lets it visit users
13 // before defs. Users can usually be expanded out to multiple operations (think
14 // mux of a bundle to muxes of each field) with a temporary subWhatever op
15 // inserted. When processing an aggregate producer, we blow out the op as
16 // appropriate, then walk the users, often those are subWhatever ops which can
17 // be bypassed and deleted. Function arguments are logically last on the
18 // operation visit order and walked left to right, being peeled one layer at a
19 // time with replacements inserted to the right of the original argument.
20 //
21 // Each processing of an op peels one layer of aggregate type off. Because new
22 // ops are inserted immediately above the current up, the walk will visit them
23 // next, effectively recusing on the aggregate types, without recusing. These
24 // potentially temporary ops(if the aggregate is complex) effectively serve as
25 // the worklist. Often aggregates are shallow, so the new ops are the final
26 // ones.
27 //

71 /// Peel one layer of an aggregate type into its components. Type may be
72 /// complex, but empty, in which case fields is empty, but the return is true.
73 static bool peelType(Type type, SmallVectorImpl<FlatBundleFieldEntry> &fields) {
74  return TypeSwitch<Type, bool>(type)
75  .Case<BundleType>([&](auto bundle) {
76  SmallString<16> tmpSuffix;
77  // Otherwise, we have a bundle type. Break it down.
78  for (size_t i = 0, e = bundle.getNumElements(); i < e; ++i) {
79  auto elt = bundle.getElement(i);
80  // Construct the suffix to pass down.
81  tmpSuffix.resize(0);
82  tmpSuffix.push_back('_');
83  tmpSuffix.append(elt.name.getValue());
84  fields.emplace_back(elt.type, i, bundle.getFieldID(i), tmpSuffix,
85  elt.isFlip);
86  }
87  return true;
88  })
89  .Case<FVectorType>([&](auto vector) {
90  // Increment the field ID to point to the first element.
91  for (size_t i = 0, e = vector.getNumElements(); i != e; ++i) {
92  fields.emplace_back(vector.getElementType(), i, vector.getFieldID(i),
93  "_" + std::to_string(i), false);
94  }
95  return true;
96  })
97  .Default([](auto op) { return false; });
98 }

100 /// Return if something is not a normal subaccess. Non-normal includes
101 /// zero-length vectors and constant indexes (which are really subindexes).
102 static bool isNotSubAccess(Operation *op) {
103  SubaccessOp sao = dyn_cast<SubaccessOp>(op);
104  if (!sao)
105  return true;
106  ConstantOp arg = dyn_cast_or_null<ConstantOp>(sao.index().getDefiningOp());
107  if (arg && sao.input().getType().cast<FVectorType>().getNumElements() != 0)
108  return true;
109  return false;
110 }

112 /// Look through and collect subfields leading to a subaccess.
113 static SmallVector<Operation *> getSAWritePath(Operation *op) {
114  SmallVector<Operation *> retval;
115  auto defOp = op->getOperand(0).getDefiningOp();
116  while (defOp && isa<SubfieldOp, SubindexOp, SubaccessOp>(defOp)) {
117  retval.push_back(defOp);
118  defOp = defOp->getOperand(0).getDefiningOp();
119  }
120  // Trim to the subaccess
121  while (!retval.empty() && isNotSubAccess(retval.back()))
122  retval.pop_back();
123  return retval;
124 }

132 /// If an annotation on one operation is replicated across multiple IR
133 /// operations as a result of type lowering, the replicated annotations may want
134 /// to track which field ID they were applied to. This function adds a fieldID
135 /// to such a replicated operation, if the annotation in question requires it.
136 static Attribute updateAnnotationFieldID(MLIRContext *ctxt, Attribute attr,
137  unsigned fieldID) {
138  DictionaryAttr dict = attr.cast<DictionaryAttr>();
139 
140  // No need to do anything if the annotation applies to the entire field.
141  if (fieldID == 0)
142  return attr;
143 
144  // Only certain annotations require precise tracking of field IDs.
145  Annotation anno(dict);
146  if (!isAnnotationSensitiveToFieldID(anno))
147  return attr;
148 
149  // Add the new ID to the existing field ID in the annotation.
150  if (auto existingFieldID = anno.getMember<IntegerAttr>("fieldID"))
151  fieldID += existingFieldID.getValue().getZExtValue();
152  NamedAttrList fields(dict);
153  fields.set("fieldID", IntegerAttr::get(IntegerType::get(ctxt, 64), fieldID));
154  return DictionaryAttr::get(ctxt, fields);
155 }

157 /// Copy annotations from \p annotations to \p loweredAttrs, except annotations
158 /// with "target" key, that do not match the field suffix.
159 static ArrayAttr filterAnnotations(MLIRContext *ctxt, ArrayAttr annotations,
160  FIRRTLType srcType,
161  FlatBundleFieldEntry field) {
162  SmallVector<Attribute> retval;
163  if (!annotations || annotations.empty())
164  return ArrayAttr::get(ctxt, retval);
165  for (auto opAttr : annotations) {
166  if (auto subAnno = opAttr.dyn_cast<SubAnnotationAttr>()) {
167  // Apply annotations to all elements if fieldID is equal to zero.
168  if (subAnno.getFieldID() == 0) {
169  retval.push_back(subAnno.getAnnotations());
170  continue;
171  }
172 
173  // Check whether the annotation falls into the range of the current field.
174  if (subAnno.getFieldID() >= field.fieldID &&
175  subAnno.getFieldID() <= field.fieldID + field.type.getMaxFieldID()) {
176  if (auto newFieldID = subAnno.getFieldID() - field.fieldID) {
177  // If the target is a subfield/subindex of the current field, create a
178  // new sub-annotation with a new field ID.
179  retval.push_back(SubAnnotationAttr::get(ctxt, newFieldID,
180  subAnno.getAnnotations()));
181  } else {
182  // Otherwise, if the current field is exactly the target, degenerate
183  // the sub-annotation to a normal annotation.
184  retval.push_back(subAnno.getAnnotations());
185  }
186  }
187  } else {
188  retval.push_back(updateAnnotationFieldID(ctxt, opAttr, field.fieldID));
189  }
190  }
191  return ArrayAttr::get(ctxt, retval);
192 }

194 static MemOp cloneMemWithNewType(ImplicitLocOpBuilder *b, MemOp op,
195  FlatBundleFieldEntry field) {
196  SmallVector<Type, 8> ports;
197  SmallVector<Attribute, 8> portNames;
198 
199  auto oldPorts = op.getPorts();
200  for (size_t portIdx = 0, e = oldPorts.size(); portIdx < e; ++portIdx) {
201  auto port = oldPorts[portIdx];
202  ports.push_back(MemOp::getTypeForPort(op.depth(), field.type, port.second));
203  portNames.push_back(port.first);
204  }
205 
206  // It's easier to duplicate the old annotations, then fix and filter them.
207  auto newMem = b->create<MemOp>(
208  ports, op.readLatency(), op.writeLatency(), op.depth(), op.ruw(),
209  portNames, (op.name() + field.suffix).str(), op.annotations().getValue(),
210  op.portAnnotations().getValue(), op.inner_symAttr());
211 
212  SmallVector<Attribute> newAnnotations;
213  for (size_t portIdx = 0, e = newMem.getNumResults(); portIdx < e; ++portIdx) {
214  auto portType = newMem.getResult(portIdx).getType().cast<BundleType>();
215  auto oldPortType = op.getResult(portIdx).getType().cast<BundleType>();
216  SmallVector<Attribute> portAnno;
217  for (auto attr : newMem.getPortAnnotation(portIdx)) {
218  if (auto subAnno = attr.dyn_cast<SubAnnotationAttr>()) {
219  auto targetIndex = oldPortType.getIndexForFieldID(subAnno.getFieldID());
220 
221  // Apply annotations to all elements if the target is the whole
222  // sub-field.
223  if (subAnno.getFieldID() == oldPortType.getFieldID(targetIndex)) {
224  portAnno.push_back(SubAnnotationAttr::get(
225  b->getContext(), portType.getFieldID(targetIndex),
226  subAnno.getAnnotations()));
227  continue;
228  }
229 
230  // Handle aggregate sub-fields, including `(r/w)data` and `(w)mask`.
231  if (oldPortType.getElement(targetIndex).type.isa<BundleType>()) {
232  // Check whether the annotation falls into the range of the current
233  // field. Note that the `field` here is peeled from the `data`
234  // sub-field of the memory port, thus we need to add the fieldID of
235  // `data` or `mask` sub-field to get the "real" fieldID.
236  auto fieldID = field.fieldID + oldPortType.getFieldID(targetIndex);
237  if (subAnno.getFieldID() >= fieldID &&
238  subAnno.getFieldID() <= fieldID + field.type.getMaxFieldID()) {
239  // Create a new sub-annotation with a new field ID. Similarly, we
240  // need to add the fieldID of `data` or `mask` sub-field in the new
241  // memory port type here.
242  auto newFieldID = subAnno.getFieldID() - fieldID +
243  portType.getFieldID(targetIndex);
244  portAnno.push_back(SubAnnotationAttr::get(
245  b->getContext(), newFieldID, subAnno.getAnnotations()));
246  }
247  }
248  } else
249  portAnno.push_back(attr);
250  }
251  newAnnotations.push_back(b->getArrayAttr(portAnno));
252  }
253  newMem.setAllPortAnnotations(newAnnotations);
254  return newMem;
255 }

322 Value TypeLoweringVisitor::getSubWhatever(Value val, size_t index) {
323  if (BundleType bundle = val.getType().dyn_cast<BundleType>()) {

332 /// Conditionally expand a subaccessop write path
333 bool TypeLoweringVisitor::processSAPath(Operation *op) {
334  // Does this LHS have a subaccessop?
335  SmallVector<Operation *> writePath = getSAWritePath(op);
336  if (writePath.empty())
337  return false;
338 
339  lowerSAWritePath(op, writePath);
340  // Unhook the writePath from the connect. This isn't the right type, but we
341  // are deleting the op anyway.
342  op->eraseOperands(0, 2);
343  // See how far up the tree we can delete things.
344  for (size_t i = 0; i < writePath.size(); ++i) {
345  if (writePath[i]->use_empty()) {
346  writePath[i]->erase();
347  } else {
348  break;
349  }
350  }
351  return true;
352 }

354 void TypeLoweringVisitor::lowerBlock(Block *block) {
355  // Lower the operations bottom up.
356  for (auto it = block->rbegin(), e = block->rend(); it != e;) {
357  auto &iop = *it;
358  builder->setInsertionPoint(&iop);
359  builder->setLoc(iop.getLoc());
360  bool removeOp = dispatchVisitor(&iop);
361  ++it;
362  // Erase old ops eagerly so we don't have dangling uses we've already
363  // lowered.
364  if (removeOp)
365  iop.erase();
366  }
367 }

369 bool TypeLoweringVisitor::lowerProducer(
370  Operation *op,
371  llvm::function_ref<Operation *(FlatBundleFieldEntry, StringRef, ArrayAttr)>
372  clone) {
373  // If this is not a bundle, there is nothing to do.
374  auto srcType = op->getResult(0).getType().cast<FIRRTLType>();
375  SmallVector<FlatBundleFieldEntry, 8> fieldTypes;
376  if (!peelType(srcType, fieldTypes))
377  return false;
378 
379  SmallVector<Value> lowered;
380  // Loop over the leaf aggregates.
381  SmallString<16> loweredName;
382  StringAttr innerSym = op->getAttrOfType<StringAttr>("inner_sym");
383  if (auto nameAttr = op->getAttr("name"))
384  if (auto nameStrAttr = nameAttr.dyn_cast<StringAttr>())
385  loweredName = nameStrAttr.getValue();
386  auto baseNameLen = loweredName.size();
387  auto oldAnno = op->getAttr("annotations").dyn_cast_or_null<ArrayAttr>();
388 
389  for (auto field : fieldTypes) {
390  if (!loweredName.empty()) {
391  loweredName.resize(baseNameLen);
392  loweredName += field.suffix;
393  }
394  // For all annotations on the parent op, filter them based on the target
395  // attribute.
396  ArrayAttr loweredAttrs =
397  filterAnnotations(context, oldAnno, srcType, field);
398  auto newOp = clone(field, loweredName, loweredAttrs);
399  // Carry over the inner_sym name, if present.
400  if (innerSym) {
401  if (!lowered.empty())
402  op->emitError("replication due to type lowering renders @")
403  << innerSym.getValue() << " ambiguous";
404  newOp->setAttr("inner_sym", innerSym);
405  }
406  lowered.push_back(newOp->getResult(0));
407  }
408 
409  processUsers(op->getResult(0), lowered);
410  return true;
411 }

413 void TypeLoweringVisitor::processUsers(Value val, ArrayRef<Value> mapping) {
414  for (auto user : llvm::make_early_inc_range(val.getUsers())) {
415  if (SubindexOp sio = dyn_cast<SubindexOp>(user)) {
416  Value repl = mapping[sio.index()];
417  sio.replaceAllUsesWith(repl);
418  sio.erase();
419  } else if (SubfieldOp sfo = dyn_cast<SubfieldOp>(user)) {
420  // Get the input bundle type.
421  Value repl = mapping[sfo.fieldIndex()];
422  sfo.replaceAllUsesWith(repl);
423  sfo.erase();
424  } else {
425  llvm_unreachable("Unknown aggregate user");
426  }
427  }

438 // module. This also maintains the name attribute for the new argument,

467 // Lower arguments with bundle type by flattening them.
468 bool TypeLoweringVisitor::lowerArg(Operation *module, size_t argIndex,
469  SmallVectorImpl<PortInfo> &newArgs,
470  SmallVectorImpl<Value> &lowering) {
471 
472  // Flatten any bundle types.
473  SmallVector<FlatBundleFieldEntry> fieldTypes;
474  auto srcType = newArgs[argIndex].type.cast<FIRRTLType>();
475  if (!peelType(srcType, fieldTypes))
476  return false;
477 
478  for (auto field : llvm::enumerate(fieldTypes)) {
479  auto newValue = addArg(module, 1 + argIndex + field.index(), srcType,
480  field.value(), newArgs[argIndex]);
481  newArgs.insert(newArgs.begin() + 1 + argIndex + field.index(),
482  newValue.second);
483  // Lower any other arguments by copying them to keep the relative order.
484  lowering.push_back(newValue.first);
485  }
486  return true;
487 }

501 void TypeLoweringVisitor::lowerSAWritePath(Operation *op,
502  ArrayRef<Operation *> writePath) {
503  SubaccessOp sao = cast<SubaccessOp>(writePath.back());
504  auto saoType = sao.input().getType().cast<FVectorType>();
505  auto selectWidth =
506  sao.index().getType().cast<FIRRTLType>().getBitWidthOrSentinel();
507 
508  for (size_t index = 0, e = saoType.getNumElements(); index < e; ++index) {
509  auto cond = builder->create<EQPrimOp>(
510  sao.index(),
511  builder->createOrFold<ConstantOp>(UIntType::get(context, selectWidth),
512  APInt(selectWidth, index)));
513  builder->create<WhenOp>(cond, false, [&]() {
514  // Recreate the write Path
515  Value leaf = builder->create<SubindexOp>(sao.input(), index);
516  for (int i = writePath.size() - 2; i >= 0; --i)
517  leaf = cloneAccess(builder, writePath[i], leaf);
518 
519  if (isa<ConnectOp>(op))
520  builder->create<ConnectOp>(leaf, op->getOperand(1));
521  else
522  builder->create<PartialConnectOp>(leaf, op->getOperand(1));
523  });
524  }

527 // Expand connects of aggregates
528 bool TypeLoweringVisitor::visitStmt(ConnectOp op) {
529  if (processSAPath(op))
530  return true;
531 
532  // Attempt to get the bundle types.
533  SmallVector<FlatBundleFieldEntry> fields;
534  if (!peelType(op.dest().getType(), fields))
535  return false;
536 
537  // Loop over the leaf aggregates.
538  for (auto field : llvm::enumerate(fields)) {
539  Value src = getSubWhatever(op.src(), field.index());
540  Value dest = getSubWhatever(op.dest(), field.index());
541  if (field.value().isOutput)
542  std::swap(src, dest);
543  if (src.getType().isa<AnalogType>())
544  builder->create<AttachOp>(ArrayRef<Value>{dest, src});
545  else
546  builder->create<ConnectOp>(dest, src);
547  }
548  return true;
549 }

551 bool TypeLoweringVisitor::visitStmt(PartialConnectOp op) {
552  if (processSAPath(op))
553  return true;
554 
555  SmallVector<FlatBundleFieldEntry> srcFields, destFields;
556  peelType(op.src().getType(), srcFields);
557  bool dValid = peelType(op.dest().getType(), destFields);
558 
559  // Ground Type
560  if (!dValid) {
561  // check for truncation
562  Value src = op.src();
563  Value dest = op.dest();
564  auto srcType = src.getType().cast<FIRRTLType>();
565  auto destType = dest.getType().cast<FIRRTLType>();
566  auto srcWidth = srcType.getBitWidthOrSentinel();
567  auto destWidth = destType.getBitWidthOrSentinel();
568 
569  if (destType == srcType) {
570  builder->create<ConnectOp>(dest, src);
571  return true;
572  }
573 
574  if (!destType.isa<IntType>() || !srcType.isa<IntType>() || destWidth < 0)
575  return false;
576 
577  if (destWidth < srcWidth) {
578  // firrtl.tail always returns uint even for sint operands.
579  IntType tmpType = destType.cast<IntType>();
580  if (tmpType.isSigned())
581  tmpType = UIntType::get(destType.getContext(), destWidth);
582  src = builder->create<TailPrimOp>(tmpType, src, srcWidth - destWidth);
583  // Insert the cast back to signed if needed.
584  if (tmpType != destType)
585  src = builder->create<AsSIntPrimOp>(destType, src);
586  } else {
587  // Need to extend arg
588  src = builder->create<PadPrimOp>(src, destWidth);
589  }
590  builder->create<ConnectOp>(dest, src);
591  return true;
592  }
593 
594  // Aggregates
595  if (FVectorType fvector = op.src().getType().dyn_cast<FVectorType>()) {
596  for (int index = 0, e = std::min(srcFields.size(), destFields.size());
597  index != e; ++index) {
598  Value src = builder->create<SubindexOp>(op.src(), index);
599  Value dest = builder->create<SubindexOp>(op.dest(), index);
600  if (src.getType() == dest.getType())
601  builder->create<ConnectOp>(dest, src);
602  else
603  builder->create<PartialConnectOp>(dest, src);
604  }
605  } else if (BundleType srcBundle = op.src().getType().dyn_cast<BundleType>()) {
606  // Pairwise connect on matching field names
607  BundleType destBundle = op.dest().getType().cast<BundleType>();
608  for (int srcIndex = 0, srcEnd = srcBundle.getNumElements();
609  srcIndex < srcEnd; ++srcIndex) {
610  auto srcName = srcBundle.getElement(srcIndex).name;
611  for (int destIndex = 0, destEnd = destBundle.getNumElements();
612  destIndex < destEnd; ++destIndex) {
613  auto destName = destBundle.getElement(destIndex).name;
614  if (srcName == destName) {
615  Value src = builder->create<SubfieldOp>(op.src(), srcIndex);
616  Value dest = builder->create<SubfieldOp>(op.dest(), destIndex);
617  if (destFields[destIndex].isOutput)
618  std::swap(src, dest);
619  if (src.getType().isa<AnalogType>())
620  builder->create<AttachOp>(ArrayRef<Value>{dest, src});
621  else if (src.getType() == dest.getType())
622  builder->create<ConnectOp>(dest, src);
623  else
624  builder->create<PartialConnectOp>(dest, src);
625  break;
626  }
627  }
628  }
629  } else {
630  op.emitError("Unknown aggregate type");
631  }
632 
633  return true;
634 }

650 // Convert an aggregate type into a flat list of fields.
651 // This is used to flatten the aggregate memory datatype.
652 // Recursively populate the results with each ground type field.
653 static bool flattenType(FIRRTLType type, SmallVectorImpl<IntType> &results) {
654  std::function<bool(FIRRTLType)> flatten = [&](FIRRTLType type) -> bool {
655  return TypeSwitch<FIRRTLType, bool>(type)
656  .Case<BundleType>([&](auto bundle) {
657  for (auto &elt : bundle.getElements())
658  if (!flatten(elt.type))
659  return false;
660  return true;
661  })
662  .Case<FVectorType>([&](auto vector) {
663  for (size_t i = 0, e = vector.getNumElements(); i != e; ++i)
664  if (!flatten(vector.getElementType()))
665  return false;
666  return true;
667  })
668  .Case<IntType>([&](auto iType) {
669  results.push_back({iType});
670  return iType.getWidth().hasValue();
671  })
672  .Default([&](auto) { return false; });
673  };
674  if (flatten(type))
675  return true;
676  return false;
677 }

679 /// Lower memory operations. A new memory is created for every leaf
680 /// element in a memory's data type.
681 bool TypeLoweringVisitor::visitDecl(MemOp op) {
682  // Attempt to get the bundle types.
683  SmallVector<FlatBundleFieldEntry> fields;
684  if (!peelType(op.getDataType(), fields))
685  return false;
686 
687  SmallVector<MemOp> newMemories;
688  SmallVector<WireOp> oldPorts;
689 
690  // Wires for old ports
691  for (unsigned int index = 0, end = op.getNumResults(); index < end; ++index) {
692  auto result = op.getResult(index);
693  auto wire = builder->create<WireOp>(
694  result.getType(),
695  (op.name() + "_" + op.getPortName(index).getValue()).str());
696  oldPorts.push_back(wire);
697  result.replaceAllUsesWith(wire.getResult());
698  }
699  // The vector of leaf elements type after flattening the data.
700  SmallVector<IntType> flatMemType;
701  // MaskGranularity : how many bits each mask bit controls.
702  size_t maskGran = 1;
703  // Total mask bitwidth after flattening.
704  uint32_t totalmaskWidths = 0;
705  // How many mask bits each field type requires.
706  SmallVector<unsigned> maskWidths;
707 
708  auto hasSubAnno = [&]() -> bool {
709  for (size_t portIdx = 0, e = op.getNumResults(); portIdx < e; ++portIdx)
710  for (auto attr : op.getPortAnnotation(portIdx)) {
711  if (auto subAnno = attr.dyn_cast<SubAnnotationAttr>())
712  return true;
713  }
714  return false;
715  };
716  // If subannotations present on aggregate fields, we cannot flatten the
717  // memory. It must be split into one memory per aggregate field.
718  if (flattenAggregateMemData)
719  if (hasSubAnno() || !flattenType(op.getDataType(), flatMemType))
720  flattenAggregateMemData = false;
721 
722  if (flattenAggregateMemData) {
723  SmallVector<Operation *, 8> flatData;
724  SmallVector<int32_t> memWidths;
725  // Get the width of individual aggregate leaf elements.
726  for (auto f : flatMemType)
727  memWidths.push_back(f.getWidth().getValue());
728 
729  maskGran = memWidths[0];
730  size_t memFlatWidth = 0;
731  // Compute the GCD of all data bitwidths.
732  for (auto w : memWidths) {
733  memFlatWidth += w;
734  maskGran = llvm::GreatestCommonDivisor64(maskGran, w);
735  }
736  for (auto w : memWidths) {
737  // How many mask bits required for each flattened field.
738  auto mWidth = w / maskGran;
739  maskWidths.push_back(mWidth);
740  totalmaskWidths += mWidth;
741  }
742  // Now create a new memory of type flattened data.
743  // ----------------------------------------------
744  SmallVector<Type, 8> ports;
745  SmallVector<Attribute, 8> portNames;
746 
747  // Create a new memoty data type of unsigned and computed width.
748  auto flatType = UIntType::get(context, memFlatWidth);
749  auto opPorts = op.getPorts();
750  for (size_t portIdx = 0, e = opPorts.size(); portIdx < e; ++portIdx) {
751  auto port = opPorts[portIdx];
752  ports.push_back(MemOp::getTypeForPort(op.depth(), flatType, port.second,
753  totalmaskWidths));
754  portNames.push_back(port.first);
755  }
756 
757  auto flatMem = builder->create<MemOp>(
758  ports, op.readLatency(), op.writeLatency(), op.depth(), op.ruw(),
759  portNames, op.name(), op.annotations().getValue(),
760  op.portAnnotations().getValue(), op.inner_symAttr());
761  // Done creating the memory.
762  // ----------------------------------------------
763  newMemories.push_back(flatMem);
764  } else {
765  // Memory for each field
766  for (auto field : fields)
767  newMemories.push_back(cloneMemWithNewType(builder, op, field));
768  }
769  // Emit an error if the memory had an `inner_sym` that was replicated across
770  // multiple memories.
771  if (newMemories.size() > 1 && op.inner_symAttr())
772  op->emitError("replication due to type lowering renders @")
773  << op.inner_symAttr().getValue() << " ambiguous";
774  // Hook up the new memories to the wires the old memory was replaced with.
775  for (size_t index = 0, rend = op.getNumResults(); index < rend; ++index) {
776  auto result = oldPorts[index];
777  auto rType = result.getType().cast<BundleType>();
778  for (size_t fieldIndex = 0, fend = rType.getNumElements();
779  fieldIndex != fend; ++fieldIndex) {
780  auto name = rType.getElement(fieldIndex).name.getValue();
781  auto oldField = builder->create<SubfieldOp>(result, fieldIndex);
782  // data and mask depend on the memory type which was split. They can also
783  // go both directions, depending on the port direction.
784  if (name == "data" || name == "mask" || name == "wdata" ||
785  name == "wmask" || name == "rdata") {
786  if (flattenAggregateMemData) {
787  // If memory was flattened instead of one memory per aggregate field.
788  Value newField =
789  getSubWhatever(newMemories[0].getResult(index), fieldIndex);
790  Value realOldField = oldField;
791  if (rType.getElement(fieldIndex).isFlip) {
792  // Cast the memory read data from flat type to aggregate.
793  newField = builder->createOrFold<BitCastOp>(
794  oldField.getType().cast<FIRRTLType>(), newField);
795  // Write the aggregate read data.
796  builder->create<ConnectOp>(realOldField, newField);
797  } else {
798  // Cast the input aggregate write data to flat type.
799  realOldField = builder->create<BitCastOp>(
800  newField.getType().cast<FIRRTLType>(), oldField);
801  // Mask bits require special handling, since some of the mask bits
802  // need to be repeated, direct bitcasting wouldn't work. Depending
803  // on the mask granularity, some mask bits will be repeated.
804  if ((name == "mask" || name == "wmask") &&
805  (maskWidths.size() < totalmaskWidths)) {
806  Value catMasks;
807  for (auto m : llvm::enumerate(maskWidths)) {
808  // Get the mask bit.
809  auto mBit = builder->createOrFold<BitsPrimOp>(
810  realOldField, m.index(), m.index());
811  // Check how many times the mask bit needs to be prepend.
812  for (size_t repeat = 0; repeat < m.value(); repeat++)
813  if (m.index() == 0 && repeat == 0)
814  catMasks = mBit;
815  else
816  catMasks = builder->createOrFold<CatPrimOp>(mBit, catMasks);
817  }
818  realOldField = catMasks;
819  }
820  // Now set the mask or write data.
821  // Ensure that the types match.
822  builder->create<ConnectOp>(
823  newField,
824  builder->createOrFold<BitCastOp>(
825  newField.getType().cast<FIRRTLType>(), realOldField));
826  }
827  } else {
828  for (auto field : fields) {
829  auto realOldField = getSubWhatever(oldField, field.index);
830  auto newField = getSubWhatever(
831  newMemories[field.index].getResult(index), fieldIndex);
832  if (rType.getElement(fieldIndex).isFlip)
833  std::swap(realOldField, newField);
834  builder->create<ConnectOp>(newField, realOldField);
835  }
836  }
837  } else {
838  for (auto mem : newMemories) {
839  auto newField =
840  builder->create<SubfieldOp>(mem.getResult(index), fieldIndex);
841  builder->create<ConnectOp>(newField, oldField);
842  }
843  }
844  }
845  }
846  return true;
847 }

1063 // Expand BitCastOp of aggregates
1064 bool TypeLoweringVisitor::visitExpr(BitCastOp op) {
1065  Value srcLoweredVal = op.input();
1066  // If the input is of aggregate type, then cat all the leaf fields to form a
1067  // UInt type result. That is, first bitcast the aggregate type to a UInt.
1068  // Attempt to get the bundle types.
1069  SmallVector<FlatBundleFieldEntry> fields;
1070  if (peelType(op.input().getType(), fields)) {
1071  size_t uptoBits = 0;
1072  // Loop over the leaf aggregates and concat each of them to get a UInt.
1073  // Bitcast the fields to handle nested aggregate types.
1074  for (auto field : llvm::enumerate(fields)) {
1075  auto fieldBitwidth = getBitWidth(field.value().type).getValue();
1076  // Ignore zero width fields, like empty bundles.
1077  if (fieldBitwidth == 0)
1078  continue;
1079  Value src = getSubWhatever(op.input(), field.index());
1080  // The src could be an aggregate type, bitcast it to a UInt type.
1081  src = builder->createOrFold<BitCastOp>(
1082  UIntType::get(context, fieldBitwidth), src);
1083  // Take the first field, or else Cat the previous fields with this field.
1084  if (uptoBits == 0)
1085  srcLoweredVal = src;
1086  else
1087  srcLoweredVal = builder->create<CatPrimOp>(src, srcLoweredVal);
1088  // Record the total bits already accumulated.
1089  uptoBits += fieldBitwidth;
1090  }
1091  } else
1092  srcLoweredVal = builder->createOrFold<AsUIntPrimOp>(srcLoweredVal);
1093  // Now the input has been cast to srcLoweredVal, which is of UInt type.
1094  // If the result is an aggregate type, then use lowerProducer.
1095  if (op.getResult().getType().isa<BundleType, FVectorType>()) {
1096  // uptoBits is used to keep track of the bits that have been extracted.
1097  size_t uptoBits = 0;
1098  auto clone = [&](FlatBundleFieldEntry field, StringRef name,
1099  ArrayAttr attrs) -> Operation * {
1100  // All the fields must have valid bitwidth, a requirement for BitCastOp.
1101  auto fieldBits = getBitWidth(field.type).getValue();
1102  // If empty field, then it doesnot have any use, so replace it with an
1103  // invalid op, which should be trivially removed.
1104  if (fieldBits == 0)
1105  return builder->create<InvalidValueOp>(field.type);
1106 
1107  // Assign the field to the corresponding bits from the input.
1108  // Bitcast the field, incase its an aggregate type.
1109  auto extractBits = builder->create<BitsPrimOp>(
1110  srcLoweredVal, uptoBits + fieldBits - 1, uptoBits);
1111  uptoBits += fieldBits;
1112  return builder->create<BitCastOp>(field.type, extractBits);
1113  };
1114  return lowerProducer(op, clone);
1115  }
1116 
1117  // If ground type, then replace the result.
1118  if (op.getType().dyn_cast<SIntType>())
1119  srcLoweredVal = builder->create<AsSIntPrimOp>(srcLoweredVal);
1120  op.getResult().replaceAllUsesWith(srcLoweredVal);
1121  return true;
1122 }

1223  }
