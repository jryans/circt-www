50 void emitInnerRefPart(TclOutputState &s, Operation *op) {
51  // Extract the name from an InstanceOp or other op with a "name".
52  StringAttr nameAttr;
53  if (auto instOp = dyn_cast<msft::InstanceOp>(op))
54  nameAttr = instOp.instanceNameAttr();
55  else if (auto name = op->getAttrOfType<StringAttr>("name"))
56  nameAttr = name;
57  assert(nameAttr && "placed ops must have a name");
58 
59  // We append new symbolRefs to the state, so s.symbolRefs.size() is the
60  // index of the InnerRefAttr we are about to add.
61  s.os << "{{" << s.symbolRefs.size() << "}}";
62 
63  // At this point, everything is contained within MSFTModuleOps.
64  auto mod = op->getParentOfType<MSFTModuleOp>();
65  assert(mod && "named op must be contained in MSFTModuleOp");
66  StringAttr modName = mod.getNameAttr();
67 
68  // Append a new inner reference for the template above.
69  s.symbolRefs.push_back(InnerRefAttr::get(modName, nameAttr));
70 }

72 void emitPath(TclOutputState &s, PlacementDB::PlacedInstance inst,
73  SymbolCache &symCache) {
74  // Traverse each part of the path.
75  RootedInstancePathAttr path = inst.path;
76  for (auto part : path.getPath()) {
77  auto inst = cast<msft::InstanceOp>(symCache.getDefinition(part));
78  assert(inst && "path instance must be in symbol cache");
79 
80  emitInnerRefPart(s, inst);
81  s.os << '|';
82  }
83 
84  // If instance name is specified, add it in between the parent entity path and
85  // the child entity path.
86  emitInnerRefPart(s, inst.op);
87 
88  // Some placements don't require subpaths.
89  if (!inst.subpath.empty()) {
90  s.os << '|';
91  s.os << inst.subpath;
92  }
93 
94  s.os << '\n';
95 }

97 /// Emit tcl in the form of:
98 /// "set_location_assignment MPDSP_X34_Y285_N0 -to $parent|fooInst|entityName"
99 static void emit(TclOutputState &s, PlacementDB::PlacedInstance inst,
100  PhysLocationAttr pla, SymbolCache &symCache) {
101 
102  s.indent() << "set_location_assignment ";
103 
104  // Different devices have different 'number' letters (the 'N' in 'N0'). M20Ks
105  // and DSPs happen to have the same one, probably because they never co-exist
106  // at the same location.
107  char numCharacter;
108  switch (pla.getPrimitiveType().getValue()) {
109  case PrimitiveType::M20K:
110  s.os << "M20K";
111  numCharacter = 'N';
112  break;
113  case PrimitiveType::DSP:
114  s.os << "MPDSP";
115  numCharacter = 'N';
116  break;
117  }
118 
119  // Write out the rest of the location info.
120  s.os << "_X" << pla.getX() << "_Y" << pla.getY() << "_" << numCharacter
121  << pla.getNum();
122 
123  // To which entity does this apply?
124  s.os << " -to $parent|";
125  emitPath(s, inst, symCache);
126 }

128 /// Emit tcl in the form of:
129 /// set_instance_assignment -name PLACE_REGION "X1 Y1 X20 Y20" -to $parent|a|b|c
130 /// set_instance_assignment -name RESERVE_PLACE_REGION OFF -to $parent|a|b|c
131 /// set_instance_assignment -name CORE_ONLY_PLACE_REGION ON -to $parent|a|b|c
132 /// set_instance_assignment -name REGION_NAME test_region -to $parent|a|b|c
133 static void emit(TclOutputState &s, PlacementDB::PlacedInstance inst,
134  PhysicalRegionRefAttr regionRef, SymbolCache &symCache) {
135  auto topModule = inst.op->getParentOfType<mlir::ModuleOp>();
136  auto physicalRegion =
137  topModule.lookupSymbol<PhysicalRegionOp>(regionRef.getPhysicalRegion());
138  assert(physicalRegion && "must reference an existant physical region");
139 
140  // PLACE_REGION directive.
141  s.indent() << "set_instance_assignment -name PLACE_REGION \"";
142  auto physicalBounds =
143  physicalRegion.bounds().getAsRange<PhysicalBoundsAttr>();
144  llvm::interleave(
145  physicalBounds, s.os,
146  [&s](PhysicalBoundsAttr bounds) {
147  s.os << 'X' << bounds.getXMin() << ' ';
148  s.os << 'Y' << bounds.getYMin() << ' ';
149  s.os << 'X' << bounds.getXMax() << ' ';
150  s.os << 'Y' << bounds.getYMax();
151  },
152  ";");
153  s.os << '"';
154  s.os << " -to $parent|";
155  emitPath(s, inst, symCache);
156 
157  // RESERVE_PLACE_REGION directive.
158  s.indent() << "set_instance_assignment -name RESERVE_PLACE_REGION OFF";
159  s.os << " -to $parent|";
160  emitPath(s, inst, symCache);
161 
162  // CORE_ONLY_PLACE_REGION directive.
163  s.indent() << "set_instance_assignment -name CORE_ONLY_PLACE_REGION ON";
164  s.os << " -to $parent|";
165  emitPath(s, inst, symCache);
166 
167  // REGION_NAME directive.
168  s.indent() << "set_instance_assignment -name REGION_NAME ";
169  s.os << physicalRegion.getName();
170  s.os << " -to $parent|";
171  emitPath(s, inst, symCache);
172 }

175 void circt::msft::populateSymbolCache(mlir::ModuleOp mod, SymbolCache &cache) {
176  // Traverse each module and each instance within the module.
177  for (auto msftMod : mod.getOps<MSFTModuleOp>()) {
178  for (auto inst : msftMod.getOps<msft::InstanceOp>()) {
179  // Use the instance symbol name.
180  StringAttr symName = inst.sym_nameAttr();
181 
182  // Add the symbol to the cache.
183  cache.addDefinition(symName, inst);
184  }
185  }
186 
187  cache.freeze();
188 }

190 /// Write out all the relevant tcl commands. Create one 'proc' per module which
191 /// takes the parent entity name since we don't assume that the created module
192 /// is the top level for the entire design.
193 LogicalResult circt::msft::exportQuartusTcl(MSFTModuleOp hwMod,
194  SymbolCache &symCache,
195  StringRef outputFile) {
196  // Build up the output Tcl, tracking symbol references in state.
197  std::string s;
198  llvm::raw_string_ostream os(s);
199  TclOutputState state(os);
200  PlacementDB db(hwMod);
201  size_t failures = db.addDesignPlacements();
202  if (failures != 0)
203  return hwMod->emitError("Could not place ") << failures << " instances";
204 
205  os << "proc " << hwMod.getName() << "_config { parent } {\n";
206 
207  db.walkPlacements([&state, &symCache](PhysLocationAttr loc,
208  PlacementDB::PlacedInstance inst) {
209  emit(state, inst, loc, symCache);
210  });
211 
212  db.walkRegionPlacements(
213  [&state, &symCache](PhysicalRegionRefAttr regionRef,
214  PlacementDB::PlacedInstance inst) {
215  emit(state, inst, regionRef, symCache);
216  });
217 
218  os << "}\n\n";
219 
220  // Create a verbatim op containing the Tcl and symbol references.
221  OpBuilder builder = OpBuilder::atBlockEnd(hwMod->getBlock());
222  auto verbatim = builder.create<sv::VerbatimOp>(
223  builder.getUnknownLoc(), os.str(), ValueRange{},
224  builder.getArrayAttr(state.symbolRefs));
225 
226  // When requested, give the verbatim op an output file.
227  if (!outputFile.empty()) {
228  auto outputFileAttr =
229  OutputFileAttr::getFromFilename(builder.getContext(), outputFile);
230  verbatim->setAttr("output_file", outputFileAttr);
231  }
232 
233  return success();
234 }
