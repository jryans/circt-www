59 /// Return an encapsulated set of information about input and output ports of
60 /// the specified module or instance. The input ports always come before the
61 /// output ports in the list.
62 /// TODO: This should really be shared with the HW dialect instead of cloned.
63 /// Consider adding a `HasModulePorts` op interface to facilitate.
64 hw::ModulePortInfo MSFTModuleOp::getPorts() {
65  SmallVector<hw::PortInfo> inputs, outputs;
66  auto argTypes = getType().getInputs();
67 
68  auto argNames = this->argNames();
69  for (unsigned i = 0, e = argTypes.size(); i < e; ++i) {
70  bool isInOut = false;
71  auto type = argTypes[i];
72 
73  if (auto inout = type.dyn_cast<hw::InOutType>()) {
74  isInOut = true;
75  type = inout.getElementType();
76  }
77 
78  auto direction =
79  isInOut ? hw::PortDirection::INOUT : hw::PortDirection::INPUT;
80  inputs.push_back({argNames[i].cast<StringAttr>(), direction, type, i});
81  }
82 
83  auto resultNames = this->resultNames();
84  auto resultTypes = getType().getResults();
85  for (unsigned i = 0, e = resultTypes.size(); i < e; ++i) {
86  outputs.push_back({resultNames[i].cast<StringAttr>(),
87  hw::PortDirection::OUTPUT, resultTypes[i], i});
88  }
89  return hw::ModulePortInfo(inputs, outputs);
90 }

118 static ParseResult parseMSFTModuleOp(OpAsmParser &parser,
119  OperationState &result) {
120  using namespace mlir::function_like_impl;
121 
122  auto loc = parser.getCurrentLocation();
123 
124  SmallVector<OpAsmParser::OperandType, 4> entryArgs;
125  SmallVector<NamedAttrList, 4> argAttrs;
126  SmallVector<NamedAttrList, 4> resultAttrs;
127  SmallVector<Type, 4> argTypes;
128  SmallVector<Type, 4> resultTypes;
129  auto &builder = parser.getBuilder();
130 
131  // Parse the name as a symbol.
132  StringAttr nameAttr;
133  if (parser.parseSymbolName(nameAttr, SymbolTable::getSymbolAttrName(),
134  result.attributes))
135  return failure();
136 
137  // Parse the parameters
138  DictionaryAttr paramsAttr;
139  if (parser.parseAttribute(paramsAttr))
140  return failure();
141  result.addAttribute("parameters", paramsAttr);
142 
143  // Parse the function signature.
144  bool isVariadic = false;
145  SmallVector<Attribute> resultNames;
146  if (hw::module_like_impl::parseModuleFunctionSignature(
147  parser, entryArgs, argTypes, argAttrs, isVariadic, resultTypes,
148  resultAttrs, resultNames))
149  return failure();
150 
151  // Record the argument and result types as an attribute. This is necessary
152  // for external modules.
153  auto type = builder.getFunctionType(argTypes, resultTypes);
154  result.addAttribute(getTypeAttrName(), TypeAttr::get(type));
155 
156  // If function attributes are present, parse them.
157  if (parser.parseOptionalAttrDictWithKeyword(result.attributes))
158  return failure();
159 
160  auto *context = result.getContext();
161 
162  if (hasAttribute("argNames", result.attributes) ||
163  hasAttribute("resultNames", result.attributes)) {
164  parser.emitError(
165  loc, "explicit argNames and resultNames attributes not allowed");
166  return failure();
167  }
168 
169  // Use the argument and result names if not already specified.
170  SmallVector<Attribute> argNames;
171  if (!entryArgs.empty()) {
172  for (auto &arg : entryArgs)
173  argNames.push_back(
174  hw::module_like_impl::getPortNameAttr(context, arg.name));
175  } else if (!argTypes.empty()) {
176  // The parser returns empty names in a special way.
177  argNames.assign(argTypes.size(), StringAttr::get(context, ""));
178  }
179 
180  result.addAttribute("argNames", ArrayAttr::get(context, argNames));
181  result.addAttribute("resultNames", ArrayAttr::get(context, resultNames));
182 
183  assert(argAttrs.size() == argTypes.size());
184  assert(resultAttrs.size() == resultTypes.size());
185 
186  // Add the attributes to the module arguments.
187  addArgAndResultAttrs(builder, result, argAttrs, resultAttrs);
188 
189  // Parse the optional module body.
190  auto regionSuccess = parser.parseOptionalRegion(
191  *result.addRegion(), entryArgs,
192  entryArgs.empty() ? ArrayRef<Type>() : argTypes);
193  if (regionSuccess.hasValue() && failed(*regionSuccess))
194  return failure();
195 
196  return success();
197 }

235 /// Parse an parameter list if present. Same format as HW dialect.
236 /// module-parameter-list ::= `<` parameter-decl (`,` parameter-decl)* `>`
237 /// parameter-decl ::= identifier `:` type
238 /// parameter-decl ::= identifier `:` type `=` attribute
239 ///
240 static ParseResult parseParameterList(OpAsmParser &parser,
241  SmallVector<Attribute> &parameters) {
242 
243  return parser.parseCommaSeparatedList(
244  OpAsmParser::Delimiter::OptionalLessGreater, [&]() {
245  std::string name;
246  Type type;
247  Attribute value;
248 
249  if (parser.parseKeywordOrString(&name) || parser.parseColonType(type))
250  return failure();
251 
252  // Parse the default value if present.
253  if (succeeded(parser.parseOptionalEqual())) {
254  if (parser.parseAttribute(value, type))
255  return failure();
256  }
257 
258  auto &builder = parser.getBuilder();
259  parameters.push_back(hw::ParamDeclAttr::get(
260  builder.getContext(), builder.getStringAttr(name),
261  TypeAttr::get(type), value));
262  return success();
263  });
264 }

278 /// Print a parameter list for a module or instance. Same format as HW dialect.
279 static void printParameterList(OpAsmPrinter &p, Operation *op,
280  ArrayAttr parameters) {
281  if (parameters.empty())
282  return;
283 
284  p << '<';
285  llvm::interleaveComma(parameters, p, [&](Attribute param) {
286  auto paramAttr = param.cast<hw::ParamDeclAttr>();
287  p << paramAttr.getName().getValue() << ": " << paramAttr.getType();
288  if (auto value = paramAttr.getValue()) {
289  p << " = ";
290  p.printAttributeWithoutType(value);
291  }
292  });
293  p << '>';

296 /// Check parameter specified by `value` to see if it is valid within the scope
297 /// of the specified module `module`. If not, emit an error at the location of
298 /// `usingOp` and return failure, otherwise return success. If `usingOp` is
299 /// null, then no diagnostic is generated. Same format as HW dialect.
300 ///
301 /// If `disallowParamRefs` is true, then parameter references are not allowed.
302 static LogicalResult checkParameterInContext(Attribute value, Operation *module,
303  Operation *usingOp,
304  bool disallowParamRefs) {
305  // Literals are always ok. Their types are already known to match
306  // expectations.
307  if (value.isa<IntegerAttr>() || value.isa<FloatAttr>() ||
308  value.isa<StringAttr>() || value.isa<hw::ParamVerbatimAttr>())
309  return success();
310 
311  // Check both subexpressions of an expression.
312  if (auto expr = value.dyn_cast<hw::ParamExprAttr>()) {
313  for (auto op : expr.getOperands())
314  if (failed(
315  checkParameterInContext(op, module, usingOp, disallowParamRefs)))
316  return failure();
317  return success();
318  }
319 
320  // Parameter references need more analysis to make sure they are valid within
321  // this module.
322  if (auto parameterRef = value.dyn_cast<hw::ParamDeclRefAttr>()) {
323  auto nameAttr = parameterRef.getName();
324 
325  // Don't allow references to parameters from the default values of a
326  // parameter list.
327  if (disallowParamRefs) {
328  if (usingOp)
329  usingOp->emitOpError("parameter ")
330  << nameAttr << " cannot be used as a default value for a parameter";
331  return failure();
332  }
333 
334  // Find the corresponding attribute in the module.
335  for (auto param : module->getAttrOfType<ArrayAttr>("parameters")) {
336  auto paramAttr = param.cast<hw::ParamDeclAttr>();
337  if (paramAttr.getName() != nameAttr)
338  continue;
339 
340  // If the types match then the reference is ok.
341  if (paramAttr.getType().getValue() == parameterRef.getType())
342  return success();
343 
344  if (usingOp) {
345  auto diag = usingOp->emitOpError("parameter ")
346  << nameAttr << " used with type " << parameterRef.getType()
347  << "; should have type " << paramAttr.getType().getValue();
348  diag.attachNote(module->getLoc()) << "module declared here";
349  }
350  return failure();
351  }
352 
353  if (usingOp) {
354  auto diag = usingOp->emitOpError("use of unknown parameter ") << nameAttr;
355  diag.attachNote(module->getLoc()) << "module declared here";
356  }
357  return failure();
358  }
359 
360  if (usingOp)
361  usingOp->emitOpError("invalid parameter value ") << value;
362  return failure();
363 }

365 static ParseResult parseMSFTModuleExternOp(OpAsmParser &parser,
366  OperationState &result) {
367  using namespace mlir::function_like_impl;
368 
369  auto loc = parser.getCurrentLocation();
370 
371  SmallVector<OpAsmParser::OperandType, 4> entryArgs;
372  SmallVector<NamedAttrList, 4> argAttrs;
373  SmallVector<NamedAttrList, 4> resultAttrs;
374  SmallVector<Type, 4> argTypes;
375  SmallVector<Type, 4> resultTypes;
376  SmallVector<Attribute> parameters;
377  auto &builder = parser.getBuilder();
378 
379  // Parse the name as a symbol.
380  StringAttr nameAttr;
381  if (parser.parseSymbolName(nameAttr, SymbolTable::getSymbolAttrName(),
382  result.attributes))
383  return failure();
384 
385  // Parse the function signature.
386  bool isVariadic = false;
387  SmallVector<Attribute> resultNames;
388  if (parseParameterList(parser, parameters) ||
389  hw::module_like_impl::parseModuleFunctionSignature(
390  parser, entryArgs, argTypes, argAttrs, isVariadic, resultTypes,
391  resultAttrs, resultNames) ||
392  // If function attributes are present, parse them.
393  parser.parseOptionalAttrDictWithKeyword(result.attributes))
394  return failure();
395 
396  // Record the argument and result types as an attribute. This is necessary
397  // for external modules.
398  auto type = builder.getFunctionType(argTypes, resultTypes);
399  result.addAttribute(getTypeAttrName(), TypeAttr::get(type));
400 
401  auto *context = result.getContext();
402 
403  if (hasAttribute("resultNames", result.attributes) ||
404  hasAttribute("parameters", result.attributes)) {
405  parser.emitError(
406  loc, "explicit `resultNames` / `parameters` attributes not allowed");
407  return failure();
408  }
409 
410  // Use the argument and result names if not already specified.
411  SmallVector<Attribute> argNames;
412  if (!entryArgs.empty()) {
413  for (auto &arg : entryArgs)
414  argNames.push_back(
415  hw::module_like_impl::getPortNameAttr(context, arg.name));
416  } else if (!argTypes.empty()) {
417  // The parser returns empty names in a special way.
418  argNames.assign(argTypes.size(), StringAttr::get(context, ""));
419  }
420 
421  // An explicit `argNames` attribute overrides the MLIR names. This is how
422  // we represent port names that aren't valid MLIR identifiers. Result and
423  // parameter names are printed quoted when they aren't valid identifiers, so
424  // they don't need this affordance.
425  if (!hasAttribute("argNames", result.attributes))
426  result.addAttribute("argNames", ArrayAttr::get(context, argNames));
427  result.addAttribute("resultNames", ArrayAttr::get(context, resultNames));
428  result.addAttribute("parameters", ArrayAttr::get(context, parameters));
429 
430  assert(argAttrs.size() == argTypes.size());
431  assert(resultAttrs.size() == resultTypes.size());
432 
433  // Add the attributes to the function arguments.
434  addArgAndResultAttrs(builder, result, argAttrs, resultAttrs);
435 
436  // Extern modules carry an empty region to work with HWModuleImplementation.h.
437  result.addRegion();
438 
439  return success();
440 }

471 static LogicalResult verifyMSFTModuleExternOp(MSFTModuleExternOp module) {
472  using namespace mlir::function_like_impl;
473  auto typeAttr = module->getAttrOfType<TypeAttr>(getTypeAttrName());
474  auto moduleType = typeAttr.getValue().cast<FunctionType>();
475  auto argNames = module->getAttrOfType<ArrayAttr>("argNames");
476  auto resultNames = module->getAttrOfType<ArrayAttr>("resultNames");
477  if (argNames.size() != moduleType.getNumInputs())
478  return module->emitOpError("incorrect number of argument names");
479  if (resultNames.size() != moduleType.getNumResults())
480  return module->emitOpError("incorrect number of result names");
481 
482  SmallPtrSet<Attribute, 4> paramNames;
483 
484  // Check parameter default values are sensible.
485  for (auto param : module->getAttrOfType<ArrayAttr>("parameters")) {
486  auto paramAttr = param.cast<hw::ParamDeclAttr>();
487 
488  // Check that we don't have any redundant parameter names. These are
489  // resolved by string name: reuse of the same name would cause ambiguities.
490  if (!paramNames.insert(paramAttr.getName()).second)
491  return module->emitOpError("parameter ")
492  << paramAttr << " has the same name as a previous parameter";
493 
494  // Default values are allowed to be missing, check them if present.
495  auto value = paramAttr.getValue();
496  if (!value)
497  continue;
498 
499  if (value.getType() != paramAttr.getType().getValue())
500  return module->emitOpError("parameter ")
501  << paramAttr << " should have type "
502  << paramAttr.getType().getValue() << "; has type "
503  << value.getType();
504 
505  // Verify that this is a valid parameter value, disallowing parameter
506  // references. We could allow parameters to refer to each other in the
507  // future with lexical ordering if there is a need.
508  if (failed(checkParameterInContext(value, module, module,
509  /*disallowParamRefs=*/true)))
510  return failure();
511  }
512  return success();
513 }
