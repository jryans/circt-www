26 /// This is an edge in the InstanceGraph. This tracks a specific instantiation
27 /// of a module.
28 class InstanceRecord {
29 public:
30  InstanceRecord(InstanceOp instance, InstanceGraphNode *parent,
31  InstanceGraphNode *target)
32  : instance(instance), parent(parent), target(target) {}
33 
34  /// Get the InstanceOp that this is tracking.
35  InstanceOp getInstance() const { return instance; }
36 
37  /// Get the module where the InstanceOp lives.
38  InstanceGraphNode *getParent() const { return parent; }
39 
40  /// Get the module which the InstanceOp is instantiating.
41  InstanceGraphNode *getTarget() const { return target; }
42 
43 private:
44  friend class InstanceGraph;
45 
46  /// The InstanceOp that this is tracking.
47  InstanceOp instance;
48 
49  /// This is the module where the InstanceOp lives.
50  InstanceGraphNode *parent;
51 
52  /// This is the module which the InstanceOp is instantiating.
53  InstanceGraphNode *target;
54 };

56 /// This is a Node in the InstanceGraph. Each node represents a Module in a
57 /// Circuit. Both external modules and regular modules can be represented by
58 /// this class. It is possible to efficiently iterate all modules instantiated
59 /// by this module, as well as all instantiations of this module.
60 class InstanceGraphNode {
61  using EdgeVec = std::deque<InstanceRecord>;
62  using UseVec = std::vector<InstanceRecord *>;
63 
64  static InstanceRecord *unwrap(EdgeVec::value_type &value) { return &value; }
65  class InstanceIterator final
66  : public llvm::mapped_iterator<EdgeVec::iterator, decltype(&unwrap)> {
67  public:
68  /// Initializes the result type iterator to the specified result iterator.
69  InstanceIterator(EdgeVec::iterator it)
70  : llvm::mapped_iterator<EdgeVec::iterator, decltype(&unwrap)>(it,
71  &unwrap) {
72  }
73  };
74 
75 public:
76  InstanceGraphNode() : module(nullptr) {}
77 
78  /// Get the module that this node is tracking.
79  Operation *getModule() const { return module; }
80 
81  /// Iterate the instance records in this module.
82  using iterator = InstanceIterator;
83  iterator begin() { return iterator(moduleInstances.begin()); }
84  iterator end() { return iterator(moduleInstances.end()); }
85  llvm::iterator_range<iterator> instances() {
86  return llvm::make_range(begin(), end());
87  }
88 
89  /// Iterate the instance records which instantiate this module.
90  using use_iterator = UseVec::iterator;
91  use_iterator uses_begin() { return moduleUses.begin(); }
92  use_iterator uses_end() { return moduleUses.end(); }
93  llvm::iterator_range<use_iterator> uses() {
94  return llvm::make_range(uses_begin(), uses_end());
95  }
96 
97 private:
98  /// Record a new instance op in the body of this module. Returns a newly
99  /// allocated InstanceRecord which will be owned by this node.
100  InstanceRecord *recordInstance(InstanceOp instance,
101  InstanceGraphNode *target);
102 
103  /// Record that a module instantiates this module.
104  void recordUse(InstanceRecord *record);
105 
106  /// The module.
107  Operation *module;
108 
109  /// List of instance operations in this module. This member owns the
110  /// InstanceRecords, which may be pointed to by other InstanceGraohNode's use
111  /// lists.
112  EdgeVec moduleInstances;
113 
114  /// List of instances which instantiate this module.
115  UseVec moduleUses;
116 
117  // Provide access to the constructor.
118  friend class InstanceGraph;
119 };

121 /// This graph tracks modules and where they are instantiated. This is intended
122 /// to be used as a cached analysis on FIRRTL circuits. This class can be used
123 /// to walk the modules efficiently in a bottom-up or top-down order.
124 ///
125 /// To use this class, retrieve a cached copy from the analysis manager:
126 /// auto &instanceGraph = getAnalysis<InstanceGraph>(getOperation());
127 class InstanceGraph {
128 
129  /// Storage for InstanceGraphNodes.
130  using NodeVec = std::deque<InstanceGraphNode>;
131 
132  /// Iterator that unwraps a unique_ptr to return a regular pointer.
133  static InstanceGraphNode *unwrap(NodeVec::value_type &value) {
134  return &value;
135  }
136  struct NodeIterator final
137  : public llvm::mapped_iterator<NodeVec::iterator, decltype(&unwrap)> {
138  /// Initializes the result type iterator to the specified result iterator.
139  NodeIterator(NodeVec::iterator it)
140  : llvm::mapped_iterator<NodeVec::iterator, decltype(&unwrap)>(it,
141  &unwrap) {
142  }
143  };
144 
145 public:
146  /// Create a new module graph of a circuit. This must be called on a FIRRTL
147  /// CircuitOp.
148  explicit InstanceGraph(Operation *operation);
149 
150  /// Get the node corresponding to the top-level module of a circuit.
151  InstanceGraphNode *getTopLevelNode();
152 
153  /// Get the module corresponding to the top-lebel module of a circuit.
154  FModuleLike getTopLevelModule();
155 
156  /// Look up an InstanceGraphNode for a module. Operation must be an FModuleOp
157  /// or an FExtModuleOp.
158  InstanceGraphNode *lookup(Operation *op);
159 
160  /// Lookup an InstanceGraphNode for a module. Operation must be an FModuleOp
161  /// or an FExtModuleOp.
162  InstanceGraphNode *operator[](Operation *op) { return lookup(op); }
163 
164  /// Look up the referenced module from an InstanceOp. This will use a
165  /// hashtable lookup to find the module, where
166  /// InstanceOp.getReferencedModule() will be a linear search through the IR.
167  Operation *getReferencedModule(InstanceOp op);
168 
169  /// Iterate through all modules.
170  using iterator = NodeIterator;
171  iterator begin() { return nodes.begin(); }
172  iterator end() { return nodes.end(); }
173 
174  //===-------------------------------------------------------------------------
175  // Methods to keep an InstanceGraph up to date.
176  //
177  // These methods are not thread safe. Make sure that modifications are
178  // properly synchronized or performed in a serial context. When the
179  // InstanceGraph is used as an analysis, this is only safe when the pass is
180  // on a CircuitOp.
181 
182  // Replaces an instance of a module with another instance. The target module
183  // of both InstanceOps must be the same.
184  void replaceInstance(InstanceOp inst, InstanceOp newInst);
185 
186 private:
187  /// Get the node corresponding to the module. If the node has does not exist
188  /// yet, it will be created.
189  InstanceGraphNode *getOrAddNode(StringRef name);
190 
191  /// Lookup an module by name.
192  InstanceGraphNode *lookup(StringRef name);
193 
194  /// The storage for graph nodes, with deterministic iteration.
195  NodeVec nodes;
196 
197  /// This maps each operation to its graph node.
198  llvm::StringMap<unsigned> nodeMap;
199 };

201 /// An absolute instance path.
202 using InstancePath = ArrayRef<InstanceOp>;
203 
204 template <typename T>
205 inline static T &formatInstancePath(T &into, const InstancePath &path) {
206  into << "$root";
207  for (auto inst : path)
208  into << "/" << inst.name() << ":" << inst.moduleName();
209  return into;
210 }
211 
212 template <typename T>
213 static T &operator<<(T &os, const InstancePath &path) {
214  return formatInstancePath(os, path);
215 }
