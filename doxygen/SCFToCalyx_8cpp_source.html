512 /// Creates register assignment operations within the provided groupOp.
513 static void buildAssignmentsForRegisterWrite(ComponentLoweringState &state,
514  PatternRewriter &rewriter,
515  calyx::GroupOp groupOp,
516  calyx::RegisterOp &reg,
517  Value inputValue) {
518  IRRewriter::InsertionGuard guard(rewriter);
519  auto loc = inputValue.getLoc();
520  rewriter.setInsertionPointToEnd(groupOp.getBody());
521  rewriter.create<calyx::AssignOp>(loc, reg.in(), inputValue);
522  rewriter.create<calyx::AssignOp>(loc, reg.write_en(),
523  state.getConstant(rewriter, loc, 1, 1));

527 static calyx::GroupOp buildWhileIterArgAssignments(
528  PatternRewriter &rewriter, ComponentLoweringState &state, Location loc,
529  scf::WhileOp whileOp, Twine uniqueSuffix, ValueRange ops) {
530  assert(whileOp);
531  /// Pass iteration arguments through registers. This follows closely
532  /// to what is done for branch ops.
533  auto groupName = "assign_" + uniqueSuffix;
534  auto groupOp = createGroup<calyx::GroupOp>(rewriter, state.getComponentOp(),
535  loc, groupName);
536  // Create register assignment for each iter_arg. a calyx::GroupDone signal
537  // is created for each register. This is later cleaned up in
538  // GroupDoneCleanupPattern.
539  for (auto arg : enumerate(ops)) {
540  auto reg = state.getWhileIterReg(whileOp, arg.index());
541  buildAssignmentsForRegisterWrite(state, rewriter, groupOp, reg,
542  arg.value());
543  }
544  return groupOp;
545 }

722  /// buildLibraryOp will build a TCalyxLibOp inside a TGroupOp based on the
723  /// source operation TSrcOp.
724  template <typename TGroupOp, typename TCalyxLibOp, typename TSrcOp>
725  LogicalResult buildLibraryOp(PatternRewriter &rewriter, TSrcOp op,
726  TypeRange srcTypes, TypeRange dstTypes) const {
727  SmallVector<Type> types;
728  llvm::append_range(types, srcTypes);
729  llvm::append_range(types, dstTypes);
730 
731  auto calyxOp = getComponentState().getNewLibraryOpInstance<TCalyxLibOp>(
732  rewriter, op.getLoc(), types);
733 
734  auto directions = calyxOp.portDirections();
735  SmallVector<Value, 4> opInputPorts;
736  SmallVector<Value, 4> opOutputPorts;
737  for (auto dir : enumerate(directions)) {
738  if (dir.value() == calyx::Direction::Input)
739  opInputPorts.push_back(calyxOp.getResult(dir.index()));
740  else
741  opOutputPorts.push_back(calyxOp.getResult(dir.index()));
742  }
743  assert(
744  opInputPorts.size() == op->getNumOperands() &&
745  opOutputPorts.size() == op->getNumResults() &&
746  "Expected an equal number of in/out ports in the Calyx library op with "
747  "respect to the number of operands/results of the source operation.");
748 
749  /// Create assignments to the inputs of the library op.
750  auto group = createGroupForOp<TGroupOp>(rewriter, op);
751  rewriter.setInsertionPointToEnd(group.getBody());
752  for (auto dstOp : enumerate(opInputPorts))
753  rewriter.create<calyx::AssignOp>(op.getLoc(), dstOp.value(),
754  op->getOperand(dstOp.index()));
755 
756  /// Replace the result values of the source operator with the new operator.
757  for (auto res : enumerate(opOutputPorts)) {
758  getComponentState().registerEvaluatingGroup(res.value(), group);
759  op->getResult(res.index()).replaceAllUsesWith(res.value());
760  }
761  return success();
762  }

768  return buildLibraryOp<TGroupOp, TCalyxLibOp, TSrcOp>(
769  rewriter, op, op.getOperandTypes(), op->getResultTypes());

800 LogicalResult BuildOpGroups::buildOp(PatternRewriter &rewriter,
801  memref::LoadOp loadOp) const {
802  Value memref = loadOp.memref();
803  auto memoryInterface = getComponentState().getMemoryInterface(memref);
804  if (noStoresToMemory(memref) && singleLoadFromMemory(memref)) {
805  // Single load from memory; we do not need to write the
806  // output to a register. This is essentially a "combinational read" under
807  // current Calyx semantics with memory, and thus can be done in a
808  // combinational group. Note that if any stores are done to this memory,
809  // we require that the load and store be in separate non-combinational
810  // groups to avoid reading and writing to the same memory in the same group.
811  auto combGroup = createGroupForOp<calyx::CombGroupOp>(rewriter, loadOp);
812  assignAddressPorts(rewriter, loadOp.getLoc(), combGroup, memoryInterface,
813  loadOp.getIndices());
814 
815  // We refrain from replacing the loadOp result with
816  // memoryInterface.readData, since multiple loadOp's need to be converted
817  // to a single memory's ReadData. If this replacement is done now, we lose
818  // the link between which SSA memref::LoadOp values map to which groups for
819  // loading a value from the Calyx memory. At this point of lowering, we
820  // keep the memref::LoadOp SSA value, and do value replacement _after_
821  // control has been generated (see LateSSAReplacement). This is *vital* for
822  // things such as InlineCombGroups to be able to properly track which
823  // memory assignment groups belong to which accesses.
824  getComponentState().registerEvaluatingGroup(loadOp.getResult(), combGroup);
825  } else {
826  auto group = createGroupForOp<calyx::GroupOp>(rewriter, loadOp);
827  assignAddressPorts(rewriter, loadOp.getLoc(), group, memoryInterface,
828  loadOp.getIndices());
829 
830  // Multiple loads from the same memory; In this case, we _may_ have a
831  // structural hazard in the design we generate. To get around this, we
832  // conservatively place a register in front of each load operation, and
833  // replace all uses of the loaded value with the register output. Proper
834  // handling of this requires the combinational group inliner/scheduler to
835  // be aware of when a combinational expression references multiple loaded
836  // values from the same memory, and then schedule assignments to temporary
837  // registers to get around the structural hazard.
838  auto reg = createReg(getComponentState(), rewriter, loadOp.getLoc(),
839  getComponentState().getUniqueName("load"),
840  loadOp.getMemRefType().getElementTypeBitWidth());
841  buildAssignmentsForRegisterWrite(getComponentState(), rewriter, group, reg,
842  memoryInterface.readData());
843  loadOp.getResult().replaceAllUsesWith(reg.out());
844  getComponentState().addBlockScheduleable(loadOp->getBlock(), group);
845  }
846  return success();
847 }

870 LogicalResult BuildOpGroups::buildOp(PatternRewriter &rewriter,
871  MulIOp mul) const {
872  Location loc = mul.getLoc();
873  Type width = mul.result().getType(), one = rewriter.getI1Type();
874  auto multPipe =
875  getComponentState().getNewLibraryOpInstance<calyx::MultPipeLibOp>(
876  rewriter, loc, {width, width, one, one, one, width, one});
877  // Pass the result from the MulIOp to the Calyx primitive.
878  mul.result().replaceAllUsesWith(multPipe.out());
879  auto reg = createReg(getComponentState(), rewriter, mul.getLoc(),
880  getComponentState().getUniqueName("mult_reg"),
881  width.getIntOrFloatBitWidth());
882  // Multiplication pipelines are not combinational, so a GroupOp is required.
883  auto group = createGroupForOp<calyx::GroupOp>(rewriter, mul);
884  getComponentState().addBlockScheduleable(mul->getBlock(), group);
885 
886  rewriter.setInsertionPointToEnd(group.getBody());
887  rewriter.create<calyx::AssignOp>(loc, multPipe.left(), mul.lhs());
888  rewriter.create<calyx::AssignOp>(loc, multPipe.right(), mul.rhs());
889  // Write the output to this register.
890  rewriter.create<calyx::AssignOp>(loc, reg.in(), multPipe.out());
891  // The write enable port is high when the pipeline is done.
892  rewriter.create<calyx::AssignOp>(loc, reg.write_en(), multPipe.done());
893  rewriter.create<calyx::AssignOp>(
894  loc, multPipe.go(), getComponentState().getConstant(rewriter, loc, 1, 1));
895  // The group is done when the register write is complete.
896  rewriter.create<calyx::GroupDoneOp>(loc, reg.done());
897 
898  // Register the values for the pipeline.
899  getComponentState().registerEvaluatingGroup(multPipe.out(), group);
900  getComponentState().registerEvaluatingGroup(multPipe.left(), group);
901  getComponentState().registerEvaluatingGroup(multPipe.right(), group);
902 
903  return success();
904 }

954 LogicalResult BuildOpGroups::buildOp(PatternRewriter &rewriter,
955  BranchOpInterface brOp) const {
956  /// Branch argument passing group creation
957  /// Branch operands are passed through registers. In BuildBBRegs we
958  /// created registers for all branch arguments of each block. We now
959  /// create groups for assigning values to these registers.
960  Block *srcBlock = brOp->getBlock();
961  for (auto succBlock : enumerate(brOp->getSuccessors())) {
962  auto succOperands = brOp.getSuccessorOperands(succBlock.index());
963  if (!succOperands.hasValue() || succOperands.getValue().size() == 0)
964  continue;
965  // Create operand passing group
966  std::string groupName = progState().blockName(srcBlock) + "_to_" +
967  progState().blockName(succBlock.value());
968  auto groupOp = createGroup<calyx::GroupOp>(rewriter, *getComponent(),
969  brOp.getLoc(), groupName);
970  // Fetch block argument registers associated with the basic block
971  auto dstBlockArgRegs =
972  getComponentState().getBlockArgRegs(succBlock.value());
973  // Create register assignment for each block argument
974  for (auto arg : enumerate(succOperands.getValue())) {
975  auto reg = dstBlockArgRegs[arg.index()];
976  buildAssignmentsForRegisterWrite(getComponentState(), rewriter, groupOp,
977  reg, arg.value());
978  }
979  /// Register the group as a block argument group, to be executed
980  /// when entering the successor block from this block (srcBlock).
981  getComponentState().addBlockArgGroup(srcBlock, succBlock.value(), groupOp);
982  }
983  return success();
984 }

1122  LogicalResult partiallyLower(calyx::AssignOp assignOp,
1123  PatternRewriter &rewriter) const override {
1124  auto &state =
1125  pls.compLoweringState(assignOp->getParentOfType<calyx::ComponentOp>());
1126 
1127  auto dest = assignOp.dest();
1128  if (!state.isInputPortOfMemory(dest).hasValue())
1129  return success();
1130

1132  unsigned srcBits = src.getType().getIntOrFloatBitWidth();

1134  if (srcBits == dstBits)
1135  return success();

1138  rewriter.getIntegerType(dstBits)};
1139  auto sliceOp = state.getNewLibraryOpInstance<calyx::SliceLibOp>(
1140  rewriter, assignOp.getLoc(), types);

1186 /// Inlines Calyx ExecuteRegionOp operations within their parent blocks.
1187 /// An execution region op (ERO) is inlined by:
1188 /// i : add a sink basic block for all yield operations inside the
1189 /// ERO to jump to
1190 /// ii : Rewrite scf.yield calls inside the ERO to branch to the sink block
1191 /// iii: inline the ERO region
1192 /// TODO(#1850) evaluate the usefulness of this lowering pattern.
1193 class InlineExecuteRegionOpPattern
1194  : public OpRewritePattern<scf::ExecuteRegionOp> {
1195  using OpRewritePattern::OpRewritePattern;
1196 
1197  LogicalResult matchAndRewrite(scf::ExecuteRegionOp execOp,
1198  PatternRewriter &rewriter) const override {
1199  /// Determine type of "yield" operations inside the ERO.
1200  TypeRange yieldTypes = execOp.getResultTypes();
1201 
1202  /// Create sink basic block and rewrite uses of yield results to sink block
1203  /// arguments.
1204  rewriter.setInsertionPointAfter(execOp);
1205  auto *sinkBlock = rewriter.splitBlock(
1206  execOp->getBlock(),
1207  execOp.getOperation()->getIterator()->getNextNode()->getIterator());
1208  sinkBlock->addArguments(yieldTypes);
1209  for (auto res : enumerate(execOp.getResults()))
1210  res.value().replaceAllUsesWith(sinkBlock->getArgument(res.index()));
1211 
1212  /// Rewrite yield calls as branches.
1213  for (auto yieldOp :
1214  make_early_inc_range(execOp.getRegion().getOps<scf::YieldOp>())) {
1215  rewriter.setInsertionPointAfter(yieldOp);
1216  rewriter.replaceOpWithNewOp<BranchOp>(yieldOp, sinkBlock,
1217  yieldOp.getOperands());
1218  }
1219 
1220  /// Inline the regionOp.
1221  auto *preBlock = execOp->getBlock();
1222  auto *execOpEntryBlock = &execOp.getRegion().front();
1223  auto *postBlock = execOp->getBlock()->splitBlock(execOp);
1224  rewriter.inlineRegionBefore(execOp.getRegion(), postBlock);
1225  rewriter.mergeBlocks(postBlock, preBlock);
1226  rewriter.eraseOp(execOp);
1227 
1228  /// Finally, erase the unused entry block of the execOp region.
1229  rewriter.mergeBlocks(execOpEntryBlock, preBlock);
1230 
1231  return success();
1232  }

1307  outPorts.size()};

1358  extMemPorts.done = compOp.getArgument(inPortsIt);
1359  extMemPorts.writeData = compOp.getArgument(outPortsIt++);

1362  .cast<MemRefType>()
1363  .getShape()

1367  extMemPorts.writeEn = compOp.getArgument(outPortsIt);
1368 
1369  /// Register the external memory ports as a memory interface within the
1370  /// component.
1371  compState.registerMemoryInterface(extMemPortIndices.getFirst(),
1372  CalyxMemoryInterface(extMemPorts));
1373  }
1374

1376  }

1622  /// before recursing into the scheduling of the block innards.

1627  const DenseSet<Block *> &path, Location loc,
1628  Block *from, Block *to,
1629  Block *parentCtrlBlock) const {
1630  /// Schedule any registered block arguments to be executed before the body
1631  /// of the branch.
1632  rewriter.setInsertionPointToEnd(parentCtrlBlock);

1641  LogicalResult buildCFGControl(DenseSet<Block *> path,
1642  PatternRewriter &rewriter,
1643  mlir::Block *parentCtrlBlock,
1644  mlir::Block *preBlock,
1645  mlir::Block *block) const {
1646  if (path.count(block) != 0)
1647  return preBlock->getTerminator()->emitError()
1648  << "CFG backedge detected. Loops must be raised to 'scf.while' or "
1649  "'scf.for' operations.";
1650 
1651  rewriter.setInsertionPointToEnd(parentCtrlBlock);
1652  LogicalResult bbSchedResult =
1653  scheduleBasicBlock(rewriter, path, parentCtrlBlock, block);
1654  if (bbSchedResult.failed())
1655  return bbSchedResult;
1656 
1657  path.insert(block);
1658  auto successors = block->getSuccessors();
1659  auto nSuccessors = successors.size();
1660  if (nSuccessors > 0) {
1661  auto brOp = dyn_cast<BranchOpInterface>(block->getTerminator());
1662  assert(brOp);
1663  if (nSuccessors > 1) {
1664  /// TODO(mortbopet): we could choose to support ie. std.switch, but it
1665  /// would probably be easier to just require it to be lowered
1666  /// beforehand.
1667  assert(nSuccessors == 2 &&
1668  "only conditional branches supported for now...");
1669  /// Wrap each branch inside an if/else.
1670  auto cond = brOp->getOperand(0);
1671  auto condGroup =
1672  getComponentState().getEvaluatingGroup<calyx::CombGroupOp>(cond);
1673  auto symbolAttr = FlatSymbolRefAttr::get(
1674  StringAttr::get(getContext(), condGroup.sym_name()));
1675 
1676  auto ifOp = rewriter.create<calyx::IfOp>(
1677  brOp->getLoc(), cond, symbolAttr, /*initializeElseBody=*/true);
1678  rewriter.setInsertionPointToStart(ifOp.getThenBody());
1679  auto thenSeqOp = rewriter.create<calyx::SeqOp>(brOp.getLoc());
1680  rewriter.setInsertionPointToStart(ifOp.getElseBody());
1681  auto elseSeqOp = rewriter.create<calyx::SeqOp>(brOp.getLoc());
1682 
1683  bool trueBrSchedSuccess =
1684  schedulePath(rewriter, path, brOp.getLoc(), block, successors[0],
1685  thenSeqOp.getBody())
1686  .succeeded();
1687  bool falseBrSchedSuccess = true;
1688  if (trueBrSchedSuccess) {
1689  falseBrSchedSuccess =
1690  schedulePath(rewriter, path, brOp.getLoc(), block, successors[1],
1691  elseSeqOp.getBody())
1692  .succeeded();
1693  }
1694 
1695  return success(trueBrSchedSuccess && falseBrSchedSuccess);
