55 /// A data structure to contain both the locations of the primitives on the
56 /// device and instance assignments to said primitives locations, aka
57 /// placements.
58 ///
59 /// Holds pointers into the IR, which may become invalid as a result of IR
60 /// transforms. As a result, this class should only be used for analysis and
61 /// then thrown away. It is permissible to persist it through transformations so
62 /// long as it is maintained along with the transformations.
63 class PlacementDB {
64 public:
65  /// Create a DB treating 'top' as the root module.
66  PlacementDB(Operation *top);
67  PlacementDB(Operation *top, const PrimitiveDB &seed);
68 
69  // TODO: Add calls to model the device primitive locations.
70 
71  /// In addition to an Operation which is the instance at the level being
72  /// modeled in MLIR, the instance path within the MLIR instance is often
73  /// necessary as most often the instance is an extern module.
74  struct PlacedInstance {
75  RootedInstancePathAttr path;
76  llvm::StringRef subpath;
77  Operation *op;
78  };
79 
80  /// Assign an instance to a primitive. Return false if another instance is
81  /// already placed at that location.
82  LogicalResult addPlacement(PhysLocationAttr, PlacedInstance);
83  /// Assign an operation to a physical region. Return false on failure.
84  LogicalResult addPlacement(PhysicalRegionRefAttr, PlacedInstance);
85  /// Using the operation attributes, add the proper placements to the database.
86  /// Return the number of placements which weren't added due to conflicts.
87  size_t addPlacements(FlatSymbolRefAttr rootMod, mlir::Operation *);
88  /// Walk the entire design adding placements root at the top module.
89  size_t addDesignPlacements();
90 
91  /// Lookup the instance at a particular location.
92  Optional<PlacedInstance> getInstanceAt(PhysLocationAttr);
93 
94  /// Find the nearest unoccupied primitive location to 'nearestToY' in
95  /// 'column'.
96  PhysLocationAttr getNearestFreeInColumn(PrimitiveType prim, uint64_t column,
97  uint64_t nearestToY);
98 
99  /// Walk the placement information in some sort of reasonable order. Bounds
100  /// restricts the walk to a rectangle of [xmin, xmax, ymin, ymax] (inclusive),
101  /// with -1 meaning unbounded.
102  void walkPlacements(function_ref<void(PhysLocationAttr, PlacedInstance)>,
103  std::tuple<int64_t, int64_t, int64_t, int64_t> bounds =
104  std::make_tuple(-1, -1, -1, -1),
105  Optional<PrimitiveType> primType = {});
106 
107  /// Walk the region placement information.
108  void walkRegionPlacements(
109  function_ref<void(PhysicalRegionRefAttr, PlacedInstance)>);
110 
111 private:
112  MLIRContext *ctxt;
113  Operation *top;
114 
115  using DimDevType = DenseMap<PrimitiveType, PlacedInstance>;
116  using DimNumMap = DenseMap<size_t, DimDevType>;
117  using DimYMap = DenseMap<size_t, DimNumMap>;
118  using DimXMap = DenseMap<size_t, DimYMap>;
119  using InstanceAndRegion = std::pair<PhysicalRegionRefAttr, PlacedInstance>;
120  using RegionPlacements = SmallVector<InstanceAndRegion>;
121 
122  /// Get the leaf node. Abstract this out to make it easier to change the
123  /// underlying data structure.
124  Optional<PlacedInstance *> getLeaf(PhysLocationAttr);
125 
126  DimXMap placements;
127  RegionPlacements regionPlacements;
128  bool seeded;
129 };
