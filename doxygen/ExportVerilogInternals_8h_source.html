30 /// It is built in a global pass over the entire design and then frozen to allow

35  /// Return the string to use for the specified port in the specified module.
36  /// Ports may be renamed for a variety of reasons (e.g. conflicting with
37  /// Verilog keywords), and this returns the legalized name to use.
38  StringRef getPortVerilogName(Operation *module,
39  const hw::PortInfo &port) const {
40  auto it = renamedPorts.find(std::make_pair(module, port.getId()));
41  return (it != renamedPorts.end() ? it->second : port.name).getValue();
42  }
43 
44  /// Return the string to use for the specified port in the specified module.
45  /// Ports may be renamed for a variety of reasons (e.g. conflicting with
46  /// Verilog keywords), and this returns the legalized name to use.
47  StringRef getPortVerilogName(Operation *module, ssize_t portArgNum) const {
48  auto numInputs = hw::getModuleNumInputs(module);
49  // portArgNum is the index into the result of getAllModulePortInfos.
50  // We need to translate it into the PortInfo::getId(), which is a unique
51  // port identifier.
52  // if input port, then (-1 - portArgNum)
53  // else numInputs - portArgNum
54  // Also ensure the correct index into the input/output list is computed.
55  StringAttr nameAttr;
56  ssize_t portId = portArgNum;
57  if (portArgNum < numInputs) {
58  nameAttr = module->getAttrOfType<ArrayAttr>("argNames")[portArgNum]
59  .cast<StringAttr>();
60  portId = -1 - portArgNum;
61  } else {
62  portId = portArgNum - numInputs;
63  nameAttr = module->getAttrOfType<ArrayAttr>("resultNames")[portId]
64  .cast<StringAttr>();
65  }
66  auto it = renamedPorts.find(std::make_pair(module, portId));
67  return (it != renamedPorts.end() ? it->second : nameAttr).getValue();
68  }
69 
70  /// Return the string to use for the specified parameter name in the specified
71  /// module. Parameters may be renamed for a variety of reasons (e.g.
72  /// conflicting with ports or Verilog keywords), and this returns the
73  /// legalized name to use.
74  StringRef getParameterVerilogName(Operation *module,
75  StringAttr paramName) const {
76  auto it = renamedParams.find(std::make_pair(module, paramName));
77  return (it != renamedParams.end() ? it->second : paramName).getValue();
78  }
79 
80  /// Return the string to use for the specified declaration. The operation is
81  /// commonly a declaration within the module, like `WireOp` and friends, which
82  /// have a `name` attribute. Values may be renamed for a variety of reasons
83  /// (e.g. conflicting with ports or Verilog keywords), and this returns the
84  /// legalized name to use.
85  StringRef getDeclarationVerilogName(Operation *op) const {
86  auto it = renamedDecls.find(op);
87  auto attr = it != renamedDecls.end() ? it->second : getDeclarationName(op);
88  return attr ? attr.getValue() : "";
89  }
90 
91  /// Return the string to use for the specified interface. Interfaces and their
92  /// signals may be renamed for a variety of reasons (e.g. conflicting with
93  /// Verilog keywords), and this returns the legalized name to use.
94  StringRef getInterfaceVerilogName(Operation *op) const {
95  auto it = renamedInterfaceOp.find(op);
96  auto attr = it != renamedInterfaceOp.end() ? it->second
97  : SymbolTable::getSymbolName(op);
98  return attr.getValue();
99  }
100 
101 private:
102  friend class GlobalNameResolver;
103  GlobalNameTable() {}
104  GlobalNameTable(const GlobalNameTable &) = delete;
105  void operator=(const GlobalNameTable &) = delete;

112 
113  void addRenamedParam(Operation *module, StringAttr oldName,

117  }
118 
119  void addRenamedDeclaration(Operation *declOp, StringRef newName) {
120  renamedDecls[declOp] = StringAttr::get(declOp->getContext(), newName);
121  }
122 
123  void addRenamedInterfaceOp(Operation *interfaceOp, StringRef newName) {
124  renamedInterfaceOp[interfaceOp] =
125  StringAttr::get(interfaceOp->getContext(), newName);
126  }
127 
128  /// This contains entries for any ports that got renamed. The key is a
129  /// moduleop/portIdx tuple, the value is the name to use. The portIdx is the
130  /// index of the port in the list returned by getAllModulePortInfos.
131  /// Note, PortInfo::getId() returns ssize_t.
132  DenseMap<std::pair<Operation *, ssize_t>, StringAttr> renamedPorts;
133 
134  /// This contains entries for any parameters that got renamed. The key is a
135  /// moduleop/paramName tuple, the value is the name to use.
136  DenseMap<std::pair<Operation *, Attribute>, StringAttr> renamedParams;
137 
138  /// This contains entries for any declarations that got renamed.
139  DenseMap<Operation *, StringAttr> renamedDecls;
140 
141  /// This contains entries for interface operations (sv.interface,
142  /// sv.interface.signal, and sv.interface.modport) that need to be renamed
143  /// due to conflicts.
144  DenseMap<Operation *, StringAttr> renamedInterfaceOp;
145 };

202 /// This class wraps an operation or a fixed string that should be emitted.
203 class StringOrOpToEmit {
204 public:
205  explicit StringOrOpToEmit(Operation *op) : pointerData(op), length(~0ULL) {}
206 
207  explicit StringOrOpToEmit(StringRef string) {
208  pointerData = (Operation *)nullptr;
209  setString(string);
210  }
211 
212  ~StringOrOpToEmit() {
213  if (const void *ptr = pointerData.dyn_cast<const void *>())
214  free(const_cast<void *>(ptr));
215  }
216 
217  /// If the value is an Operation*, return it. Otherwise return null.
218  Operation *getOperation() const {
219  return pointerData.dyn_cast<Operation *>();
220  }
221 
222  /// If the value wraps a string, return it. Otherwise return null.
223  StringRef getStringData() const {
224  if (const void *ptr = pointerData.dyn_cast<const void *>())
225  return StringRef((const char *)ptr, length);
226  return StringRef();
227  }
228 
229  /// This method transforms the entry from an operation to a string value.
230  void setString(StringRef value) {
231  assert(pointerData.is<Operation *>() && "shouldn't already be a string");
232  length = value.size();
233  void *data = malloc(length);
234  memcpy(data, value.data(), length);
235  pointerData = (const void *)data;
236  }
237 
238  // These move just fine.
239  StringOrOpToEmit(StringOrOpToEmit &&rhs)
240  : pointerData(rhs.pointerData), length(rhs.length) {
241  rhs.pointerData = (Operation *)nullptr;
242  }
243 
244 private:
245  StringOrOpToEmit(const StringOrOpToEmit &) = delete;
246  void operator=(const StringOrOpToEmit &) = delete;
247  PointerUnion<Operation *, const void *> pointerData;
248  size_t length;
249 };

305 /// Return true for operations that must always be inlined into a containing
306 /// expression for correctness.
307 static inline bool isExpressionAlwaysInline(Operation *op) {
308  // We need to emit array indexes inline per verilog "lvalue" semantics.
309  if (isa<sv::ArrayIndexInOutOp>(op) || isa<sv::ReadInOutOp>(op))
310  return true;
311 
312  // An SV interface modport is a symbolic name that is always inlined.
313  if (isa<sv::GetModportOp>(op) || isa<sv::ReadInterfaceSignalOp>(op))
314  return true;
315 
316  // XMRs can't be spilled if they are on the lhs. Conservatively never spill
317  // them.
318  if (isa<sv::XMROp>(op))
319  return true;
320 
321  return false;
322 }
