34 /// Parses a type registered to this dialect
35 Type MooreDialect::parseType(DialectAsmParser &parser) const {
36  llvm::StringRef mnemonic;
37  auto loc = parser.getCurrentLocation();
38  if (parser.parseKeyword(&mnemonic))
39  return Type();
40  Type type;
41  auto parseResult = generatedTypeParser(parser, mnemonic, type);
42  if (parseResult.hasValue())
43  return type;
44  parser.emitError(loc, "Failed to parse type moore.") << mnemonic << "\n";
45  return Type();
46 }
