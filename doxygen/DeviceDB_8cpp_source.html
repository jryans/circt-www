79 /// Assign an instance to a primitive. Return false if another instance is
80 /// already placed at that location.
81 LogicalResult PlacementDB::addPlacement(PhysLocationAttr loc,
82  PlacedInstance inst) {
83 
84  Optional<PlacedInstance *> leaf = getLeaf(loc);
85  if (!leaf)
86  return inst.op->emitOpError("Could not apply placement. Invalid location: ")
87  << loc;
88  PlacedInstance *cell = *leaf;
89  if (cell->op != nullptr)
90  return inst.op->emitOpError("Could not apply placement ")
91  << loc << ". Position already occupied by " << cell->op << ".";
92  *cell = inst;
93  return success();
94 }

110 /// Using the operation attributes, add the proper placements to the database.
111 /// Return the number of placements which weren't added due to conflicts.
112 size_t PlacementDB::addPlacements(FlatSymbolRefAttr rootMod,
113  mlir::Operation *op) {
114  size_t numFailed = 0;
115  for (NamedAttribute attr : op->getAttrs()) {
116  StringRef attrName = attr.getName();
117  llvm::TypeSwitch<Attribute>(attr.getValue())
118 
119  // Handle switch instance.
120  .Case([&](SwitchInstanceAttr instSwitch) {
121  for (auto caseAttr : instSwitch.getCases()) {
122  RootedInstancePathAttr instPath = caseAttr.getInst();
123 
124  // Filter out all paths which aren't related to this DB.
125  if (instPath.getRootModule() != rootMod)
126  continue;
127 
128  // If we recognize the type, validate and add it.
129  if (auto loc = caseAttr.getAttr().dyn_cast<PhysLocationAttr>()) {
130  if (!attrName.startswith("loc:")) {
131  op->emitOpError(
132  "PhysLoc attributes must have names starting with 'loc'");
133  ++numFailed;
134  continue;
135  }
136  LogicalResult added = addPlacement(
137  loc, PlacedInstance{instPath, attrName.substr(4), op});
138  if (failed(added))
139  ++numFailed;
140  } else if (auto loc = caseAttr.getAttr()
141  .dyn_cast<PhysicalRegionRefAttr>()) {
142  LogicalResult added =
143  addPlacement(loc, PlacedInstance{instPath, StringRef(), op});
144  if (failed(added))
145  ++numFailed;
146  }
147  }
148  })
149 
150  // Physloc outside of a switch instance is not valid.
151  .Case([op, &numFailed](PhysLocationAttr) {
152  ++numFailed;
153  op->emitOpError("PhysLoc attribute must be inside an "
154  "instance switch attribute");
155  })
156 
157  // Ignore attributes we don't understand.
158  .Default([](Attribute) {});
159  }
160  return numFailed;
161 }

183 PhysLocationAttr PlacementDB::getNearestFreeInColumn(PrimitiveType prim,
184  uint64_t columnNum,
185  uint64_t nearestToY) {
186  // Simplest possible algorithm.
187  PhysLocationAttr nearest = {};
188  walkPlacements(
189  [&nearest, nearestToY](PhysLocationAttr loc, PlacedInstance inst) {
190  if (inst.op)
191  return;
192  if (!nearest) {
193  nearest = loc;
194  return;
195  }
196  int64_t curDist =
197  std::abs((int64_t)nearestToY - (int64_t)nearest.getY());
198  int64_t replDist = std::abs((int64_t)nearestToY - (int64_t)loc.getY());
199  if (replDist < curDist)
200  nearest = loc;
201  },
202  std::make_tuple(columnNum, columnNum, -1, -1), prim);
203  return nearest;
204 }

206 Optional<PlacementDB::PlacedInstance *>
207 PlacementDB::getLeaf(PhysLocationAttr loc) {
208  PrimitiveType primType = loc.getPrimitiveType().getValue();
209 
210  DimNumMap &nums = placements[loc.getX()][loc.getY()];
211  if (!seeded)
212  return &nums[loc.getNum()][primType];
213  if (!nums.count(loc.getNum()))
214  return {};
215 
216  DimDevType &primitives = nums[loc.getNum()];
217  if (primitives.count(primType) == 0)
218  return {};
219  return &primitives[primType];
220 }

222 /// Walker for placements.
223 void PlacementDB::walkPlacements(
224  function_ref<void(PhysLocationAttr, PlacedInstance)> callback,
225  std::tuple<int64_t, int64_t, int64_t, int64_t> bounds,
226  Optional<PrimitiveType> primType) {
227  uint64_t xmin = std::get<0>(bounds) < 0 ? 0 : std::get<0>(bounds);
228  uint64_t xmax = std::get<1>(bounds) < 0 ? std::numeric_limits<uint64_t>::max()
229  : (uint64_t)std::get<1>(bounds);
230  uint64_t ymin = std::get<2>(bounds) < 0 ? 0 : std::get<2>(bounds);
231  uint64_t ymax = std::get<3>(bounds) < 0 ? std::numeric_limits<uint64_t>::max()
232  : (uint64_t)std::get<3>(bounds);
233 
234  // TODO: Since the data structures we're using aren't sorted, the best we can
235  // do is iterate and filter. Once we get to performance, we'll figure out the
236  // right data structure.
237 
238  // X loop.
239  for (auto colF = placements.begin(), colE = placements.end(); colF != colE;
240  ++colF) {
241  size_t x = colF->getFirst();
242  if (x < xmin || x > xmax)
243  continue;
244  DimYMap yMap = colF->getSecond();
245 
246  // Y loop.
247  for (auto rowF = yMap.begin(), rowE = yMap.end(); rowF != rowE; ++rowF) {
248  size_t y = rowF->getFirst();
249  if (y < ymin || y > ymax)
250  continue;
251  DimNumMap numMap = rowF->getSecond();
252 
253  // Num loop.
254  for (auto numF = numMap.begin(), numE = numMap.end(); numF != numE;
255  ++numF) {
256  size_t num = numF->getFirst();
257  DimDevType devMap = numF->getSecond();
258 
259  // DevType loop.
260  for (auto devF = devMap.begin(), devE = devMap.end(); devF != devE;
261  ++devF) {
262  PrimitiveType devtype = devF->getFirst();
263  if (primType && devtype != *primType)
264  continue;
265  PlacedInstance inst = devF->getSecond();
266 
267  // Marshall and run the callback.
268  PhysLocationAttr loc = PhysLocationAttr::get(
269  ctxt, PrimitiveTypeAttr::get(ctxt, devtype), x, y, num);
270  callback(loc, inst);
271  }
272  }
273  }
274  }
275 }
