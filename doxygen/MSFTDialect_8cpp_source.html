29 // We implement the OpAsmDialectInterface so that MSFT dialect operations
30 // automatically interpret the name attribute on operations as their SSA name.

32  using OpAsmDialectInterface::OpAsmDialectInterface;

34  /// Get a special name to use when printing the entry block arguments of the
35  /// region contained by an operation in this dialect.
36  void getAsmBlockArgumentNames(Block *block,
37  OpAsmSetValueNameFn setNameFn) const override {
38  // Assign port names to the bbargs if this is a module.
39  auto modOp = dyn_cast<MSFTModuleOp>(block->getParentOp());
40  if (!modOp)
41  return;
42  ArrayAttr argNames = modOp.argNamesAttr();
43  for (size_t i = 0, e = block->getNumArguments(); i != e; ++i) {
44  auto name = argNames[i].cast<StringAttr>().getValue();
45  if (!name.empty())
46  setNameFn(block->getArgument(i), name);
47  }
48  }
49 };
50 } // end anonymous namespace

74 static msft::InstanceOp getInstance(MSFTModuleOp root,
75  ArrayRef<StringRef> path) {
76  assert(!path.empty());
77 
78  // Unfortunately, instance names are not symbols and modules are not symbol
79  // tables, so we have to do a full walk.
80  StringRef searchName = path[0];
81  msft::InstanceOp match = nullptr;
82  root.walk([&](msft::InstanceOp inst) {
83  if (inst.instanceName() != searchName)
84  return WalkResult::advance();
85  match = inst;
86  return WalkResult::interrupt();
87  });
88  if (!match)
89  return nullptr;
90  if (path.size() == 1)
91  return match;
92 
93  // We're not the leaf, so recurse.
94  auto subPath = path.slice(1);
95  Operation *submoduleOp = match.getReferencedModule();
96  auto submodule = dyn_cast<msft::MSFTModuleOp>(submoduleOp);
97  if (!submodule)
98  return nullptr;
99  return getInstance(submodule, subPath);
100 }
