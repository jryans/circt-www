28 static ParseResult parsePipelineWhileOp(OpAsmParser &parser,
29  OperationState &result) {
30  // Parse initiation interval.
31  IntegerAttr ii;
32  if (parser.parseKeyword("II") || parser.parseEqual() ||
33  parser.parseAttribute(ii))
34  return failure();
35  result.addAttribute("II", ii);
36 
37  // Parse iter_args assignment list.
38  SmallVector<OpAsmParser::OperandType> regionArgs, operands;
39  if (succeeded(parser.parseOptionalKeyword("iter_args"))) {
40  if (parser.parseAssignmentList(regionArgs, operands))
41  return failure();
42  }
43 
44  // Parse function type from iter_args to results.
45  FunctionType type;
46  if (parser.parseColon() || parser.parseType(type))
47  return failure();
48 
49  // Function result type is the pipeline result type.
50  result.addTypes(type.getResults());
51 
52  // Resolve iter_args operands.
53  for (auto [operand, type] : llvm::zip(operands, type.getInputs()))
54  if (parser.resolveOperand(operand, type, result.operands))
55  return failure();
56 
57  // Parse condition region.
58  Region *condition = result.addRegion();
59  parser.parseRegion(*condition, regionArgs, type.getInputs());
60 
61  // Parse stages region.
62  if (parser.parseKeyword("do"))
63  return failure();
64  Region *stages = result.addRegion();
65  parser.parseRegion(*stages, regionArgs, type.getInputs());
66 
67  return success();
68 }

70 static void printPipelineWhileOp(OpAsmPrinter &p, PipelineWhileOp op) {
71  // Print the initiation interval.
72  p << " II = " << ' ' << op.II();
73 
74  // Print iter_args assignment list.
75  p << " iter_args(";
76  llvm::interleaveComma(
77  llvm::zip(op.stages().getArguments(), op.iterArgs()), p,
78  [&](auto it) { p << std::get<0>(it) << " = " << std::get<1>(it); });
79  p << ") : ";
80 
81  // Print function type from iter_args to results.
82  auto type = FunctionType::get(op.getContext(), op.stages().getArgumentTypes(),
83  op.getResultTypes());
84  p.printType(type);
85 
86  // Print condition region.
87  p.printRegion(op.condition(), /*printEntryBlockArgs=*/false);
88  p << " do";
89 
90  // Print stages region.
91  p.printRegion(op.stages(), /*printEntryBlockArgs=*/false);
92 }

94 static LogicalResult verifyPipelineWhileOp(PipelineWhileOp op) {
95  // Verify the condition block is "combinational" based on an allowlist of
96  // Arithmetic ops.
97  Block &conditionBlock = op.condition().front();
98  Operation *nonCombinational;
99  WalkResult conditionWalk = conditionBlock.walk([&](Operation *op) {
100  if (isa<StaticLogicDialect>(op->getDialect()))
101  return WalkResult::advance();
102 
103  if (!isa<arith::AddIOp, arith::AndIOp, arith::BitcastOp, arith::CmpIOp,
104  arith::ConstantOp, arith::IndexCastOp, arith::MulIOp, arith::OrIOp,
105  SelectOp, arith::ShLIOp, arith::ExtSIOp, arith::CeilDivSIOp,
106  arith::DivSIOp, arith::FloorDivSIOp, arith::RemSIOp,
107  arith::ShRSIOp, arith::SubIOp, arith::TruncIOp, arith::DivUIOp,
108  arith::RemUIOp, arith::ShRUIOp, arith::XOrIOp, arith::ExtUIOp>(
109  op)) {
110  nonCombinational = op;
111  return WalkResult::interrupt();
112  }
113 
114  return WalkResult::advance();
115  });
116 
117  if (conditionWalk.wasInterrupted())
118  return op.emitOpError("condition must have a combinational body, found ")
119  << *nonCombinational;
120 
121  // Verify the condition block terminates with a value of type i1.
122  TypeRange conditionResults =
123  conditionBlock.getTerminator()->getOperandTypes();
124  if (conditionResults.size() != 1)
125  return op.emitOpError(
126  "condition must terminate with a single result, found ")
127  << conditionResults;
128 
129  if (conditionResults.front() != IntegerType::get(op.getContext(), 1))
130  return op.emitOpError("condition must terminate with an i1 result, found ")
131  << conditionResults.front();
132 
133  // Verify the stages block contains at least one stage and a terminator.
134  Block &stagesBlock = op.stages().front();
135  if (stagesBlock.getOperations().size() < 2)
136  return op.emitOpError("stages must contain at least one stage");
137 
138  // Verify the stages block contains only `staticlogic.pipeline.stage` and
139  // `staticlogic.pipeline.terminator` ops.
140  for (Operation &inner : stagesBlock)
141  if (!isa<PipelineStageOp, PipelineTerminatorOp>(inner))
142  return op.emitOpError(
143  "stages may only contain 'staticlogic.pipeline.stage' or "
144  "'staticlogic.pipeline.terminator' ops, found ")
145  << inner;
146 
147  return success();
148 }

150 void PipelineWhileOp::build(OpBuilder &builder, OperationState &state,
151  TypeRange resultTypes, IntegerAttr ii,
152  ValueRange iterArgs) {
153  OpBuilder::InsertionGuard g(builder);
154 
155  state.addTypes(resultTypes);
156  state.addAttribute("II", ii);
157  state.addOperands(iterArgs);
158 
159  Region *condRegion = state.addRegion();
160  Block &condBlock = condRegion->emplaceBlock();
161  condBlock.addArguments(iterArgs.getTypes());
162  builder.setInsertionPointToEnd(&condBlock);
163  builder.create<PipelineRegisterOp>(builder.getUnknownLoc(), ValueRange());
164 
165  Region *stagesRegion = state.addRegion();
166  Block &stagesBlock = stagesRegion->emplaceBlock();
167  stagesBlock.addArguments(iterArgs.getTypes());
168  builder.setInsertionPointToEnd(&stagesBlock);
169  builder.create<PipelineTerminatorOp>(builder.getUnknownLoc(), ValueRange(),
170  ValueRange());
171 }

193 static LogicalResult verifyPipelineRegisterOp(PipelineRegisterOp op) {
194  PipelineStageOp stage = op->getParentOfType<PipelineStageOp>();
195 
196  // If this doesn't terminate a stage, it is terminating the condition.
197  if (stage == nullptr)
198  return success();
199 
200  // Verify stage terminates with the same types as the result types.
201  TypeRange registerTypes = op.getOperandTypes();
202  TypeRange resultTypes = stage.getResultTypes();
203  if (registerTypes != resultTypes)
204  return op.emitOpError("operand types (")
205  << registerTypes << ") must match result types (" << resultTypes
206  << ")";
207 
208  return success();
209 }

215 static LogicalResult verifyPipelineTerminatorOp(PipelineTerminatorOp op) {
216  PipelineWhileOp pipeline = op->getParentOfType<PipelineWhileOp>();
217 
218  // Verify pipeline terminates with the same `iter_args` types as the pipeline.
219  auto iterArgs = op.iter_args();
220  TypeRange terminatorArgTypes = iterArgs.getTypes();
221  TypeRange pipelineArgTypes = pipeline.iterArgs().getTypes();
222  if (terminatorArgTypes != pipelineArgTypes)
223  return op.emitOpError("'iter_args' types (")
224  << terminatorArgTypes << ") must match pipeline 'iter_args' types ("
225  << pipelineArgTypes << ")";
226 
227  // Verify `iter_args` are defined by a pipeline stage.
228  for (auto iterArg : iterArgs)
229  if (iterArg.getDefiningOp<PipelineStageOp>() == nullptr)
230  return op.emitOpError(
231  "'iter_args' must be defined by a 'staticlogic.pipeline.stage'");
232 
233  // Verify pipeline terminates with the same result types as the pipeline.
234  auto results = op.results();
235  TypeRange terminatorResultTypes = results.getTypes();
236  TypeRange pipelineResultTypes = pipeline.getResultTypes();
237  if (terminatorResultTypes != pipelineResultTypes)
238  return op.emitOpError("'results' types (")
239  << terminatorResultTypes << ") must match pipeline result types ("
240  << pipelineResultTypes << ")";
241 
242  // Verify `results` are defined by a pipeline stage.
243  for (auto result : results)
244  if (result.getDefiningOp<PipelineStageOp>() == nullptr)
245  return op.emitOpError(
246  "'results' must be defined by a 'staticlogic.pipeline.stage'");
247 
248  return success();
249 }
