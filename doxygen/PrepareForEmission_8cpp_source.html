9 // This file implements the "prepare" pass that walks the IR before the emitter
10 // gets involved. This allows us to do some transformations that would be
11 // awkward to implement inline in the emitter.

13 // NOTE: This file covers the preparation phase of `ExportVerilog` which mainly
14 // legalizes the IR and makes adjustments necessary for emission. This is the
15 // place to mutate the IR if emission needs it. The IR cannot be modified during
16 // emission itself, which happens in parallel.

33 // Check if the value is from read of a wire or reg or is a port.
34 static bool isSimpleReadOrPort(Value v) {
35  if (v.isa<BlockArgument>())
36  return true;
37  auto vOp = v.getDefiningOp();
38  if (!vOp)
39  return false;
40  auto read = dyn_cast<ReadInOutOp>(vOp);
41  if (!read)
42  return false;
43  auto readSrc = read.input().getDefiningOp();
44  if (!readSrc)
45  return false;
46  return isa<WireOp, RegOp>(readSrc);
47 }

49 // Given an invisible instance, make sure all inputs are driven from
50 // wires or ports.
51 static void lowerBoundInstance(InstanceOp op) {
52  Block *block = op->getParentOfType<HWModuleOp>().getBodyBlock();
53  auto builder = ImplicitLocOpBuilder::atBlockBegin(op.getLoc(), block);
54 
55  SmallString<32> nameTmp;
56  nameTmp = (op.instanceName() + "_").str();
57  auto namePrefixSize = nameTmp.size();
58 
59  size_t nextOpNo = 0;
60  for (auto &port : getModulePortInfo(op).inputs) {
61  auto src = op.getOperand(nextOpNo);
62  ++nextOpNo;
63 
64  if (isSimpleReadOrPort(src))
65  continue;
66 
67  nameTmp.resize(namePrefixSize);
68  if (port.name)
69  nameTmp += port.name.getValue().str();
70  else
71  nameTmp += std::to_string(nextOpNo - 1);
72 
73  auto newWire = builder.create<WireOp>(src.getType(), nameTmp);
74  auto newWireRead = builder.create<ReadInOutOp>(newWire);
75  auto connect = builder.create<AssignOp>(newWire, src);
76  newWireRead->moveBefore(op);
77  connect->moveBefore(op);
78  op.setOperand(nextOpNo - 1, newWireRead);
79  }
80 }

90 // Ensure that each output of an instance are used only by a wire
91 static void lowerInstanceResults(InstanceOp op) {
92  Block *block = op->getParentOfType<HWModuleOp>().getBodyBlock();
93  auto builder = ImplicitLocOpBuilder::atBlockBegin(op.getLoc(), block);
94 
95  SmallString<32> nameTmp;
96  nameTmp = (op.instanceName() + "_").str();
97  auto namePrefixSize = nameTmp.size();
98 
99  size_t nextResultNo = 0;
100  for (auto &port : getModulePortInfo(op).outputs) {
101  auto result = op.getResult(nextResultNo);
102  ++nextResultNo;
103 
104  if (onlyUseIsAssign(result))
105  continue;
106 
107  bool isOneUseOutput = false;
108  if (result.hasOneUse()) {
109  OpOperand &use = *result.getUses().begin();
110  isOneUseOutput = dyn_cast_or_null<OutputOp>(use.getOwner()) != nullptr;
111  }
112 
113  if (!isOneUseOutput) {
114  nameTmp.resize(namePrefixSize);
115  if (port.name)
116  nameTmp += port.name.getValue().str();
117  else
118  nameTmp += std::to_string(nextResultNo - 1);
119 
120  auto newWire = builder.create<WireOp>(result.getType(), nameTmp);
121  while (!result.use_empty()) {
122  auto newWireRead = builder.create<ReadInOutOp>(newWire);
123  OpOperand &use = *result.getUses().begin();
124  use.set(newWireRead);
125  newWireRead->moveBefore(use.getOwner());
126  }
127 
128  auto connect = builder.create<AssignOp>(newWire, result);
129  connect->moveAfter(op);
130  }
131  }
132 }

134 // Given a side effect free "always inline" operation, make sure that it
135 // exists in the same block as its users and that it has one use for each one.
136 static void lowerAlwaysInlineOperation(Operation *op) {
137  assert(op->getNumResults() == 1 &&
138  "only support 'always inline' ops with one result");
139 
140  // Nuke use-less operations.
141  if (op->use_empty()) {
142  op->erase();
143  return;
144  }
145 
146  // Moving/cloning an op should pull along its operand tree with it if they
147  // are always inline. This happens when an array index has a constant
148  // operand for example.
149  auto recursivelyHandleOperands = [](Operation *op) {
150  for (auto operand : op->getOperands()) {
151  if (auto *operandOp = operand.getDefiningOp())
152  if (isExpressionAlwaysInline(operandOp))
153  lowerAlwaysInlineOperation(operandOp);
154  }
155  };
156 
157  // If this operation has multiple uses, duplicate it into N-1 of them in
158  // turn.
159  while (!op->hasOneUse()) {
160  OpOperand &use = *op->getUses().begin();
161  Operation *user = use.getOwner();
162 
163  // Clone the op before the user.
164  auto *newOp = op->clone();
165  user->getBlock()->getOperations().insert(Block::iterator(user), newOp);
166  // Change the user to use the new op.
167  use.set(newOp->getResult(0));
168 
169  // If any of the operations of the moved op are always inline, recursively
170  // handle them too.
171  recursivelyHandleOperands(newOp);
172  }
173 
174  // Finally, ensures the op is in the same block as its user so it can be
175  // inlined.
176  Operation *user = *op->getUsers().begin();
177  if (op->getBlock() != user->getBlock()) {
178  op->moveBefore(user);
179 
180  // If any of the operations of the moved op are always inline, recursively
181  // move/clone them too.
182  recursivelyHandleOperands(op);
183  }
184  return;
185 }

187 /// Lower a variadic fully-associative operation into an expression tree. This
188 /// enables long-line splitting to work with them.
189 static Value lowerFullyAssociativeOp(Operation &op, OperandRange operands,
190  SmallVector<Operation *> &newOps) {
191  Value lhs, rhs;
192  switch (operands.size()) {
193  case 0:
194  assert(0 && "cannot be called with empty operand range");
195  break;
196  case 1:
197  return operands[0];
198  case 2:
199  lhs = operands[0];
200  rhs = operands[1];
201  break;
202  default:
203  auto firstHalf = operands.size() / 2;
204  lhs = lowerFullyAssociativeOp(op, operands.take_front(firstHalf), newOps);
205  rhs = lowerFullyAssociativeOp(op, operands.drop_front(firstHalf), newOps);
206  break;
207  }
208 
209  OperationState state(op.getLoc(), op.getName());
210  state.addOperands(ValueRange{lhs, rhs});
211  state.addTypes(op.getResult(0).getType());
212  auto *newOp = Operation::create(state);
213  op.getBlock()->getOperations().insert(Block::iterator(&op), newOp);
214  newOps.push_back(newOp);
215  return newOp->getResult(0);
216 }

218 /// When we find that an operation is used before it is defined in a graph
219 /// region, we emit an explicit wire to resolve the issue.
220 static void lowerUsersToTemporaryWire(Operation &op) {
221  Block *block = op.getBlock();
222  auto builder = ImplicitLocOpBuilder::atBlockBegin(op.getLoc(), block);
223 
224  for (auto result : op.getResults()) {
225  auto newWire = builder.create<WireOp>(result.getType());
226 
227  while (!result.use_empty()) {
228  auto newWireRead = builder.create<ReadInOutOp>(newWire);
229  OpOperand &use = *result.getUses().begin();
230  use.set(newWireRead);
231  newWireRead->moveBefore(use.getOwner());
232  }
233 
234  auto connect = builder.create<AssignOp>(newWire, result);
235  connect->moveAfter(&op);
236  }
237 }

239 /// Transform "a + -cst" ==> "a - cst" for prettier output. This returns the
240 /// first operation emitted.
241 static Operation *rewriteAddWithNegativeConstant(comb::AddOp add,
242  hw::ConstantOp rhsCst) {
243  ImplicitLocOpBuilder builder(add.getLoc(), add);
244 
245  // Get the positive constant.
246  auto negCst = builder.create<hw::ConstantOp>(-rhsCst.getValue());
247  auto sub = builder.create<comb::SubOp>(add.getOperand(0), negCst);
248  add.getResult().replaceAllUsesWith(sub);
249  add.erase();
250  if (rhsCst.use_empty())
251  rhsCst.erase();
252  return negCst;
253 }

255 /// Given an operation in a procedural region, scan up the region tree to find
256 /// the first operation in a graph region (typically an always or initial op).
257 static Operation *findParentInNonProceduralRegion(Operation *op) {
258  Operation *parentOp = op->getParentOp();
259  assert(parentOp->hasTrait<ProceduralRegion>() &&
260  "we should only be hoisting from procedural");
261  while (parentOp->getParentOp()->hasTrait<ProceduralRegion>())
262  parentOp = parentOp->getParentOp();
263  return parentOp;
264 }

266 /// This function is invoked on side effecting Verilog expressions when we're in
267 /// 'disallowLocalVariables' mode for old Verilog clients. This ensures that
268 /// any side effecting expressions are only used by a single BPAssign to a
269 /// sv.reg operation. This ensures that the verilog emitter doesn't have to
270 /// worry about spilling them.
271 ///
272 /// This returns true if the op was rewritten, false otherwise.
273 static bool rewriteSideEffectingExpr(Operation *op) {
274  assert(op->getNumResults() == 1 && "isn't a verilog expression");
275 
276  // Check to see if this is already rewritten.
277  if (op->hasOneUse()) {
278  if (auto assign = dyn_cast<BPAssignOp>(*op->user_begin()))
279  if (assign.dest().getDefiningOp<RegOp>())
280  return false;
281  }
282 
283  // Otherwise, we have to transform it. Insert a reg at the top level, make
284  // everything using the side effecting expression read the reg, then assign to
285  // it after the side effecting operation.
286  Value opValue = op->getResult(0);
287 
288  // Scan to the top of the region tree to find out where to insert the reg.
289  Operation *parentOp = findParentInNonProceduralRegion(op);
290  OpBuilder builder(parentOp);
291  auto reg = builder.create<RegOp>(op->getLoc(), opValue.getType());
292 
293  // Everything using the expr now uses a read_inout of the reg.
294  auto value = builder.create<ReadInOutOp>(op->getLoc(), reg);
295  opValue.replaceAllUsesWith(value);
296 
297  // We assign the side effect expr to the reg immediately after that expression
298  // is computed.
299  builder.setInsertionPointAfter(op);
300  builder.create<BPAssignOp>(op->getLoc(), reg, opValue);
301  return true;
302 }

304 /// This function is called for non-side-effecting Verilog expressions when
305 /// we're in 'disallowLocalVariables' mode for old Verilog clients. It hoists
306 /// non-constant expressions out to the top level so they don't turn into local
307 /// variable declarations.
308 static bool hoistNonSideEffectExpr(Operation *op) {
309  // Never hoist "always inline" expressions - they will never generate a
310  // temporary and in fact must always be emitted inline.
311  if (isExpressionAlwaysInline(op) && !isa<sv::ReadInOutOp>(op))
312  return false;
313 
314  // Scan to the top of the region tree to find out where to move the op.
315  Operation *parentOp = findParentInNonProceduralRegion(op);
316 
317  // We can typically hoist all the way out to the top level in one step, but
318  // there may be intermediate operands that aren't hoistable. If so, just
319  // hoist one level.
320  bool cantHoist = false;
321  if (llvm::any_of(op->getOperands(), [&](Value operand) -> bool {
322  // The operand value dominates the original operation, but may be
323  // defined in one of the procedural regions between the operation and
324  // the top level of the module. We can tell this quite efficiently by
325  // looking for ops in a procedural region - because procedural regions
326  // live in graph regions but not visa-versa.
327  Operation *operandOp = operand.getDefiningOp();
328  if (!operandOp) // References to ports are always ok.
329  return false;
330 
331  if (operandOp->getParentOp()->hasTrait<ProceduralRegion>()) {
332  cantHoist |= operandOp->getBlock() == op->getBlock();
333  return true;
334  }
335  return false;
336  })) {
337 
338  // If the operand is in the same block as the expression then we can't hoist
339  // this out at all.
340  if (cantHoist)
341  return false;
342 
343  // Otherwise, we can hoist it, but not all the way out in one step. Just
344  // hoist one level out.
345  parentOp = op->getParentOp();
346  }
347 
348  op->moveBefore(parentOp);
349  return true;
350 }

352 /// For each module we emit, do a prepass over the structure, pre-lowering and
353 /// otherwise rewriting operations we don't want to emit.
354 void ExportVerilog::prepareHWModule(Block &block,
355  const LoweringOptions &options) {
356 
357  // First step, check any nested blocks that exist in this region. This walk
358  // can pull things out to our level of the hierarchy.
359  for (auto &op : block) {
360  // If the operations has regions, prepare each of the region bodies.
361  for (auto &region : op.getRegions()) {
362  if (!region.empty())
363  prepareHWModule(region.front(), options);
364  }
365  }
366 
367  // Next, walk all of the operations at this level.
368 
369  // True if these operations are in a procedural region.
370  bool isProceduralRegion = block.getParentOp()->hasTrait<ProceduralRegion>();
371  for (Block::iterator opIterator = block.begin(), e = block.end();
372  opIterator != e;) {
373  auto &op = *opIterator++;
374 
375  // Lower variadic fully-associative operations with more than two operands
376  // into balanced operand trees so we can split long lines across multiple
377  // statements.
378  // TODO: This is checking the Commutative property, which doesn't seem
379  // right in general. MLIR doesn't have a "fully associative" property.
380  if (op.getNumOperands() > 2 && op.getNumResults() == 1 &&
381  op.hasTrait<mlir::OpTrait::IsCommutative>() &&
382  mlir::MemoryEffectOpInterface::hasNoEffect(&op) &&
383  op.getNumRegions() == 0 && op.getNumSuccessors() == 0 &&
384  op.getAttrs().empty()) {
385  // Lower this operation to a balanced binary tree of the same operation.
386  SmallVector<Operation *> newOps;
387  auto result = lowerFullyAssociativeOp(op, op.getOperands(), newOps);
388  op.getResult(0).replaceAllUsesWith(result);
389  op.erase();
390 
391  // Make sure we revisit the newly inserted operations.
392  opIterator = Block::iterator(newOps.front());
393  continue;
394  }
395 
396  // Turn a + -cst ==> a - cst
397  if (auto addOp = dyn_cast<comb::AddOp>(op)) {
398  if (auto cst = addOp.getOperand(1).getDefiningOp<hw::ConstantOp>()) {
399  assert(addOp.getNumOperands() == 2 && "commutative lowering is done");
400  if (cst.getValue().isNegative()) {
401  Operation *firstOp = rewriteAddWithNegativeConstant(addOp, cst);
402  opIterator = Block::iterator(firstOp);
403  continue;
404  }
405  }
406  }
407 
408  // Name legalization should have happened in a different pass for these sv
409  // elements and we don't want to change their name through re-legalization
410  // (e.g. letting a temporary take the name of an unvisited wire). Adding
411  // them now ensures any temporary generated will not use one of the names
412  // previously declared.
413  if (auto instance = dyn_cast<InstanceOp>(op)) {
414  // Anchor return values to wires early
415  lowerInstanceResults(instance);
416  // Anchor ports of bound instances
417  if (instance->hasAttr("doNotPrint"))
418  lowerBoundInstance(instance);
419  }
420 
421  // Force any expression used in the event control of an always process to be
422  // a trivial wire, if the corresponding option is set.
423  if (!options.allowExprInEventControl) {
424  auto enforceWire = [&](Value expr) {
425  // Direct port uses are fine.
426  if (expr.isa<BlockArgument>())
427  return;
428  // If this is a read from a wire, we're fine.
429  if (auto read = expr.getDefiningOp<ReadInOutOp>())
430  if (read.input().getDefiningOp<WireOp>())
431  return;
432  auto builder = ImplicitLocOpBuilder::atBlockBegin(op.getLoc(), &block);
433  auto newWire = builder.create<WireOp>(expr.getType());
434  builder.setInsertionPoint(&op);
435  builder.create<AssignOp>(newWire, expr);
436  auto newWireRead = builder.create<ReadInOutOp>(newWire);
437  op.replaceUsesOfWith(expr, newWireRead);
438  };
439  if (auto always = dyn_cast<AlwaysOp>(op)) {
440  for (auto clock : always.clocks())
441  enforceWire(clock);
442  continue;
443  }
444  if (auto always = dyn_cast<AlwaysFFOp>(op)) {
445  enforceWire(always.clock());
446  if (auto reset = always.reset())
447  enforceWire(reset);
448  continue;
449  }
450  }
451 
452  // If the target doesn't support local variables, hoist all the expressions
453  // out to the nearest non-procedural region.
454  if (options.disallowLocalVariables && isVerilogExpression(&op) &&
455  isProceduralRegion) {
456 
457  // Force any side-effecting expressions in nested regions into a sv.reg
458  // if we aren't allowing local variable declarations. The Verilog emitter
459  // doesn't want to have to have to know how to synthesize a reg in the
460  // case they have to be spilled for whatever reason.
461  if (!mlir::MemoryEffectOpInterface::hasNoEffect(&op)) {
462  if (rewriteSideEffectingExpr(&op))
463  continue;
464  }
465 
466  // Hoist other expressions out to the parent region.
467  //
468  // NOTE: This effectively disables inlining of expressions into if
469  // conditions, $fwrite statements, and instance inputs. We could be
470  // smarter in ExportVerilog itself, but we'd have to teach it to put
471  // spilled expressions (due to line length, multiple-uses, and
472  // non-inlinable expressions) in the outer scope.
473  if (hoistNonSideEffectExpr(&op))
474  continue;
475  }
476 
477  // Duplicate "always inline" expression for each of their users and move
478  // them to be next to their users.
479  if (isExpressionAlwaysInline(&op)) {
480  lowerAlwaysInlineOperation(&op);
481  continue;
482  }
483  }
484 
485  // Now that all the basic ops are settled, check for any use-before def issues
486  // in graph regions. Lower these into explicit wires to keep the emitter
487  // simple.
488  if (!isProceduralRegion) {
489  SmallPtrSet<Operation *, 32> seenOperations;
490 
491  for (auto &op : llvm::make_early_inc_range(block)) {
492  // Check the users of any expressions to see if they are
493  // lexically below the operation itself. If so, it is being used out
494  // of order.
495  bool haveAnyOutOfOrderUses = false;
496  for (auto *userOp : op.getUsers()) {
497  // If the user is in a suboperation like an always block, then zip up
498  // to the operation that uses it.
499  while (&block != &userOp->getParentRegion()->front())
500  userOp = userOp->getParentOp();
501 
502  if (seenOperations.count(userOp)) {
503  haveAnyOutOfOrderUses = true;
504  break;
505  }
506  }
507 
508  // Remember that we've seen this operation.
509  seenOperations.insert(&op);
510 
511  // If all the uses of the operation are below this, then we're ok.
512  if (!haveAnyOutOfOrderUses)
513  continue;
514 
515  // If this is a reg/wire declaration, then we move it to the top of the
516  // block. We can't abstract the inout result.
517  if (op.getNumResults() == 1 &&
518  op.getResult(0).getType().isa<InOutType>() &&
519  op.getNumOperands() == 0) {
520  op.moveBefore(&block.front());
521  continue;
522  }
523 
524  // If this is a constant, then we move it to the top of the block.
525  if (isConstantExpression(&op)) {
526  op.moveBefore(&block.front());
527  continue;
528  }
529 
530  // Otherwise, we need to lower this to a wire to resolve this.
531  lowerUsersToTemporaryWire(op);
532  }
533  }
