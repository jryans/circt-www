66 template <typename BaseTy>
67 bool type_isa(Type type) {
68  // First check if the type is the requested type.
69  if (type.isa<BaseTy>())
70  return true;
71 
72  // Then check if it is a type alias wrapping the requested type.
73  if (auto alias = type.dyn_cast<TypeAliasType>())
74  return alias.getInnerType().isa<BaseTy>();
75 
76  return false;
77 }
78 
79 template <typename BaseTy>
80 BaseTy type_cast(Type type) {
81  assert(type_isa<BaseTy>(type) && "type must convert to requested type");
82 
83  // If the type is the requested type, return it.
84  if (type.isa<BaseTy>())
85  return type.cast<BaseTy>();
86 
87  // Otherwise, it must be a type alias wrapping the requested type.
88  return type.cast<TypeAliasType>().getInnerType().cast<BaseTy>();
89 }
90 
91 template <typename BaseTy>
92 BaseTy type_dyn_cast(Type type) {
93  if (!type_isa<BaseTy>(type))
94  return BaseTy();
95 
96  return type_cast<BaseTy>(type);
97 }
