31 /// Parse a time attribute.
32 /// Syntax: timeattr ::= #llhd.time<[time][timeUnit], [delta]d, [epsilon]e>
33 Attribute TimeAttr::parse(AsmParser &p, Type type) {
34  llvm::StringRef timeUnit;
35  unsigned time = 0;
36  unsigned delta = 0;
37  unsigned eps = 0;
38 
39  // parse the time value
40  if (p.parseLess() || p.parseInteger(time) || p.parseKeyword(&timeUnit))
41  return {};
42 
43  // parse the delta step value
44  if (p.parseComma() || p.parseInteger(delta) || p.parseKeyword("d"))
45  return {};
46 
47  // parse the epsilon value
48  if (p.parseComma() || p.parseInteger(eps) || p.parseKeyword("e") ||
49  p.parseGreater())
50  return {};
51 
52  // return a new instance of time attribute
53  auto loc = p.getEncodedSourceLoc(p.getCurrentLocation());
54  return getChecked(mlir::detail::getDefaultDiagnosticEmitFn(loc),
55  p.getContext(), time, timeUnit, delta, eps);
56 }

63 LogicalResult TimeAttr::verify(function_ref<InFlightDiagnostic()> emitError,
64  TimeType type, unsigned time,
65  llvm::StringRef timeUnit, unsigned delta,
66  unsigned epsilon) {
67  // Check the time unit is a legal SI unit
68  std::vector<std::string> legalUnits{"ys", "zs", "as", "fs", "ps",
69  "ns", "us", "ms", "s"};
70  if (std::find(legalUnits.begin(), legalUnits.end(), timeUnit) ==
71  legalUnits.end())
72  return emitError() << "Illegal time unit.";
73 
74  return success();
75 }

81 /// Parses a type registered to this dialect. Parse out the mnemonic then invoke
82 /// the tblgen'd type parser dispatcher.
83 Type LLHDDialect::parseType(DialectAsmParser &parser) const {
84  llvm::StringRef mnemonic;
85  if (parser.parseKeyword(&mnemonic))
86  return {};
87 
88  Type type;
89  if (generatedTypeParser(parser, mnemonic, type).hasValue())
90  return type;
91 
92  emitError(parser.getEncodedSourceLoc(parser.getCurrentLocation()),
93  "Invalid LLHD type!");
94  return {};
95 }
