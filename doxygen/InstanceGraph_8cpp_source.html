24 InstanceGraph::InstanceGraph(Operation *operation) {
25  auto circuitOp = cast<CircuitOp>(operation);
26 
27  // We insert the top level module first in to the node map. Getting the node
28  // here is enough to ensure that it is the first one added.
29  getOrAddNode(circuitOp.name());
30 
31  for (auto &op : *circuitOp.getBody()) {
32  if (auto extModule = dyn_cast<FExtModuleOp>(op)) {
33  auto *currentNode = getOrAddNode(extModule.getName());
34  currentNode->module = extModule;
35  }
36  if (auto module = dyn_cast<FModuleOp>(op)) {
37  auto *currentNode = getOrAddNode(module.getName());
38  currentNode->module = module;
39  // Find all instance operations in the module body.
40  module.body().walk([&](InstanceOp instanceOp) {
41  // Add an edge to indicate that this module instantiates the target.
42  auto *targetNode = getOrAddNode(instanceOp.moduleName());
43  auto *instanceRecord =
44  currentNode->recordInstance(instanceOp, targetNode);
45  targetNode->recordUse(instanceRecord);
46  });
47  }
48  }
49 }

78 InstanceGraphNode *InstanceGraph::getOrAddNode(StringRef name) {
79  // Try to insert an InstanceGraphNode. If its not inserted, it returns
80  // an iterator pointing to the node.
81  auto itAndInserted = nodeMap.try_emplace(name, 0);
82  auto &index = itAndInserted.first->second;
83  if (itAndInserted.second) {
84  // This is a new node, we have to add an element to the NodeVec.
85  nodes.emplace_back();
86  // Store the node storage index in to the map.
87  index = nodes.size() - 1;
88  return &nodes.back();
89  }
90  return &nodes[index];
91 }

97 void InstanceGraph::replaceInstance(InstanceOp inst, InstanceOp newInst) {
98  assert(inst.moduleName() == newInst.moduleName() &&
99  "Both instances must be targeting the same module");
100 
101  // Find the instance record of this instance.
102  auto *node = lookup(inst.moduleName());
103  auto it = llvm::find_if(node->uses(), [&](InstanceRecord *record) {
104  return record->getInstance() == inst;
105  });
106  assert(it != node->uses_end() && "Instance of module not recorded in graph");
107 
108  // We can just replace the instance op in the InstanceRecord without updating
109  // any instance lists.
110  (*it)->instance = newInst;
111 }

113 ArrayRef<InstancePath> InstancePathCache::getAbsolutePaths(Operation *op) {
114  assert((isa<FModuleOp, FExtModuleOp>(op))); // extra parens makes parser smile
115 
116  // If we have reached the circuit root, we're done.
117  if (op == instanceGraph.getTopLevelNode()->getModule()) {
118  static InstancePath empty{};
119  return empty; // array with single empty path
120  }
121 
122  // Fast path: hit the cache.
123  auto cached = absolutePathsCache.find(op);
124  if (cached != absolutePathsCache.end())
125  return cached->second;
126 
127  // For each instance, collect the instance paths to its parent and append the
128  // instance itself to each.
129  SmallVector<InstancePath, 8> extendedPaths;
130  for (auto inst : instanceGraph[op]->uses()) {
131  auto instPaths = getAbsolutePaths(inst->getParent()->getModule());
132  extendedPaths.reserve(instPaths.size());
133  for (auto path : instPaths) {
134  extendedPaths.push_back(appendInstance(path, inst->getInstance()));
135  }
136  }
137 
138  // Move the list of paths into the bump allocator for later quick retrieval.
139  ArrayRef<InstancePath> pathList;
140  if (!extendedPaths.empty()) {
141  auto paths = allocator.Allocate<InstancePath>(extendedPaths.size());
142  std::copy(extendedPaths.begin(), extendedPaths.end(), paths);
143  pathList = ArrayRef<InstancePath>(paths, extendedPaths.size());
144  }
145 
146  absolutePathsCache.insert({op, pathList});
147  return pathList;
148 }
